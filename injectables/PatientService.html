import { Injectable } from '@angular/core';
import { AngularFirestore } from '@angular/fire/compat/firestore';
import { getFirestore, collection, query, where, getDocs, doc } from '@angular/fire/firestore'
import { Patient } from '../type/patient.type';
import { getStorage, ref, uploadBytes } from "firebase/storage";
import { arrayUnion, deleteDoc } from 'firebase/firestore';

@Injectable({
  providedIn: 'root'
})
export class PatientService {

  constructor(private db: AngularFirestore) { }

  /**
   * Get patient data that matches document id and collection name from firebase.
   * 
   * @param id  Patient unique document id value
   * @param col Collection name
   * @returns   Observable<Action<DocumentSnapshot<unknown>>>
   */
  getPatient(id: string, col: string) {
    // col for specific collections (patients || studentPatients)
    return this.db
      .collection(col)
      .doc(id)
      .snapshotChanges();
  }

  /**
   * Create patient document based on the provided patient argument and image file.
   * Use callback function to do a task after patient is created.
   * 
   * @param patient Patient object that contains all the necessary data
   * @param img     Image of the patient (optional)
   * @param fn      callback function (optional)
   */
  createPatient(patient: Patient, img?: File, fn?: () => void) {
    // Convert Patient object to a JSON object
    let obj = patient.toFirebase();
    if(img) {
      patient.setProfilePicName(img.name);
      const storage = getStorage();
      const storageRef = ref(storage, patient.getProfilePicURL());

      uploadBytes(storageRef, img)
        .then((snapshot) => {
          console.log('Image uploaded successfully!');
        })
        .catch((error) => {
          console.error("Error uploading image:", error);
        });
    }

    // Use "set" method, since we separate the ID from the data; ID
    // is used as the document ID
    this.db.collection('patients')
      .doc(patient.getId())
      .set(obj)
      .then(fn)
      .catch(e => {
        console.log(e);
    });
  }

  /**
   *  Delete patient document that matches document id from patients collection
   * 
   * @param id  Patient unique document id value 
   */
  async deletePatient(id: string) {
    this.db.collection('patients')
      .doc(id)
      .delete()
      .catch((err) => {
        console.log("Error deleting patient:", err);
      })
  }

  /**
   * Create studentPatient document based on the provided patient argument and image file.
   * 
   * @param patient Patient object that contains all the necessary data
   */
  createStudentPatient(patient: Patient) {
    // Convert Patient object to a JSON object
    let obj = patient.toFirebase();

    // Use "add" method to create unique ID as 
    // patient ID cannot be used for the document ID
    this.db.collection('studentPatients').add(obj)
    .then(res => {
        console.log(res);
    })
    .catch(e => {
        console.log(e);
    })
  }

  /**
   * Update the specified document with the provided data (Document will be entirely updated)
   * 
   * @param id    Patient unique document id value
   * @param data  Data used to update the document
   * @param col   Collection name to perform update
   */
  async setPatient(id: string, data: any, col: string) {
    try {
      const res = await this.db
        .collection(col)
        .doc(id)
        .update(data);
    } catch (e) {
      console.log(e);
    }
  }

  /**
   * Update the specific field value of the firebase document. Only different value
   * will be added at the end of the field
   * 
   * @param id    Patient unique document id value
   * @param field Key name of the field that will be updated
   * @param data  Data used to update the document
   * @param col   Collection name to perform update
   */
  updateField(id: string, field: string, data: any, col: string) {
    const updateRef = this.db.collection(col).doc(id);
    updateRef.update({
      [field]: arrayUnion(data)
    })
  }

  /**
   *  Check both student id and patient id and returns querySnapshot of 
   * document that fits condition
   * 
   * @param {string} sId   string constraint value for thet "studentId" field
   * @param {string} pId   string constraint value for thet "id" field (patientId)
   * @param col Collection name 
   * @returns 
   */
  async patientDuplicateCheck(sId: string, pId:string, col: string) {
    const db = getFirestore();

    // query(collectionReference, condition)
    // Goes through studentpatients collection and find a patient assigned to a specific user
    const q = query(collection(db, col), where("studentId", "==", sId), where("id", "==", pId));

    const querySnapshot = await getDocs(q);

    return querySnapshot
  }

  /**
   * Check studentId with the provided id and return querySnapshot of document that fits the condition
   * 
   * @param {string} id   string constraint value for thet "studentId" field
   * @returns QuerySnapshot of DocumentData
   */
  async getStudentPatientbyId(id: string) {
    const db = getFirestore();

    // query(collectionReference, condition)
    // Goes through studentpatients collection and find a patient assigned to a specific user
    const q = query(collection(db, "studentPatients"), where("studentId", "==", id));

    const querySnapshot = await getDocs(q);

    return querySnapshot
  }

  /**
   * @param {string} studentList
   * @param {string} patientId 
   * @param {string} col
   *
   * : Delete patients/studentPatients document based on student id and patient id
   */
  deleteStudentPatient(studentId: string, patientuid: string, col: string) {
    const db = getFirestore();
    let counter = 0;
    let deleted = true;
    
    // Retrieve document(s) that matches provided arguments
    this.patientDuplicateCheck(studentId, patientuid, col).then(
      patientuid => {
        console.log("size: ", patientuid.size)
        counter++;
        if(!patientuid.empty){
          // Delete all matching document(s)
          patientuid.forEach(document => deleteDoc(doc(db, col, document.id)));
        }
        // When delete failed at any point
        else if(counter >= (patientuid.size - 1) && !deleted) {
          alert("Deletion failed for one or more student(s)")
        }
        else{
          deleted = false;
          console.log("deleteStudentPatient: ", patientuid, " does not exist");
        }
      }
    )
    console.log("deleteStudentPatient working!");
  }
}
