import { Component, Inject, OnInit } from '@angular/core';
import { FormBuilder, FormControl } from '@angular/forms';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { Order } from 'src/app/type/order.type';

export interface AdministerMedDialogData {
  patientName: string;
  patientId: string;
  med: Order; 
}

@Component({
  selector: 'app-administer-med-dialog',
  templateUrl: './administer-med-dialog.component.html',
  styleUrls: ['./administer-med-dialog.component.scss']
})
export class AdministerMedDialogComponent implements OnInit {

  // Variables to store data
  patientName: string = "";
  patientId: string = "";
  med!: Order;

  form = this.fb.group({
    medId: [''],
  })

  // Variables to handle error messages
  submitted: boolean = false;
  error: string = "";

  constructor(public dialogRef: MatDialogRef<AdministerMedDialogComponent>,
              @Inject(MAT_DIALOG_DATA) public data: AdministerMedDialogData,
              private fb: FormBuilder) { }

  ngOnInit(): void {
    // Set submitted to false whenever the form's value changes;
    // this gets rid of any error messages when the user starts typing
    this.form.valueChanges.subscribe(() => {
      this.submitted = false;
    })

    this.patientName = this.data.patientName;
    this.patientId = this.data.patientId;
    this.med = this.data.med;
  }

  /**
   *  close the dialog and sends medication data to the parent component
   */
  onCloseClick(): void {
    // Pass back 'false' to MAR tab to indicate cancelling of administration
    this.dialogRef.close(this.med);
  }

  /**
   *  Submit the form and checks to see correct medication is administered
   */
  onSubmit(): void {
    if(this.correctMedId) {
      this.med.quantityGiven += 1;
    }
    else {
      this.submitted = true;
      this.error = this.getError();
    }
  }

  /**
   *  Returns various error messages to be displayed on client side.
   * The error messages are checked below in descending precedence.
   */
   getError(): string {
    if(!this.correctMedId) {
      return "This is the incorrect medication to administer!"
    }
    return "Something went wrong - try again."
  }

  get medIdFormControl(): FormControl {
    return this.form.controls.medId as FormControl;
  }

  get correctMedId(): boolean {
    return (this.medIdFormControl.value as string).trim() === this.data.med.getMedId();
  }

  /**
   *  Returns whether the administered quantity is greater than the 
   * provider order quantity.
   */
  get isOverdosed(): boolean {
    return this.med.quantityGiven > this.med.quantity
  }

}
