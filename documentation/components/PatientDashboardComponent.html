export class PatientDashboardComponent extends PageComponent implements OnInit {

  // Stores current patient information
  currentPatient = new Patient();
  patientId = "";
  collection = "";
  profilePicSrc = "";

  // Styles for the snapshot (general info) container
  isSnapshotOpen = true;

  // Needed to realign tab ink bar when patient snapshot opens and closes
  @ViewChild('tabs', {static: false}) tabs!: MatTabGroup;

  // Needed to repaint the MAR scheduler so that it loads correctly on the first open
  // and reloads each time thereafter
  @ViewChild(MarTabComponent, {static: false}) mar!: MarTabComponent;

  constructor(private router: Router,
              private route: ActivatedRoute,
              private patientService: PatientService,
              userService: UserService) { 
    super(userService);
  }

  ngOnInit(): void {
    super.ngOnInit();

    // Get id from the router URL, set from PatientList component
    let pId = this.route.snapshot.paramMap.get('id');
    let col = this.route.snapshot.paramMap.get('collection')
    // TODO: change default patient or make "patient not loaded" page
    // if(!pId && !col) {
      // this.router.navigate(['/page-not-found']);
    // }
    // else {
      this.patientId = pId ? pId : "mqJfmaNgmmHaylUmhRAI"
      this.collection = col ? col : "patients"
      this.getCurrentPatient(this.patientId, this.collection);
    // }
  }

  /**
   * 
   * @param patientId The ID of the patient to retrieve from Firebase.
   * @param collection The collection to get the patient from.
   *  Subscribes to specified patient object in Firebase; converts any data
   * coming through to a Patient object for use within the app. Gets the patient's profile
   * picture from Firebase Storage.
   */
  getCurrentPatient(patientId: string, collection: string) {
    this.patientService.getPatient(patientId, collection)
      .subscribe(
        (action) => {
          let patient = new Patient();
          let obj = { id: action.payload.id, ...action.payload.data() as object };
          Object.assign(patient, obj)
          this.currentPatient = patient;
          this.currentPatient.toJSObject();

          // Get profile picture
          const storage = getStorage();
          getDownloadURL(ref(storage, this.currentPatient.getProfilePicURL()))
            .then((url) => {
              this.profilePicSrc = url;
              this.stopLoading();
            })
            .catch((error) => {
              console.error("Error in loading profile picture:", error);
            });
        },
        (err) => console.error("Error loading patient data!"),
      )
  }

  /**
   * 
   * @param update Update passed upwards by one of the child tab components.
   *  Calls the PatientService to update the specified field of the 
   * current patient. Assumes update is pre-formatted for Firebase.
   */
  updateCurrentPatient(update: Update) {
    this.patientService.setPatient(this.patientId, {[update.field]: update.data}, this.collection);
  }

  /**
   * Toggles the patient snapshot to be open or closed by switching a boolean.
   */
  toggleSnapshot() {
    this.isSnapshotOpen = !this.isSnapshotOpen;
  }

  /**
   * 
   * @param event object sent by openCloseSnapshot animation on complete.
   *  Realigns the tab ink bar when the size of the tab group changes due to
   * snapshot open/close animation.
   */
  onAnimationEvent(event: any) {
    this.tabs.realignInkBar();
  }

  /**
   * tabClick
   * 
   * @param event MatTabChange event containing the tab clicked
   * 
   * Runs when the user first clicks the MAR tab; repaints the scheduler
   * in the MAR so that it loads normally. Without this, appointments do
   * not load on first opening the MAR.
   */
  tabClick(event: MatTabChangeEvent) {
    if(event.tab.textLabel == "MAR") {
      this.mar.refresh();
    }
  }

  /**
   *  Downloads the current patient's MRN as a QR code image.
   */
  downloadQR() {
    QRCode.toDataURL(this.currentPatient.getId(), (err, url) => {
      saveAs(url, this.currentPatient.getId() + ".png");
    })
  }
}
