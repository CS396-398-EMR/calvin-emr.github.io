import { Component, OnInit } from '@angular/core';
import { FormBuilder, FormControl, FormGroup } from '@angular/forms';
import { AssessmentForm } from 'src/app/shared/assessment-form.component';
import { FormOption } from 'src/app/type/form-option.type';

@Component({
  selector: 'app-musculoskeletal-form',
  templateUrl: './musculoskeletal-form.component.html',
  styleUrls: ['./musculoskeletal-form.component.scss', '../assessments-tab.component.scss']
})
export class MusculoskeletalFormComponent extends AssessmentForm implements OnInit {

  constructor() { 
    super();
    this.controlList = [
      {control: "mobilityAndTransfers", default: "No assists" , wnl: false},
      {control: "assistiveDevices", default: false, wnl: false}
    ];
  }

  ngOnInit(): void {
    // Add listener to changes in the fall risk form, which updates
    //  fall score each time a change occurs
    this.fallRiskFormGroup.valueChanges.subscribe(() => {
      this.fallScoreFormControl.setValue(this.fallRiskScore);
    })
  }

  /**
   * Other
   */

  mobilityAndTransfersIndex = 0;
  assistiveDevicesIndex = 1;

  mobilityAndTransfersOptions = [
    new FormOption("No assists"),
    new FormOption("Standby assist"),
    new FormOption("One assist"),
    new FormOption("Two assists"),
    new FormOption("Max assists"),
  ]

  deviceOptions = [
    new FormOption("Gait Belt", "gaitBelt"),
    new FormOption("Walker", "walker"),
    new FormOption("Cane", "cane"),
    new FormOption("Trapeze", "trapeze"),
  ]

  get mobilityAndTransfersDefault() {
    return this.controlList[this.mobilityAndTransfersIndex].default;
  }

  get assistiveDevicesDefault() {
    return this.controlList[this.assistiveDevicesIndex].default;
  }

  get mobilityAndTransfersFormControl(): FormControl {
    return this.formGroup.controls.mobilityAndTransfers as FormControl;
  }

  get assistiveDevicesFormGroup(): FormGroup {
    return this.formGroup.controls.assistiveDevices as FormGroup;
  }


  /**
   * Fall Risk
   */

   ageOptions = [
    new FormOption("60 - 69 years (1 point)"),
    new FormOption("70 - 79 years (2 points)"),
    new FormOption("greater than or equal to 80 years (3 points)"),
  ]

  fallHistoryOption = "One fall within 6 months before admission (5 points)";

  ebuOptions = [
    new FormOption("Incontinence (2 points)"),
    new FormOption("Urgency or frequency (2 points)"),
    new FormOption("Urgency/frequency and incontinence (4 points)"),
  ]

  medicationsOptions = [
    new FormOption("On 1 high fall risk drug (3 points)"),
    new FormOption("On 2 or more high fall risk drugs (5 points)"),
    new FormOption("Sedated procedure within past 24 hours (7 points)"),
  ]

  pceOptions = [
    new FormOption("One present (1 point)"),
    new FormOption("Two present (2 points)"),
    new FormOption("3 or more present (3 points)"),
  ]

  mobilityFallRiskOptions = [
    new FormOption("Requires assistance or supervision for mobility, transfer, or ambulation (2 points)", "assistance"),
    new FormOption("Unsteady gait (2 points)", "gait"),
    new FormOption("Visual or auditory impairment affecting mobility (2 points)", "impairment")
  ]

  cognitionOptions = [
    new FormOption("Altered awareness of immediate physical environment (1 point)", "awareness"),
    new FormOption("Impulsive (2 points)", "impulsive"),
    new FormOption("Lack of understanding of one's physical and cognitive limitations (4 points)", "understanding"),
  ]

  get fallRiskDefault() {
    return this.controlList[0].default;
  }

  get fallRiskFormGroup(): FormGroup {
    return this.formGroup.controls.fallRisk as FormGroup;
  }

  get ageFormControl(): FormControl {
    return this.formGroup.get("fallRisk.age") as FormControl;
  }

  get fallHistoryFormControl(): FormControl {
    return this.formGroup.get("fallRisk.fallHistory") as FormControl;
  }

  get ebuFormControl(): FormControl {
    return this.formGroup.get("fallRisk.ebu") as FormControl;
  }

  get medicationsFormControl(): FormControl {
    return this.formGroup.get("fallRisk.medications") as FormControl;
  }

  get pceFormControl(): FormControl {
    return this.formGroup.get("fallRisk.pce") as FormControl;
  }

  get mobilityFallRiskFormGroup(): FormGroup {
    return this.formGroup.get("fallRisk.mobility") as FormGroup;
  }

  get cognitionFormGroup(): FormGroup {
    return this.formGroup.get("fallRisk.cognition") as FormGroup;
  }

  get fallScoreFormControl(): FormControl {
    return this.formGroup.controls.fallScore as FormControl;
  }

  get ageScore() {
    switch(this.ageFormControl.value) {
      case "60 - 69 years (1 point)": return 1;
      case "70 - 79 years (2 points)": return 2;
      case "greater than or equal to 80 years (3 points)": return 3;
      default: return 0;
    }
  }

  get fallHistoryScore() {
    return this.fallHistoryFormControl.value ? 5 : 0;
  }

  get ebuScore() {
    switch(this.ebuFormControl.value) {
      case "Incontinence (2 points)": return 2;
      case "Urgency or frequency (2 points)": return 2;
      case "Urgency/frequency and incontinence (4 points)": return 4;
      default: return 0;
    }
  }

  get medicationsScore() {
    switch(this.medicationsFormControl.value) {
      case "On 1 high fall risk drug (3 points)": return 3;
      case "On 2 or more high fall risk drugs (5 points)": return 5;
      case "Sedated procedure within past 24 hours (7 points)": return 7;
      default: return 0;
    }
  }

  get pceScore() {
    switch(this.pceFormControl.value) {
      case "One present (1 point)": return 1;
      case "Two present (2 points)": return 2;
      case "3 or more present (3 points)": return 3;
      default: return 0;
    }
  }

  get mobilityFallRiskScore() {
    let score = 0;
    let controls = this.mobilityFallRiskFormGroup.controls;

    if(controls.assistance.value) { score += 2 };
    if(controls.gait.value) { score += 2 };
    if(controls.impairment.value) { score += 2 };

    return score;
  }

  get cognitionScore() {
    let score = 0;
    let controls = this.cognitionFormGroup.controls;

    if(controls.awareness.value) { score += 1 };
    if(controls.impulsive.value) { score += 2 };
    if(controls.understanding.value) { score += 4 };

    return score;
  }

  get fallRiskScore() {
    return this.ageScore + this.fallHistoryScore + this.ebuScore + this.medicationsScore
      + this.pceScore + this.mobilityFallRiskScore + this.cognitionScore;
  }

  get fallRiskCategory(): string {
    if(this.fallRiskScore < 6) { return "Low"}
    else if(this.fallRiskScore >= 6 && this.fallRiskScore <= 13) { return "Medium"}
    else { return "High"}
  }
}
