import { Component, OnInit, Inject } from '@angular/core';
import { MatDialog, MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { Patient } from 'src/app/type/patient.type';

@Component({
  selector: 'app-patient-list-dialog',
  templateUrl: './patient-list-dialog.component.html',
  styleUrls: ['./patient-list-dialog.component.scss']
})
export class PatientListDialogComponent implements OnInit {

  title: string = "";
  prompt: string = "";
  buttonLabel: string = "";
  type: string = "";
  displayedColumns: string[] = ["id", "name"];
  dataSource: Patient[] = [];
  selectedPatients: any[] = [];
  selectedRows: number[] = [];

  constructor(
      public dialogRef: MatDialogRef<PatientListDialogComponent>,
      @Inject(MAT_DIALOG_DATA) public data: PatientListDialogData
    ){ }

  ngOnInit(): void {
    this.title = this.data.title;
    this.prompt = this.data.prompt;
    this.buttonLabel = this.data.buttonLabel;
    this.dataSource = this.data.patients;
    this.type = this.data.type;

    console.log("type: ", this.type)
  }
  /**
   * Quits dialog without assigning patient to the student
   */
  onNoClick(): void {
    this.dialogRef.close();
  }


  /**
   * Add selected row into the selected patient array to be returned once dialog closes.
   * Different logic applies for studentPatient type as it requires entire patient data
   * whereas default (patient) type only requires patient's uid
   * 
   * @param {Patient} row   Patient data of the selected row 
   * @param {number} index  Index number of selected row
   */
  selectPatient(row: Patient, index: number): void {
    let pid = row.getId();

    // If patient has been selected already, de-select it; otherwise, add 
    // patient ID and index to arrays
    if(this.selectedPatients.includes(pid)) {
      this.selectedPatients = this.selectedPatients.filter((id) => id != pid)
      this.selectedRows = this.selectedRows.filter((i) => i != index)
    }
    // Same as above, but modified conditional statement for studentPatient
    else if(this.selectedPatients.find(x => x.id === pid)){
      this.selectedPatients = this.selectedPatients.filter((row) => row.getId() != pid)
      this.selectedRows = this.selectedRows.filter((i) => i != index)
    }
    else {
      switch(this.type){
        case "studentPatient":
          this.selectedPatients.push(row);
          this.selectedRows.push(index);
          break
        default:
          this.selectedPatients.push(row.getId());
          this.selectedRows.push(index);
      }
    }
  }

  /**
   * Returns appropriate boolean value if the index value of the row 
   * is included in the selectedRows array
   * 
   * @param {number} index  Row's index number    
   * @returns {bool}        
   */
  isSelected(index: number) {
    return this.selectedRows.includes(index);
  }
}

export interface PatientListDialogData {
  type: string;
  title: string;
  prompt: string;
  buttonLabel: string;
  patients: Patient[];
}
