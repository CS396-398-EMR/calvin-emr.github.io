import { Component, Input, OnInit } from '@angular/core';
import { TabComponent } from 'src/app/shared/tab.component';
import { TableColumn } from 'src/app/type/table-column.type';
import { Order } from 'src/app/type/order.type';
import { ProgressNotes } from 'src/app/type/progress-notes.type';
import { ActionName, TableAction, TableEvent } from 'src/app/type/table-action.enum';
import { Update } from 'src/app/type/update.type';
import { Timestamp } from 'firebase/firestore';
import { Patient } from 'src/app/type/patient.type';
import { UserService } from 'src/app/services/user.service';
import { MatDialog } from '@angular/material/dialog';
import { OrderEditDialog } from 'src/app/shared/order-edit-dialog/order-edit-dialog.component';

@Component({
  selector: 'app-provider-charts-tab',
  templateUrl: './provider-charts-tab.component.html',
  styleUrls: ['./provider-charts-tab.component.scss']
})
export class ProviderChartsTabComponent extends TabComponent implements OnInit {

  // Provider order table configurations
  columns: TableColumn[] = [
    new TableColumn("Medication", "medName",
      (element: Order) => element.medName),
    new TableColumn("Dosage", "dosage",
      (element: Order) => element.dosage),
    new TableColumn("Quantity", "quantity",
      (element: Order) => element.quantity),
    new TableColumn("Quantity Given", "quantityGiven",
      (element: Order) => element.quantityGiven),
    new TableColumn("Route", "route",
      (element: Order) => element.route),
    new TableColumn("Order", "order",
      (element: Order) => element.order),
    new TableColumn("When to Give", "startDate",
      (element: Order) => {
        return new Date(element.startDate.seconds * 1000).toLocaleString()
      }),
    new TableColumn("Latest to Give", "endDate",
    (element: Order) => {
      return new Date(element.endDate.seconds * 1000).toLocaleString()
    }),
    new TableColumn("Signature", "signature", 
      (element: Order) => element.signature),
    new TableColumn("Administered", "administered",
      (element: Order) => element.administered),
  ];

  // Progress Notes tab (PN = Progress Notes)
  columnsPN: TableColumn[] = [
    new TableColumn("Day/Time", "date",
      (element: ProgressNotes) => {
        // return element.date.toDate()
        return new Date(element.date.seconds * 1000).toLocaleString()
      }
    ),
    new TableColumn("Notes", "notes",
      (element: ProgressNotes) => element.notes),
    new TableColumn("Signature", "signature",
      (element: ProgressNotes) => element.signature)
  ];

  actions: TableAction[] = [];

  @Input() providerOrders: Order[] = [];
  @Input() progressNotes: ProgressNotes[] = [];
  @Input() historyAndPhysical: Patient = new Patient();

  constructor(private userService: UserService,
    public dialog: MatDialog) {
    super();
    if(this.userService.getCurrentUserRole() == "admin") {
      this.columns.push(
        new TableColumn("Day/Time", "date",
        (element: Order) => {
          return new Date(element.dateCreated.seconds * 1000).toLocaleString()
        })
      )
    }
  }

  ngOnChanges(): void {
    // EMR table only has actions if user is admin
    if (this.isAdmin) {
      this.actions = [
        new TableAction("Edit", "edit"),
        new TableAction("Delete", "delete", "Undo Delete", "undo"),
      ]
    }
    else {
      this.actions = []
    }
    
    this.stopLoading();
  }

  onActionClick(event: TableEvent) {
    if (event.action.name == ActionName.Edit) {
      this.editOrder(event.data as Order);
    }
    else if (event.action.name == ActionName.Delete) {
      this.providerOrders = this.providerOrders.filter((order) => {
        return order != event.data as Order;
      })
    }

    this.updateProviderOrders();
  }

  addOrder(order: Order) {
    this.providerOrders.push(order);
    this.updateProviderOrders()
  }

  editOrder(order: Order) {
    const dialogRef = this.dialog.open(OrderEditDialog, {
      width: '50%',
      data: order,
      disableClose: true,
    });

    dialogRef.afterClosed().subscribe((newOrder: Order) => {
      if(newOrder) {
        let indexToRemove = this.providerOrders.findIndex(o => o.id == order.id)
        this.providerOrders.splice(indexToRemove, 1, newOrder)
        this.updateProviderOrders();
      }
    })
  }

  addNote(note: ProgressNotes) {
    this.progressNotes.push(note);
    this.updateProgressNotes();
  }

  updateProviderOrders(): void {
    // Merge providerOrders array with empty array - this triggers the
    //  ngOnChanges() in the emr-table component, since pushing to an
    //  array is not seen as a change by onChanges()
    this.providerOrders = ([] as any[]).concat(this.providerOrders);
    this.updatePatient("providerOrders", this.providerOrders);
  }

  updateProgressNotes(): void {
    // Merge progressNotes array with empty array - this triggers the
    //  ngOnChanges() in the emr-table component, since pushing to an
    //  array is not seen as a change by onChanges()
    this.progressNotes = ([] as any[]).concat(this.progressNotes);
    super.updatePatient("progressNotes", this.progressNotes);
  }

  updatePatient(field: string, data: any[]): void {
    // Create new update object
    let update = new Update(field, data);

    // Reformat update to JSON
    update = JSON.parse(JSON.stringify(update))

    // Reformat datestrings in data to Timestamp
    update.data.forEach((obj: any) => {
      obj.dateCreated = Timestamp.fromMillis(obj.dateCreated.seconds * 1000)
      obj.startDate = Timestamp.fromMillis(obj.startDate.seconds * 1000);
      obj.endDate = Timestamp.fromMillis(obj.endDate.seconds * 1000);
    });

    this.updatePatientEvent.emit(update);
  }

  get isAdmin() {
    return this.userService.getCurrentUserRole() == "admin";
  }
}
