import { Component, OnInit } from '@angular/core';
import { User } from '../../type/user.type';
import { Patient } from 'src/app/type/patient.type';
import { UserService } from 'src/app/services/user.service';
import { PageComponent } from 'src/app/shared/page.component';
import { PatientService } from 'src/app/services/patient.service';
import { TableColumn } from 'src/app/type/table-column.type';
import { Router } from '@angular/router';
import { onAuthStateChanged, getAuth } from 'firebase/auth';
import { MatDialog, MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { PatientListDialogComponent } from 'src/app/component/patient-list-dialog/patient-list-dialog.component';
import { ActionName, TableEvent } from 'src/app/type/table-action.enum';
import { FileUploadDialogComponent } from 'src/app/shared/file-upload-dialog/file-upload-dialog.component';
import { StudentPatientSelectDialogComponent } from 'src/app/component/student-patient-select-dialog/student-patient-select-dialog.component';
import { arrayUnion } from 'firebase/firestore';
import { inspect } from 'util'
import { isEqual, transform, isArray, isObject, merge } from 'lodash';

@Component({
  selector: 'app-student-list',
  templateUrl: './student-list.component.html',
  styleUrls: ['./student-list.component.scss']
})
export class StudentListComponent extends PageComponent implements OnInit {
  currentUser = new User();
  patients: Patient[] = [];
  students: any[] = [];
  
  // Variable used for accepting return value from modal
  selectedPatient: string = "";

  // Student table configuration
  public studentColumns: TableColumn[] = [
    new TableColumn("UID", "uid",
      (element: any) => element.user.getId()),
    new TableColumn("Student Name", "studentName",
      (element: any) => element.user.lastNameFirstName),
  ];

  // Used for importing users
  importedStudents: string[] = [];
  importErrors: string[] = [];

  constructor(private router: Router,
              userService: UserService, 
              private patientService: PatientService,
              private dialog: MatDialog) { 
    super(userService);
  }

  ngOnInit(): void {
    super.ngOnInit();
    // Checks for login status
    onAuthStateChanged(getAuth(), (user) => {
      // User is logged in
      if (user) {
        this.getCurrentUser(user.uid)
        this.getAllStudents();
      }
      // User is not logged in (Currently used for guest login)
      else {
        this.getCurrentUser("MR2H0y5piRSpCF7Ty8KtiS9ai0X2")   // TODO: remove id
      }
    })
    console.log("current user: ", this.students)
  }

  /**
   * Get logged in user data from the firebase based on user's id.
   * 
   * @param uid
   */
  getCurrentUser(uid: string) {
    // Get logged in user data from the firebase
    this.userService.getUser(uid)
      .subscribe(
        // Success
        (action) => {
          this.setCurrentUser(action);
        },
        // Error
        (err) => console.error("Error loading user data!"),
        // Completion Handler
        () => console.log("User loaded successfully!")
      );
  }

  /**
   * Flip "checked" boolean value of the student after clicking the checkbox
   * 
   * @param event 
   */
  studentCheckToggle(event: any) {
    // Get the index value of checked student row
    let index = this.students.findIndex(obj => obj.user.getId() == event);
    // flip checked boolean value of a index row
    this.students[index].checked = this.students[index].checked? false : true;
  }

  /**
   * Populate the array with the 
   * 
   * @param event 
   */
  goToStudentPatient(event: TableEvent) {
    if(event.action.name == ActionName.Navigate) {
      let studentId = event.data  // id is passed from emr-table
      let tempData: any[] = []    // temporarily store patient's full name and uid to be passed down to the dialog

      // Retrieve studentPatient document from firebase linked to the template patient uID
      this.patientService.getStudentPatientbyId(studentId as any).then(
        (res) => {
          res.forEach(doc => 
            {
              tempData.push({
                name: doc.get("firstName") + 
                  " " + doc.get("middleName") + 
                  " " + doc.get("lastName"), 
                uid: doc.id})
            })
          // Open dialog to allow admin to select from the list of patients assigned to student
          this.openStudentPatientSelectDialog(tempData)
        },
        (err) => {
          console.log(err);
        }
      )
    }
    else{
      console.error("Error in navigating to patient: event is not category 'Navigate")
    }
  }

  /**
   *  Get all "patient" documents linked to current user
   */
  getPatients() {
    this.currentUser.patientIds.forEach((id) => {
      this.patientService.getPatient(id, "patients")
        .subscribe(
          (action) => {
            let patient = new Patient();
            let obj = { id: action.payload.id, ...action.payload.data() as object };
            Object.assign(patient, obj)
            this.patients.push(patient);

            if(this.patients.length == this.currentUser.patientIds.length) {
              this.stopLoading();
            }
          },
          (err) => console.error("Error loading patient data!"),
        )
    })
  }
  
  /**
   * Set student-list page's current user using DocumentSnapshot data from the firebase 
   * @param {DocumentSnapshot} res 
   */
  setCurrentUser(res: any) {
    // Put JSON data into JavaScript object
    Object.assign(this.currentUser, res.payload.data());
    this.currentUser.setId(res.payload.id);

    // Get each patient that the user has
    if(this.currentUser.patientIds.length > 0) {
      this.getPatients();
    }
    else {
      this.isLoading = false;
    }
  }

  /**
   * Retrieve QuerySnapshot from the firebase backend
   * that meets the condition of role being a "student"
   * from the "users" document
   */
  getAllStudents(){
    this.userService.getAllUsers().then(
      (res) => {
        res.forEach((doc: any) => 
          {
            this.addStudent(doc)
          })
      },
      // When no data is received
      (err) => {
        console.log(err);
      }
    )
  }

  /**
   * Add student object to the "students" array
   * 
   * @param res User firebase document containing user information
   */
  addStudent(res: any) {
    let student = new User();
    Object.assign(student, res.data())

    this.students.push(
      {
        user: new User(res.id, student.email, student.firstName, student.lastName, student.role),
        checked:false
      }
    )
  }

  /**
   * Create student credentials using uploaded student credential data
   * from student upload dialog
   * 
   * @param {string} studentListString 
   */
  createStudents(studentListString: string) {
    // Ignore first row, since first row is column headers
    let newStudents = studentListString.split('\r').slice(1);
    this.isLoading = true;
    newStudents.forEach((studentString) => {
      // Try to add data; if data is not valid, add to errors array
      try {
        let userParams = studentString.split(',').filter(x => x != "").map(x => x.trim());
        // Parameter order: id, email, first name, last name, role
        let student = new User(userParams[0], userParams[1], userParams[2], userParams[3],
                                "student");
        if(!student.hasValidEmail) {
          throw new Error();
        }

        this.userService.createUser(student);
      }
      catch(e) {
        this.importErrors.push(studentString)
      }
    })

    this.isLoading = false;

    // Display errors, if any
    if(this.importErrors.length > 0) {
      let errorString = this.importErrors.join("\n")
      console.error("Could not import the following data:", errorString, 
                    "\nVerify that your file is correctly formatted and that no data is missing.")
      // TODO: make this a dialog box
    }
  }

  // Open modal dialog
  openDialog(operation: string): void {
    const dialogRef = this.dialog.open(PatientListDialogComponent, {
      width: '500px',
      data: {
        title: "List of Patients",
        prompt: `Which patients would you like to ${operation}?`,
        buttonLabel: `${operation} Patients`,
        patients: this.patients,
        type: "studentPatient"
      }
    });

    // Temporary studentList to perform action to selected users
    let tempStudentList: string[] = []
    this.students.filter(this.studentChecked).forEach(
      obj => tempStudentList.push(obj.user.getId())
    )

    // Ater dialog is closed, perform an appropriate action based on the operation
    dialogRef.afterClosed().subscribe(results => {
      results.forEach(
        (result: Patient) => {
          if(result){
            switch(operation) {
              case "assign":
                this.assignPatient(result, tempStudentList);
                break
              case "delete":
                this.deletePatients(result.getId(), tempStudentList, "studentPatients")
                break;
              case "update":
                this.updatePatient(result.getId(), result, tempStudentList, "studentPatients")
                break;
              default:
                alert("Invalid operation")
            }
          }
        }
      )
      console.log('The dialog was closed');
    });
  }

  /**
   * Function used for filter function. Returns true for users checked
   * at the student list table and false for otherwise.
   * 
   * @param user
   * @returns boolean
   */
  studentChecked(user: any) {
    return user.checked
  }

  /**
   * Assign studentPatient associated with the selected patient template
   * 
   * @param patientUid  identification number for the selected patient template
   * @param studentList list of selected students
   */
  assignPatient(patient: Patient, studentList: string[]): void {
    if(studentList.length > 0){
      studentList.forEach(student => {
        // Checks if student already had a patient assigned
        this.patientService.patientDuplicateCheck(student, patient.getId(), "studentPatients").then(
          res => {
            // If not assigned
            if(res.empty) {
              // append student id value to the studentPatient to identify who is responsible for this student patient
              patient.setStudentId(student)
              this.patientService.createStudentPatient(patient)
              this.assignStudentPatientToUser(student)
            }
            // If assigned 
            else {
              alert(`student ${student} already has patient ${patient.getId()} assigned`)
            }
          }
        )
      })
    }
    else {
      alert("no student was selected")
    }
  }

  /**
   * Compare two objects and return the differences and common values
   * Potentially used for the updating studentPatient object in firebase
   * Code URL: https://newbedev.com/javascript-typescript-get-difference-between-two-objects-code-example
   * 
   * @param origObj   Object 1 for comparison
   * @param newObj    Object 2 for comparison
   * @returns object type that contains difference or commonality of two objects
   */

  objectDifference(origObj:Object, newObj:Object) {
    let changes = (newObj:Object, origObj:Object) => {
      let arrayIndexCounter = 0;
      return transform(newObj, function(result: any, value, key) {
        if (!isEqual(value, origObj[key])) {
          let resultKey = isArray(origObj) ? arrayIndexCounter++ : key
          result[resultKey] = (isObject(value) && isObject(origObj[key])) ? changes(value, origObj[key]) : value
        }
      })
    }
    return changes(newObj, origObj)
  }

  objectSame(origObj:Object, newObj:Object) {
    let changes = (newObj:Object, origObj:Object) => {
      let arrayIndexCounter = 0;
      return transform(newObj, function(result: any, value, key) {
        if (isEqual(value, origObj[key])) {
          let resultKey = isArray(origObj) ? arrayIndexCounter++ : key
          result[resultKey] = (isObject(value) && isObject(origObj[key])) ? changes(value, origObj[key]) : value
        }
      })
    }
    return changes(newObj, origObj)
  }
}
