import { Component, OnInit } from '@angular/core';
import { Router } from '@angular/router';
import { User } from '../../type/user.type';
import { FormControl, Validators } from '@angular/forms';

import { getFirestore, doc, setDoc } from "firebase/firestore"; //May want to create service for storing info to firestore
import { getAuth, onAuthStateChanged } from "firebase/auth";



@Component({
  selector: 'app-user-profile',
  templateUrl: './user-profile.component.html',
  styleUrls: ['./user-profile.component.scss']
})
export class UserProfileComponent implements OnInit {
  firstName = new FormControl('', [Validators.required]);
  middleName = new FormControl('');
  lastName = new FormControl('', [Validators.required]);
  calvinID = new FormControl('', [Validators.required, Validators.pattern("^[0-9]*$"), Validators.maxLength(7), Validators.minLength(7)]);
  permissionLevel = new FormControl('', [Validators.required]);

  constructor(private router: Router) { }

  ngOnInit(): void {
  }

  getEmailErrorMessage() {
    if (this.firstName.hasError('required')) {
      return 'You must enter a value';
    }
    return this.firstName.hasError('email') ? 'Not a valid email' : '';
  }

  onSubmit(){
    // TODO: Create user service that would parse user information to create more uniformed coding style
    // Refer to patient.service and patient-list.component
    
    // let usr = new User()

    let uid = "";
    let fName = this.firstName.value;
    let middleName = this.middleName.value;
    let lName = this.lastName.value;
    let calvinID = this.calvinID.value;
    let permissionLevel = this.permissionLevel.value;

    const auth = getAuth();

    // Retrieve user's uid to use it for creating user document
    onAuthStateChanged(auth, (user) => {
      if (user) {
        this.addUserProfile(user.uid, fName, lName, calvinID, permissionLevel, middleName);
        alert("Profile setting successful");
        this.router.navigateByUrl('/patients');
      } else {
        alert("user not logged in");
      }
    });
  }

  /**
   * @description Create firebase user document that is linked to the user's authentication uid.
   * Document ID is used as a "foreign key" to link with the user account's UID
   * 
   * @param uid             Unique identification number assigned to the registered user
   * @param fName           First name of the user
   * @param lName           Last name of the user
   * @param calvinID        Calvin Identification Number (Can be replaced with other form of id)
   * @param permissionLevel User's role (student vs professor/admin)
   * @param mName           Middle name of the user
   */
  async addUserProfile(uid: string, fName: string, lName: string, calvinID: number, permissionLevel: string, mName:string) {
    // link to DB
    const db = getFirestore();

    await setDoc(doc(db, "users", uid), {
      firstName: fName,
      lastName: lName,
      userID: calvinID,
      role: permissionLevel
    });
  }

}
