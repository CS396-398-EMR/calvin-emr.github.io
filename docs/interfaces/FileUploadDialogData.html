import { Component, Inject, Input, OnInit, Type } from '@angular/core';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { ProgressSpinnerMode } from '@angular/material/progress-spinner';

export interface FileUploadDialogData {
  /**
   * The title to be displayed at the top of the dialog.
   */
  title: string;

  /**
   * The type of the object to create.
   */
  type: Type<any>;

  /**
   * A function that returns true if the object is valid (or created correctly), 
   * and false otherwise.
   */
  valid: (obj: Type<any>) => boolean;
}

@Component({
  selector: 'app-file-upload-dialog',
  templateUrl: './file-upload-dialog.component.html',
  styleUrls: ['./file-upload-dialog.component.scss']
})
export class FileUploadDialogComponent implements OnInit {

  /**
   * Flag used to indicate processing of uploading information.
   */
  isLoading: boolean = false;

  /**
   * The mode of the ProgressSpinner. Defaults to 'indeterminate'.
   */
  mode: ProgressSpinnerMode = 'indeterminate';

  /**
   * The value of the ProgressSpinner that indicates how far the page has loaded.
   * Used only with the 'determinate' mode.
   */
  value = 50;

  /**
   * Stores the Type object passed through the dialog data.
   */
  Obj: Type<any>;

  /**
   * List to hold all errors encountered when processing uploaded file.
   */
  importErrors: string[] = [];

  result: Type<any>[] = [];

  file: Blob | null = null;
  fileName: string = "No file selected.";
  lastModifiedDate: string = "N/A";
  lastModifiedTime: string = "";
  fileSize: number = 0;
  fileSelected: boolean = false;

  constructor(
    public dialogRef: MatDialogRef<FileUploadDialogComponent>,
    @Inject(MAT_DIALOG_DATA) public data: FileUploadDialogData
  ) { 
    this.Obj = data.type;
  }

  ngOnInit(): void {
  }

  /**
   * Reads imported file and convert it into string with an appropriate format
   * 
   * @param event 
   */
  onFileChange(event: any) {
    this.file = event.target.files[0];
    this.fileName = event.target.files[0].name;
    this.lastModifiedDate = event.target.files[0].lastModifiedDate.toDateString();
    this.lastModifiedTime = event.target.files[0].lastModifiedDate.toTimeString().substring(0, 8)
    this.fileSize = Math.round(event.target.files[0].size / 1000)
    this.fileSelected = this.file ? true : false;
  }

  onCancel() {
    this.dialogRef.close(null);
  }

  onUpload() {
    if(this.file) {
      this.startLoading();
      const fileReader = new FileReader();
      fileReader.onload = (e) => {
        const fileResult = e.target?.result as string;
        this.processFile(fileResult);
        this.dialogRef.close(this.result);
      }
      fileReader.readAsText(this.file);
    }
  }

  /**
   * This function uses the type passed by the caller to construct objects
   * of that type and pass them back to the caller. Since this type is generic,
   * the user uploading the file must ensure that the parameters are in the correct order;
   * otherwise, the object will not be created correctly.
   * 
   * @param result The string created by the FileReader.
   */
  processFile(result: string) {
    // Clear any previous import errors
    this.importErrors = [];

    // Ignore first row, since first row is column headers
    let newObjects = result.split('\r').slice(1);

    newObjects.forEach((objString) => {
      // Try to add data; if data is not valid, add to errors array
      try {
        let params = objString.split(',').filter(x => x != "").map(x => x.trim());
        let obj = new this.Obj(...params);

        if(!this.data.valid(obj)) {
          throw new Error();
        }

        this.result.push(obj);
      }
      catch(e) {
        this.importErrors.push(objString)
      }
    })

    this.stopLoading();

    // Display errors, if any
    if(this.importErrors.length > 0) {
      let errorString = this.importErrors.join("\n")
      console.error("Could not import the following data:", errorString, 
                    "\nVerify that your file is correctly formatted and that no data is missing.")
      // TODO: make this a dialog box
    }
  }

  startLoading() {
    this.isLoading = true;
  }

  stopLoading() {
    this.isLoading = false;
  }
}
