import { Component, Inject, OnInit } from '@angular/core';
import { FormControl, Validators } from '@angular/forms';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { EmailAuthProvider, getAuth, reauthenticateWithCredential } from 'firebase/auth';
import { Subscription } from 'rxjs';
import { cancelButtonLabel, invalidPasswordError, notLoggedInError, returnKeyCode, verifyButtonLabel, verifyPrompt, verifyTitle } from 'src/app/global';

export interface LoginDialogData {
  title: string;
  prompt: string;
  cancelButtonLabel: string;
  verifyButtonLabel: string;
}

@Component({
  selector: 'app-login-dialog',
  templateUrl: './login-dialog.component.html',
  styleUrls: ['./login-dialog.component.scss']
})
export class LoginDialogComponent implements OnInit {

  title: string = verifyTitle;
  prompt: string = verifyPrompt;
  cancelButtonLabel: string = cancelButtonLabel;
  verifyButtonLabel: string = verifyButtonLabel;

  label: string = 'Enter your current password';
  password: FormControl = new FormControl('', [Validators.required]);
  passwordSub: Subscription;
  error: string = '';

  /**
   * Change the dialog's labels based on data passed in from the caller, if any.
   * Subscribe to the password valueChanges to clear any errors when the user
   * begins typing in the password form field.
   * 
   * @param dialogRef The reference to this dialog.
   * @param data The data injected into this dialog.
   */
  constructor(public dialogRef: MatDialogRef<LoginDialogComponent>,
    @Inject(MAT_DIALOG_DATA) public data: LoginDialogData) {
      if(data) {
        if(data.title) this.title = data.title;
        if(data.prompt) this.prompt = data.prompt;
        if(data.cancelButtonLabel) this.cancelButtonLabel = data.cancelButtonLabel;
        if(data.verifyButtonLabel) this.verifyButtonLabel = data.verifyButtonLabel;
      }
      this.passwordSub = this.password.valueChanges.subscribe(() => {
        this.error = '';
      })
  }

  ngOnInit(): void { }

  /**
   * Calls the onVerify function when the "Enter/Return" key is pressed.
   * 
   * @param event Object passed by (keyup) property of input tag.
   */
   handleKeyUp(event: any) {
    if(event.keyCode == returnKeyCode) {
      this.onVerify();
    }
  }

  /**
   * 
   */
  onVerify(): void {
    const auth = getAuth();
    const user = auth.currentUser;
    const email = user?.email;

    if(user && email) {
      const credential = EmailAuthProvider.credential(email, this.password.value)
      reauthenticateWithCredential(user, credential).then(() => {
        this.dialogRef.close(true)
      }).catch(() => {
        this.error = invalidPasswordError;
      })
    }
    else {
      this.error = notLoggedInError;
    }
  }

  /**
   * Send a false value back to the caller, indicating that the user
   * cancelled.
   */
  onCancel(): void {
    this.dialogRef.close(false);
  }

  get hasError(): boolean {
    return this.error != null && this.error != '';
  }

}
