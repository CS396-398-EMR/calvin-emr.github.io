  constructor(private fb: FormBuilder, private userService: UserService) { 
    super();
  }

  ngOnInit(): void {
  }

  onSubmit(): void {
    this.form.enable();

    let assessment = formToAssessment(this.form)
    this.assessments.push(assessment);
  
    this.assessments = ([] as any[]).concat(this.assessments);
    this.updatePatient(nursingAssessmentsField, this.assessments)

    this.form.reset();
    alert("Assessment submitted!")
  }

  /**
   * @returns Whether or not current user has admin role.
   */
  get isAdmin(): boolean {
    return this.userService.isCurrentUserAdmin;
  }
