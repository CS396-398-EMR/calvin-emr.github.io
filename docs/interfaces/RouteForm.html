import { Component, Inject, OnInit, ViewChild } from '@angular/core';
import { AbstractControl, FormArray, FormBuilder, FormControl, FormGroupDirective, Validators } from '@angular/forms';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { FormOption } from 'src/app/types/form-option.type';
import { Medication } from 'src/app/types/medication.type';
import { ActionName, TableEvent } from 'src/app/types/table-action.enum';
import { uniqueEntryValidator } from 'src/app/directives/unique-entry.directive';

export interface RouteForm {
  routeName: string;
  otherName: string;
}

@Component({
  selector: 'app-med-dialog',
  templateUrl: './med-dialog.component.html',
  styleUrls: ['./med-dialog.component.scss']
})
export class MedDialogComponent implements OnInit {

  @ViewChild('documentEditForm') documentEditForm!: FormGroupDirective; 

  /**
   * List of medication ids passed from parent (medication-list) to validate id input.
   */
  ids: string[] = [];

  /**
   * Form group for the dialog's form.
   */
  form = this.fb.group({
    name: ['', Validators.required],
    id: ['', [Validators.required, Validators.minLength(4)]],
    routes: this.fb.array([]),
  });

  /**
   * List of possible medication administration routes.
   */
  routes = [
    new FormOption("IM"),
    new FormOption("Inhalation"),
    new FormOption("IV"),
    new FormOption("NS"),
    new FormOption("Oral"),
    new FormOption("SL"),
    new FormOption("SQ"),
    new FormOption("Topical"),
    new FormOption("Rectal"),
    new FormOption("Other"),
  ]
  
  
  /**
   * Specifies the content to render within the dialog. Three types:
   * Add, Edit, and Delete.
   */
  
  type: string = "";

  /**
   * The title text to display at the top of the dialog box.
   */
  title: string = "";

  /**
   * The text label of the form's submit button.
   */
  btnLabel: string = "";

  /**
   * Medication that is being added/edited/deleted.
   */
  med: Medication = new Medication();

  /**
   * Assigns the dialog's properties values based on the action passed to it.
   */
  constructor(
    public dialogRef: MatDialogRef<MedDialogComponent>,
    @Inject(MAT_DIALOG_DATA) public data: TableEvent,
    private fb: FormBuilder
  ) { 
    // Split the passed data into the medication and the list of medication IDs
    let obj = data.data as {medication: Medication, ids: string[]}
    Object.assign(this.med, obj.medication);

    switch(data.action.name) {
      case ActionName.Add: {
        this.type = ActionName.Add;
        this.title = "New Medication";
        this.btnLabel = "Add";
        this.ids = obj.ids;
        this.addRoute();    // Add a route to routes FormArray

        // Add validator to medication id to ensure it is unique
        this.idFormControl.addValidators([uniqueEntryValidator(this.ids)]);
        break;
      }
      case ActionName.Edit: {
        this.type = ActionName.Edit;
        this.title = "Edit Medication";
        this.btnLabel = "Update";
        this.idFormControl.setValue(this.med.id);
        this.nameFormControl.setValue(this.med.name);

        // Set route controls
        this.med.getRoutes().forEach((route) => {
          this.addRoute(route);
        })

        break;
      }
      case ActionName.Delete: {
        this.type = ActionName.Delete;
        this.title = "Delete Medication";
        this.btnLabel = "Delete";
        break;
      }
      default: {
        this.title = "Error";
      }
    }
  }

  ngOnInit(): void { }

  /**
   * Generates a random ID for a new medication and fills the ID form
   * field with the value.
   */
  generateID() {
    let id = ("D9" + Date.now()).slice(0, 14);
    this.form.controls.id.setValue(id);
  }

  /**
   * Adds a route FormControl to the routes FormArray. Default values are empty strings,
   * but if name is specified, the FormControls will be pre-filled based on the name.
   * If name is "Other - ...", with the "..." being a custom route string, then the 
   * otherName FormControl will be set to whatever "..." is.
   * 
   * @param name route name as string
   */
  addRoute(name?: string) {
    // Defaults for a route that has not been filled in
    let rName = "";
    let oName = "";
    let disabled = true;

    // Get route name; if route name is "Other", get other name as well
    if(name) {
      let routeNames = name.split(' - ');
      rName = routeNames[0];

      if(rName == "Other") {
        if(routeNames.length > 1) {
          oName = routeNames[1];
        }
        disabled = false;
      } 
    }

    // Create form group to push to form array
    const routeForm = this.fb.group({ 
      routeName: [rName],
      otherName: [{value: oName, disabled: disabled}]
    });
    this.routesFormArray.push(routeForm);
  }

  /**
   * Deletes the FormControl at the specified index.
   * 
   * @param routeIndex the index of the route FormControl in the FormArray
   */
  deleteRoute(routeIndex: number) {
    this.routesFormArray.removeAt(routeIndex);
  }

  /**
   * Enables the secondary input if user selects "Other" as the route.
   * 
   * @param fg FormGroup passed from template (assumed to be the FormGroups added to routes FormArray)
   */
  routeChanged(fg: AbstractControl) {
    if(this.isOther(fg)) {
      fg.get('otherName')?.enable();
    }
    else {
      fg.get('otherName')?.disable();
      fg.get('otherName')?.setValue('');
    }
  }

  /**
   * Checks if routeName contol is "Other".
   * 
   * @param fg FormGroup (assumed to be the FormGroups added to routes FormArray)
   * @returns boolean value
   */
  isOther(fg: AbstractControl) {
    return fg.get('routeName')?.value == "Other";
  }

  /**
   * Sets values of medication to pass back to caller based on the type of dialog that is open.
   */
  onSubmit(): void {
    // Set medication values based on form
    if(this.type == ActionName.Add || this.type == ActionName.Edit) {
      this.med.id = this.form.get('id')?.value
      this.med.name = this.form.get('name')?.value
      this.med.routes = this.getRoutes();  
    }

    this.dialogRef.close(this.med);
  }

  /**
   * Gets the RouteForm objects from the routes FormArray and
   * transforms them into strings.
   * If the route name is "Other" and the other route name is "...",
   * then they will be combined into one string: "Other - ..."
   * 
   * @returns array of routes as strings
   * 
   */
  getRoutes(): string[] {
    let routeObjs = this.form.get('routes')?.value
    let routeStr = "";
    let routes: string[] = []

    routes = routeObjs.map((obj: RouteForm) => {
      routeStr = obj.routeName;
      if(routeStr == "Other") {
        routeStr += " - " + obj.otherName
      }
      return routeStr;
    }).filter((routeStr: string) => {
      return routeStr != "";
    })

    return routes;
  }


  /**
   * Sets the (nonempty) Medication object's properties to the current values
   * in the form group and passes the updated object to the MedicationService.
   * The ID is not changed, since we want to update and not create a new object.
   * Closes the dialog.
   */
  updateMedication(): void {
    this.med.name = this.form.get('name')?.value
    this.med.routes = this.getRoutes();
  }

  get isAddType(): boolean {
    return this.type == ActionName.Add;
  }

  get isEditType(): boolean {
    return this.type == ActionName.Edit;
  }

  get hasIdError(): boolean {
    return this.idFormControl.invalid && 
          (this.idFormControl.dirty || this.idFormControl.touched);
  }

  get hasIdRequiredError(): boolean {
    return this.idFormControl.errors?.required;
  }

  get hasIdMinLengthError(): boolean {
    return this.idFormControl.errors?.minlength;
  }

  get hasIdUniqueEntryError(): boolean {
    return this.idFormControl.errors?.uniqueEntry;
  }

  get isSubmitDisabled(): boolean {
    return !this.form.valid && this.type != ActionName.Delete;
  }

  get routesFormArray(): FormArray {
    return this.form.controls["routes"] as FormArray;
  }

  get idFormControl(): FormControl {
    return this.form.controls.id as FormControl;
  }

  get nameFormControl(): FormControl {
    return this.form.controls.name as FormControl;
  }
}
