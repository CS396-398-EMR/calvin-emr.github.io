import { adminRole, calvinEmail, calvinStudentEmail, studentRole } from "../global";

/**
 * Represents a user for the application.
 */
export class User {
  /**
   * The user's unique identifier. This is their document ID in Firestore, as well
   * as the ID of their authentication account.
   */
  id: string;

  /**
   * The user's email.
   */
  email: string;

  /**
   * The user's first name.
   */
  firstName: string;

  /**
   * The user's last name.
   */
  lastName: string;

  /**
   * The user's role. Role determines the user's permissions level, enabling or disabling
   * them from seeing specific pages and components.
   */
  role: string;

  /**
   * A list of IDs representing the user's assigned patients. The IDs are used to get the 
   * full patient document from Firestore.
   */
  patientIds: string[];

  /**
   * A list of IDs representing the user's students. The IDs are used to get the full student
   * documents from Firestore.
   */
  studentIds?: string[];

  constructor(id?: string, email?: string, firstName?: string, lastName?: string,
              role?: string) {
    this.id = id ? id : "";
    this.email = email ? email : "";
    this.firstName = firstName ? firstName : "";
    this.lastName = lastName ? lastName : "";
    this.role = role ? role : studentRole;      // user objects default to "student" role
    this.patientIds = [];
  }

  /**
   * Sets the user's ID to the ID passed to the function.
   * 
   * @param id the user's new unique identifier.
   */
  setId(id: string) {
    this.id = id;
  }

  /**
   * @returns the user's ID.
   */
  getId() {
    return this.id;
  }

  /**
   * @returns The user's first name followed by their last name.
   */
  get firstNameLastName() {
    return this.firstName + ' ' + this.lastName;
  }

  /**
   * @returns The user's last name followed by a comma, followed by their first name.
   */
  get lastNameFirstName() {
    return this.lastName + ", " + this.firstName;
  }

  /**
   * Used when creating new users through uploading a CSV. See the student-list component
   * for more information in how this accessor is used.
   * 
   * @returns Whether the user's email is a Calvin University email or not.
   */
  get hasValidEmail() {
    return this.email.toLocaleLowerCase().match(calvinEmail) || this.email.toLocaleLowerCase().match(calvinStudentEmail);
  }

  /**
   * @returns Whether user is an admin or not.
   */
  get isAdmin() {
    return this.role.toLocaleLowerCase() == adminRole;
  }
}
