import { Timestamp } from 'firebase/firestore';

/**
 * Used to represent a developer on the Calvin EMR Development Team. 
 * Mostly used to render the About page dynamically by pulling developer
 * information from Firestore.
 */
export class DeveloperProfile {
  /**
   * Whether the developer is a student or not.
   */
  isStudent: boolean;

  /**
   * The developer's first name.
   */
  firstName: string;

  /**
   * The developer's last name.
   */
  lastName: string;

  /**
   * The name of the developer's image stored in Firebase Storage.
   * This is used to get the image URL when the "About Us" page loads.
   */
  imageName: string;

  /**
   * The URL of the developer's image stored in Firebase Storage.
   * This is set during the loading of the "About Us" page.
   */
  imageURL: string;

  /**
   * The developer's first degree.
   */
  degreeFirst: string;

  /**
   * The developer's second degree, if any.
   */
  degreeSecond: string;

  /**
   * The developer's primary academic major.
   */
  majorFirst: string;

  /**
   * The developer's secondary academic major, if any.
   */
  majorSecond: string;
  
  /**
   * A description of the developer's academic minors, if any.
   */
  minors: string;
  
  /**
   * The developer's graduation date.
   */
  graduation: Timestamp;

  /**
   * The URL of the developer's LinkedIn page.
   */
  linkedIn: string;

  /**
   * A description of the developer's hobbies.
   */
  hobby: string;
  
  constructor(){
      this.isStudent = true;
      this.firstName = "";
      this.lastName = "";

      this.imageName = "";
      this.imageURL = "";

      this.degreeFirst = "";
      this.degreeSecond = "";
      this.majorFirst = "";
      this.majorSecond = "";
      this.minors = "";
      
      this.graduation = Timestamp.fromDate(new Date());
      this.linkedIn = "";

      this.hobby = "";
  }
}
