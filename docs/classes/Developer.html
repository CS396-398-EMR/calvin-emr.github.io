import { Timestamp } from 'firebase/firestore';

/**
 * Used to represent a developer on the Calvin EMR Development Team.
 * Mostly used to render the About page dynamically by pulling developer
 * information from Firestore.
 */
export class Developer {
  /**
   * The developer's first name.
   */
  firstName: string;

  /**
   * The developer's last name.
   */
  lastName: string;

  /**
   * The name of the developer's image stored in Firebase Storage.
   * This is used to get the image URL when the "About Us" page loads.
   */
  imageName: string;

  /**
   * The URL of the developer's image stored in Firebase Storage.
   * This is set during the loading of the "About Us" page.
   */
  imageUrl: string;

  /**
   * Array that stores all of the developer's degree information.
   */
  degreeInfo: DegreeInfo[];

  /**
   * The developer's graduation date.
   */
  graduation: Timestamp;

  /**
   * Whether the developer has a LinkedIn account or not.
   */
  hasLinkedIn: boolean;

  /**
   * The URL of the developer's social media/contact information page.
   */
  contactLink: string;

  /**
   * A description of the developer's hobbies.
   */
  hobby: string;

  constructor(devDTO: any) {
    this.firstName = devDTO.firstName;
    this.lastName = devDTO.lastName;
    this.imageName = devDTO.imageName;
    this.imageUrl = devDTO.imageUrl;
    this.degreeInfo = devDTO.degreeInfo;
    this.graduation = devDTO.graduation;
    this.hasLinkedIn = devDTO.hasLinkedIn;
    this.contactLink = devDTO.contactLink;
    this.hobby = devDTO.hobby;
  }

  getDegreeInfoArray(): string[] {
    let separator = '';
    return this.degreeInfo.map((info: DegreeInfo) => {
      if (info.label.toLocaleLowerCase() === 'minor') {
        separator = 'in';
      }
      return `${info.label} ${separator} ${info.content}`;
    });
  }
}

export interface DegreeInfo {
  label: string;
  content: string;
}
