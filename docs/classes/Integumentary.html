/**

export class Cardiovascular {
  // Heart tones

  /**
   * Whether the patient's pulse pattern is irregular (true) or not (false).
   */

  /**
   * The patient's heart tones.
   *
   * Options: murmur or gallup.
   */

  /**
   * The patient's heart rhythm. If the nurse selects "other", they should be able to enter
   * a free form text input into the rhythm field.
   *
   * Options: atrial fibrillation, PVC's, PAC's, SVT, atrial flutter, or other - describe.
   */

  // Edema

  /**
   * Whether there is the presence of JVD (true) or not (false).
   */
  jvd: boolean;

  /**
   * Whether there is the presence of generalized non-pitting edema (true) or not (false).
   */
  gnpe: boolean;

  /**
   * The level of pitting edema in the patient's right hand.
   *
   * Options: 1+, 2+, 3+, or 4+.
   */
  rightHand: string;

  /**
   * The level of pitting edema in the patient's right foot.
   *
   * Options: 1+, 2+, 3+, or 4+.
   */
  rightFoot: string;

  /**
   * The level of pitting edema in the patient's left foot.
   *
   * Options: 1+, 2+, 3+, or 4+.
   */
  leftFoot: string;

  /**
   * The level of pitting edema in the patient's left hand.
   *
   * Options: 1+, 2+, 3+, or 4+.
   */
  leftHand: string;

  // Other

  /**
   * The patient's generalized perfusion.
   *
   * Options: cold, pale, cyanotic, or palpitations.
   */
  generalizedPerfusion: string;

  /**
   * Whether the patient's cap refill is greater than 3 seconds (true) or not (false).
   */
  capRefill: boolean;

  // Pulses

  /**
   * The strength of the pulse in the right upper extremity.
   *
   * Options: +3 strong; +2 normal; +1 weak, thready; or +0 absent.
   */
  rue: string;

  /**
   * The strength of the pulse in the right lower extremity.
   *
   * Options: +3 strong; +2 normal; +1 weak, thready; or +0 absent.
   */
  rle: string;

  /**
   * The strength of the pulse in the left upper extremity.
   *
   * Options: +3 strong; +2 normal; +1 weak, thready; or +0 absent.
   */
  lue: string;

  /**
   * The strength of the pulse in the left lower extremity.
   *
   * Options: +3 strong; +2 normal; +1 weak, thready; or +0 absent.
   */
  lle: string;

  constructor(
    pulsePattern?: boolean,
    heartTones?: string,
    rhythm?: string,
    jvd?: boolean,
    gnpe?: boolean,
    rightHand?: string,
    rightFoot?: string,
    leftFoot?: string,
    leftHand?: string,
    generalizedPerfusion?: string,
    capRefill?: boolean,
    rue?: string,
    rle?: string,
    lue?: string,
    lle?: string
  ) {
    this.pulsePattern = pulsePattern != null ? pulsePattern : false;
    this.heartTones = heartTones ? heartTones : noString;
    this.rhythm = rhythm ? rhythm : noString;
    this.jvd = jvd != null ? jvd : false;
    this.gnpe = gnpe != null ? gnpe : false;
    this.rightHand = rightHand ? rightHand : noString;
    this.rightFoot = rightFoot ? rightFoot : noString;
    this.leftFoot = leftFoot ? leftFoot : noString;
    this.leftHand = leftHand ? leftHand : noString;
    this.generalizedPerfusion = generalizedPerfusion ? generalizedPerfusion : noString;
    this.capRefill = capRefill != null ? capRefill : false;

   */
  rul: string;

  /**
   * The sound of the right lower lobe.
   *

   */

  /**
   * The sound of the right middle lobe.
   *
   * Options: coarse rhonchi, crackles/rales, wheezes expiratory, wheezes inspiratory,
   * or other - describe.
   */
  rml: string;

  /**
   * The sound of the left upper lobe.
   *
   * Options: coarse rhonchi, crackles/rales, wheezes expiratory, wheezes inspiratory,
   * or other - describe.
   */
  lul: string;

  /**
   * The sound of the left lower lobe.
   *
   * Options: coarse rhonchi, crackles/rales, wheezes expiratory, wheezes inspiratory,

   * Whether a cough is present (true) or not (false).

    this.cough = cough ? cough : false;

  }
}

/**
 * Holds information about the patient's musculoskeletal assessment.

export class Musculoskeletal {
  // Fall Risk Assessment Tool

  /**
   * The patient's age range.
   *
   * Options:
   * "60 - 69 years (1 point)",
   * "70 -79 years (2 points)",
   * or "greater than or equal to 80 years (3 points)".
   */
  age: string;

  /**
   * Whether the patient has had "one fall within 6 months before admission" (true)
   * or not (false).
   */
  fallHistory: boolean;

  /**
   * AKA "elimination, bowel and urine."
   *
   * Options:
   * "Incontinence (2 points)",
   * "Urgency or frequency (2 points)",
   * or "Urgency/frequency and incontinence (4 points)".
   */
  ebu: string;

  /**
   * The number of high fall risk drugs the patient is currently on.
   *
   * Options:
   * "On 1 high fall risk drug (3 points)",
   * "On 2 or more high fall risk drugs (5 points)",
   * or "Sedated procedure within past 24 hours (7 points)".
   */
  medications: string;

  /**
   * AKA "patient care equipment." The number of pieces of equipment that tethers
   * the patient.
   *
   * Options:
   * "One present (1 point)",
   * "Two present (2 points)",
   * or "3 or more present (3 points)".
   */
  pce: string; // AKA patient care equipment

  /**
   * The patient's fall risk score based on their mobility.
   *
   * Options:
   * "Requires assistance or supervision for mobility, transfer, or ambulation (2 points)",
   * "Unsteady gait (2 points)", and
   * "Visual or auditory impairment affecting mobility (2 points)".
   */
  mobilityFallRisk: string[];

  /**
   * The patient's fall risk score based on their cognitive ability.
   *
   * Options:
   * "Altered awareness of immediate physical environment (1 point)",
   * "Impulsive (2 points)", and
   * "Lack of understanding of one's physical and cognitive limitations (4 points)".
   */
  cognition: string[];

  /**
   * The patient's fall score based on the scoring provided by the above options.
   */
  fallScore: number;

  /**
   * The mobility/transfers the patient has.
   *
   * Options: standby assist, one assist, two assists, or max assist.
   */
  mobilityAndTransfers: string;

  /**
   * The assistive devices a patient uses for mobility.
   *
   * Options: gait belt, walker, cane, and trapeze.
   */
  assistiveDevices: string[];

  constructor(
    age?: string,
    fallHistory?: boolean,
    ebu?: string,
    medications?: string,
    pce?: string,
    mobilityFallRisk?: string[],
    cognition?: string[],
    fallScore?: number,
    mobilityAndTransfers?: string,
    devices?: string[]
  ) {
    this.age = age ? age : noString;
    this.fallHistory = fallHistory != null ? fallHistory : false;
    this.ebu = ebu ? ebu : noString;
    this.medications = medications ? medications : noString;
    this.pce = pce ? pce : noString;
    this.mobilityFallRisk = mobilityFallRisk ? mobilityFallRisk : [];
    this.cognition = cognition ? cognition : [];
    this.fallScore = fallScore ? fallScore : 0;
    this.mobilityAndTransfers = mobilityAndTransfers ? mobilityAndTransfers : noString;
    this.assistiveDevices = devices ? devices : [];
  }
}

/**
 * Holds information about the patient's nursing assessment.
 *
 * In the sub-assessment categories (cardiovascular, respiratory, etc.), there may be fields with multiple options available.
 * Some of these options are mutually exclusive; some can have multiple options selected. In the sub-assessment classes,
 * multiple-select fields are usually string[] types, and use the word "and" when listing the options in the documentation.
 * Single-select fields are usually non-list types, and use the word "or" when listing the options in the documentation.
 * The other types that use the word "or" are boolean fields: these fields are also "single-select", but only have two options,
 * hence the boolean variable type.
 */
export class Assessment {
  /**
   * The date and time the assessment was created.
   */
  date: Timestamp;

  /**
   * The cardiovascular assessment information.
   */
  cardiovascular: Cardiovascular;

  /**
   * The respiratory assessment information.
   */
  respiratory: Respiratory;

  /**
   * The neurological assessment information.
   */
  neurological: Neurological;

  /**
   * The musculoskeletal assessment information.
   */
  musculoskeletal: Musculoskeletal;

  /**
   * The gastrointestinal assessment information.
   */
  gastrointestinal: Gastrointestinal;

  /**
   * The genitourinary assessment information.
   */
  genitourinary: Genitourinary;

  /**
   * The psychosocial assessment information.
   */
  psychosocial: Psychosocial;

  /**
   * The integumentary assessment information.
   */
  integumentary: Integumentary;

  constructor(
    date?: Timestamp,
    cardiovascular?: Cardiovascular,
    respiratory?: Respiratory,
    neurological?: Neurological,
    musculoskeletal?: Musculoskeletal,
    gastrointestinal?: Gastrointestinal,
    genitourinary?: Genitourinary,
    psychosocial?: Psychosocial,
    integumentary?: Integumentary
  ) {
    this.date = date ? date : Timestamp.now();
    this.cardiovascular = cardiovascular ? cardiovascular : new Cardiovascular();
    this.respiratory = respiratory ? respiratory : new Respiratory();
    this.neurological = neurological ? neurological : new Neurological();
    this.musculoskeletal = musculoskeletal ? musculoskeletal : new Musculoskeletal();
    this.gastrointestinal = gastrointestinal ? gastrointestinal : new Gastrointestinal();
    this.genitourinary = genitourinary ? genitourinary : new Genitourinary();
    this.psychosocial = psychosocial ? psychosocial : new Psychosocial();
    this.integumentary = integumentary ? integumentary : new Integumentary();
  }

  /**
   *
   * @returns The assessment's Timestamp as a date string, formatted as "MM/DD/YYYY @ HH:MM"
   */
  get dateString(): string {
    return this.date.toDate().toLocaleString([], HOUR24_DATE_CONFIG).replace(',', ' @');
  }

  /**
   * Prepares the Assessment's format for Firebase by converting the assessment to JSON,
   * with the JSON object's date being a Timestamp.
   *
   * @returns an Assessment represented as JSON.
   */
  toFirebase() {
    let obj = JSON.parse(JSON.stringify(Object.assign({}, this)));
    obj.date = Timestamp.fromMillis(obj.date.seconds * 1000);

    // TODO: change date in gastrointestinal to Timestamp

    return obj;
  }
}
