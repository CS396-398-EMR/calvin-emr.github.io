import { SimPerson } from "./sim-person.type";
import { MedicalCondition } from "./medical-condition.type";
import { GeneralInfo } from "./general-info.type";
import { HistoryAndPhysical } from './hist-and-phys.type';
import { Timestamp } from "firebase/firestore";
import { Vital } from "./vitals.type";
import { ProgressNote } from './progress-note.type';
import { Assessment } from "./assessment.type";
import { ProviderOrder } from "./order.type";


export class Patient {
  id: string;           // ID is the unique identifier for patient
  mrn: string;          // MRN is the ID of the parent patient; used to reference changes in parent patient
  firstName: string;
  lastName: string;
  middleName: string | null;
  dateCreated: Timestamp;
  currentDate: Timestamp; // used to build all other times around a set time

  // studentPatient identification values
  studentId: string;

  // General info
  generalInfo: GeneralInfo;

  interdisciplinaryTeam: SimPerson[];
  activeMedicalProblems: MedicalCondition[];
  allergies: MedicalCondition[];

  // History and Physical Information
  historyAndPhysical: HistoryAndPhysical;
  // Provider orders
  providerOrders: ProviderOrder[];
  progressNotes: ProgressNote[];

  // Nursing Assessment
  assessments: Assessment[];

  // Vitals
  vitals: Vital[];

  // Profile picture URL
  profilePicName?: string;

  constructor(id?: string, mrn?: string, firstName?: string, lastName?: string, studentId?: string) {
    this.id = id ? id : "";
    this.mrn = mrn ? mrn : "";
    this.firstName = firstName ? firstName : "";
    this.lastName = lastName ? lastName : "";
    this.middleName = "";
    this.dateCreated = Timestamp.fromDate(new Date());
    this.currentDate = Timestamp.fromDate(new Date());
    this.studentId = studentId ? studentId : "";
    this.generalInfo = new GeneralInfo();
    this.historyAndPhysical = new HistoryAndPhysical();
    this.interdisciplinaryTeam = [];
    this.activeMedicalProblems = [];
    this.allergies = [];
    this.providerOrders = [];
    this.progressNotes = [];
    this.assessments = [];
    this.vitals = [];
  }

  getId() {
    return this.id;
  }

  getMRN() {
    return this.mrn;
  }

  getStudentId() {
    return this.studentId;
  }

  getLastFirstMiddleName() {
    return this.middleName ?
      this.lastName + ", " + this.firstName + " " + this.middleName[0] + "." :
      this.lastName + ", " + this.firstName;
  }

  getCurrentDateAsDate(): Date {
    return new Date(this.currentDate.seconds * 1000)
    // return this.currentDate.toDate();
  }

  getCurrentDateAsTimestamp(): Timestamp {
    return this.currentDate;
  }

  getCodeStatus(): string {
    return this.generalInfo.codeStatus.toUpperCase();
  }

  getDOBString(): string {
    return this.generalInfo.dob.toDate().toLocaleDateString();
  }

  /**
   * This calculates age based on UTC, so the age may be off by one depending on
   * the location of the machine the app is being run on.
   * 
   * @returns The user's age calculated from their date of birth.
   */
  getAge(): number {
    const currentDate = new Date();
    const birthDate = this.generalInfo.dob.toDate();

    // Subtract current year from birth year to get the age
    let age = currentDate.getUTCFullYear() - birthDate.getUTCFullYear();

    const monthDiff = currentDate.getUTCMonth() - birthDate.getUTCMonth();

    // Subtract one year if birth month is ahead of current month
    if(monthDiff < 0) {
      age -= 1;
    }
    // Or subtract one year if the birth months are the same, but the birth day is ahead
    //  of the current day
    else if(monthDiff == 0) {
      const dayDiff = currentDate.getUTCDate() - birthDate.getUTCDate();
      if(dayDiff < 0) {
        age -= 1;
      }
    }

    return age;
  }

  getSex() {
    switch (this.generalInfo.sex.toLocaleLowerCase()) {
      case "female": return "Female";
      case "male": return "Male";
      default: return "Other";
    }
  }

  getWeight(): number {
    return this.generalInfo.weight;
  }

  getBMI(): number {
    try {
      let heightMeters = 0.0254 * this.generalInfo.heightInches
      return Math.round(this.generalInfo.weight / Math.pow(heightMeters, 2) * 100) / 100; // to get 2 decimal precision
    } catch (err) {
      console.error(err)
      return -1;
    };
  }

  getIsolation(): string {
    return this.generalInfo.isolation;
  }

  getHeightString(): string {
    let feet = Math.floor(this.generalInfo.heightInches / 12);
    let inches = this.generalInfo.heightInches % 12;
    return "" + feet + "\'" + inches + "\"";
  }

  // start getters for history and physical information
  getAllergies(): string { return this.historyAndPhysical.allergies; }
  getCardiovascular(): string { return this.historyAndPhysical.cardiovascular; }
  getChiefComplaint(): string { return this.historyAndPhysical.chiefComplaint; }
  getContagiousDiseases(): string { return this.historyAndPhysical.contagiousDiseases; }
  getCurrentMeds(): string { return this.historyAndPhysical.currentMeds; }
  getDevelopmental(): string { return this.historyAndPhysical.developmental; }
  getDietHistory(): string { return this.historyAndPhysical.dietHistory; }
  getEndocrine(): string { return this.historyAndPhysical.endocrine; }
  getFamilyHistory(): string { return this.historyAndPhysical.familyHistory; }
  getGastrointestinal(): string { return this.historyAndPhysical.gastrointestinal; }
  getGenitoUrinary(): string { return this.historyAndPhysical.genitourinary; }
  getHEENT(): string { return this.historyAndPhysical.heent; }
  getHistOfCurrProb(): string { return this.historyAndPhysical.histOfCurrProb; }
  getImpressions(): string { return this.historyAndPhysical.impressions; }
  getImpressionsDay(): string { return this.historyAndPhysical.impressionsDay; }
  getInformant(): string { return this.historyAndPhysical.informant; }
  getInjuries(): string { return this.historyAndPhysical.injuries; }
  getIntegument(): string { return this.historyAndPhysical.integument; }
  getLymphatic(): string { return this.historyAndPhysical.lymphatic; }
  getMusculoskeletal(): string { return this.historyAndPhysical.musculoskeletal; }
  getNeurologic(): string { return this.historyAndPhysical.neurologic; }
  getOtherPastMedHist(): string { return this.historyAndPhysical.otherPastMedHist; }
  getPlan(): string { return this.historyAndPhysical.plan; }
  getPreviousIllnesses(): string { return this.historyAndPhysical.previousIllnesses; }
  getProvSignature(): string { return this.historyAndPhysical.provSignature; }
  getRespiratory(): string { return this.historyAndPhysical.respiratory; }
  getSocialHistory(): string { return this.historyAndPhysical.socialHistory; }
  getSurgicalHistory(): string { return this.historyAndPhysical.surgicalHistory; }
  // getters for physical examination portion of hist and phys
  getPECardiovascular(): string { return this.historyAndPhysical.peCardiovascular; }
  getPEDevelopmental(): string { return this.historyAndPhysical.peDevelopmental; }
  getPEEndocrine(): string { return this.historyAndPhysical.peEndocrine; }
  getPEGastrointestinal(): string { return this.historyAndPhysical.peGastrointestinal; }
  getPEGeneral(): string { return this.historyAndPhysical.peGeneral; }
  getPEGenitourinary(): string { return this.historyAndPhysical.peGenitourinary; }
  getPEHEENT(): string { return this.historyAndPhysical.peHEENT; }
  getPELymphatic(): string { return this.historyAndPhysical.peLymphatic; }
  getPEMusculoskeletal(): string { return this.historyAndPhysical.peMusculoskeletal; }
  getPENeurologic(): string { return this.historyAndPhysical.peNeurologic; }
  getPERespiratory(): string { return this.historyAndPhysical.peRespiratory; }
  getPESkin(): string { return this.historyAndPhysical.peSkin; }
  getPEVitalSigns(): string { return this.historyAndPhysical.peVitalSigns; }
  // end getters for hist and phy info

  getProfilePicURL(): string {
    return this.profilePicName ? 
      this.id + "/" + this.profilePicName :
      "blank-profile.webp";
  }

  setId(id: string): void {
    this.id = id;
  }

  setStudentId(sid: string): void {
    this.studentId = sid;
  }

  setMRN(mrn: string): void {
    this.mrn = mrn;
  }

  setDOB(date: Timestamp): void {
    this.generalInfo.dob = date;
  }

  setHeight(heightInches: number): void {
    this.generalInfo.heightInches = heightInches;
  }

  /**
   * setProfilePicName
   *
   * @param imgName name of the image used for the user's profile picture
   *  in Firebase, image paths are of the form <id>/profile.<jpg|png|gif>
   *
   */
  setProfilePicName(imgName: string) {
    this.profilePicName = imgName;
  }

  toJSObject(): void {
    this.providerOrders = this.providerOrders.map((providerOrder) => {
      return new ProviderOrder(
        providerOrder.id,
        providerOrder.medId,
        providerOrder.medName,
        providerOrder.route,
        providerOrder.dosage,
        providerOrder.quantity,
        providerOrder.quantityGiven,
        providerOrder.order,
        providerOrder.signature,
        providerOrder.startDate,
        providerOrder.endDate,
        providerOrder.dateCreated);
    })

    this.progressNotes = this.progressNotes.map((note) => {
      return new ProgressNote(
        note.date,
        note.notes,
        note.signature,
        note.deleted);
    })

    this.vitals = this.vitals.map((vital) => {
      return new Vital(
        vital.date,
        vital.heartRate,
        vital.oxySat,
        vital.oxygen,
        vital.respirations,
        vital.temp,
        vital.systolic,
        vital.diastolic,
        vital.deleted);
    })
  }

  /**
   * Prepares the Vital's format for Firebase by converting the vital to an object, 
   * with the object's date being a Timestamp.
   * 
   * @returns a Vital represented as JSON.
   */
  toFirebase() {
    let obj = JSON.parse(JSON.stringify(Object.assign({}, this)))

    // Convert {seconds, nanoseconds} object produced by JSON.parse to Timestamp object
    obj.dateCreated = Timestamp.fromMillis(obj.dateCreated.seconds * 1000);
    obj.currentDate = Timestamp.fromMillis(obj.currentDate.seconds * 1000);

    // Call toFirebase() in contained classes
    this.datesToFirebase(obj.assessments);
    this.ordersToFirebase(obj.providerOrders);
    this.datesToFirebase(obj.progressNotes);
    this.datesToFirebase(obj.vitals);

    return obj;
  }

  // add assessmentsToFirebase to handle gastrointestinal dates

  datesToFirebase(objs: any[]) {
    objs = objs.map((obj) => {
      if(obj.date) {
        obj.date = Timestamp.fromMillis(obj.date.seconds * 1000);
      }
    })
  }

  ordersToFirebase(orders: any[]) {
    orders = orders.map((order) => {
      if(order.startDate && order.endDate && order.dateCreated) {
        order.startDate = Timestamp.fromMillis(order.startDate.seconds * 1000);
        order.endDate = Timestamp.fromMillis(order.endDate.seconds * 1000);
        order.dateCreated = Timestamp.fromMillis(order.dateCreated.seconds * 1000);
      }
    })
  }
}
