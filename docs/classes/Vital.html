import { Timestamp } from "@firebase/firestore";

/**
 * Holds information about patient vitals.
 */
export class Vital {
  /**
   * The date that the vital was created. Also used as the primary for the vital.
   */
  date: Timestamp;

  /**
   * Heart rate in beats per minute (bpm).
   */
  heartRate: number;

  /**
   * Oxygen saturation in percentage (0-100%).
   */
  oxySat: number;

  oxygen: string;

  /**
   * The number of respirations per minute.
   */
  respirations: number;

  /**
   * Temperature in degrees Celcius.
   */
  temp: number;

  /**
   * Systolic blood pressure.
   */
  systolic: number;

  /**
   * Diastolic blood pressure
   */
  diastolic: number;

  /**
   * Flag indicating whether the vital was deleted by the user or not.
   */
  deleted: boolean;

  constructor(date?: Timestamp, heartRate?: number, oxySat?: number, oxygen?: string, respirations?: number, temp?: number, systolic?: number, diastolic?: number, deleted?: boolean) {
    this.date = date ? date : Timestamp.fromDate(new Date());
    this.heartRate = heartRate ? heartRate : 0;
    this.oxySat = oxySat ? oxySat : 0;
    this.oxygen = oxygen ? oxygen : '';
    this.respirations = respirations ? respirations : 0;
    this.temp = temp ? temp : 0;
    this.systolic = systolic ? systolic : 0;
    this.diastolic = diastolic ? diastolic : 0;
    this.deleted = deleted != null ? deleted : false;
  }

  /**
   * Prepares the Vital's format for Firebase by converting the vital to a JSON object, 
   * with the JSON object's date being a Timestamp.
   * 
   * @returns a Vital represented as JSON.
   */
  toFirebase() {
    let obj = JSON.parse(JSON.stringify(Object.assign({}, this)))
    obj.date = Timestamp.fromMillis(obj.date.seconds * 1000);

    return obj;
  }

  /**
   * Used to display different color for heart rates not in normal range
   * (see the emr-table component for more details).
   * Normal range: 40-100 bpm.
   * 
   * @returns whether the heart rate is within a normal range.
   */
  isHeartRateWithinNormalRange(): boolean {
    return this.heartRate >= 40 && this.heartRate <= 100;
  }

  /**
   * Used to display different color for oxygen saturations not in normal range
   * (see the emr-table component for more details).
   * Normal range: 92-100%.
   * 
   * @returns whether the oxygen saturation is within a normal range.
   */
  isOxySatWithinNormalRange(): boolean {
    return this.oxySat >= 92 && this.oxySat <= 100;
  }

  /**
   * Used to display different color for respirations not in normal range
   * (see the emr-table component for more details).
   * Normal range: 12-25 respirations.
   * 
   * @returns whether the number of respirations is within a normal range.
   */
  isRespirationsWithinNormalRange(): boolean {
    return this.respirations >= 12 && this.respirations <= 25;
  }

  /**
   * Used to display different color for temperatures not in normal range
   * (see the emr-table component for more details).
   * Normal range: 36.1-37.2 °C.
   * 
   * @returns whether the temperature is within a normal range.
   */
  isTempWithinNormalRange(): boolean {
    return this.temp >= 36.1 && this.temp <= 37.2;
  }

  /**
   * Used to display different color for systolic blood pressures not in normal range
   * (see the emr-table component for more details).
   * Normal range: below 180.
   * 
   * @returns whether the systolic blood pressure is within a normal range.
   */
  isSystolicWithinNormalRange(): boolean {
    return this.systolic < 180;
  }

  /**
   * Used to display different color for diastolic blood pressures not in normal range
   * (see the emr-table component for more details).
   * Normal range: 60-100.
   * 
   * @returns whether the diastolic blood pressure is within a normal range.
   */
  isDiastolicWithinNormalRange(): boolean {
    return this.diastolic >= 60 && this.diastolic <= 100;
  }
}
