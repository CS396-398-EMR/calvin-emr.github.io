import { Timestamp } from "firebase/firestore";
import { hour24DateOptions, hour24TimeOptions } from "../global";

class Order {
  /**
   * The order's unique identifier.
   */
  id: string;

  /**
   * The ID of the medication ordered.
   */
  medId: string;

  /**
   * The name of the medication ordered.
   */
  medName: string;

  /**
   * The route of administration for the medication ordered.
   */
  route: string;

  /**
   * The dosage of the medication ordered. This should be a multiple of the
   * medication's single dose amount (which is found in the medication's name).
   * As of the time of writing, dosage has no functionality within the MAR or
   * Provider Orders - it is merely a display value.
   */
  dosage: string;

  /**
   * The number of times a nurse has to administer a medication. This is the
   * number of times a nurse has to scan a medication's barcode to fully
   * administer it.
   */
  quantity: number;

  /**
   * The number of times a nurse has administered a medication. This can be
   * greater than the quantity ordered.
   */
  quantityGiven: number;

  /**
   * Additional notes or instructions about the order.
   */
  order: string;

  /**
   * The signature of the provider who ordered the medication.
   */
  signature: string;

  // givenDate?: Date;            // date/time when the med was given to the patient
  
  // recurrence?: string;

  /**
   * The date the order was created.
   */
  dateCreated: Timestamp;

  constructor(id?: string, medId?: string, medName?: string, route?: string, dosage?: string, quantity?: number, 
    quantityGiven?: number, order?: string, signature?: string, dateCreated?: Timestamp) {
    this.id = id ? id : "";
    this.medId = medId ? medId : "";
    this.medName = medName ? medName : "";
    this.route = route ? route : "";
    this.dosage = dosage ? dosage : "";
    this.quantity = quantity ? quantity : 0;
    this.quantityGiven = quantityGiven ? quantityGiven : 0;
    this.order = order ? order : "";
    this.signature = signature ? signature : "";
    this.dateCreated = dateCreated ? dateCreated : Timestamp.now();
    // this.givenDate = givenDate ? givenDate : new Date(); // might not need this
  }

  /**
   * @returns The ID of the order's medication.
   */
  getMedId() {
    return this.medId;
  }

  /**
   * @returns Whether the medication has been fully/excessively administered or not.
   */
  get administered() {
    return this.quantityGiven >= this.quantity;
  }

  getAdministeredString() {
    if(this.quantityGiven > this.quantity) { return 'Overdosed'; }
    else if(this.quantityGiven == this.quantity) { return 'Yes'; }
    else if(this.quantityGiven > 0 && this.quantityGiven < this.quantity) { return 'Partial'; }
    else { return 'No'; }
  }
}

export class ProviderOrder extends Order {
  /**
   * The scheduled date and time on when to administer the medication.
   */
  startDate: Timestamp;

  /**
   * The last date and time on when to administer the medication.
   */
  endDate: Timestamp;

  constructor(id?: string, medId?: string, medName?: string, route?: string, dosage?: string, quantity?: number, 
    quantityGiven?: number, order?: string, signature?: string, startDate?: Timestamp, endDate?: Timestamp,
    dateCreated?: Timestamp) {
    super(id, medId, medName, route, dosage, quantity, quantityGiven, order, signature, dateCreated);
    this.startDate = startDate ? startDate : Timestamp.now();
    this.endDate = endDate ? endDate : Timestamp.now();
  }

  /**
   * Prepares the Order's format for Firebase by converting the order to a JSON object, 
   * with the JSON object's dates being Timestamps.
   * 
   * @returns an Order represented as JSON.
   */
  toFirebase() {
    let obj = JSON.parse(JSON.stringify(Object.assign({}, this)))
    obj.startDate = this.startDate;
    obj.endDate = this.endDate;
    obj.dateCreated = this.dateCreated;
    return obj;
  }

  /**
   * @returns A date as string, formatted as 'DD/MM/YYYY @ HH:MM' (24-hour time).
   */
  getDateString(separator: string) {
    return this.startDate.toDate().toLocaleString([], hour24DateOptions).replace(',', ' ' + separator);
  }
}


export class MAROrder extends Order {
  /**
   * The scheduled date and time on when to administer the medication.
   */
  startDate: Date;

  /**
   * The last date and time on when to administer the medication.
   */
  endDate: Date;

  constructor(id?: string, medId?: string, medName?: string, route?: string, dosage?: string, quantity?: number, 
    quantityGiven?: number, order?: string, signature?: string, startDate?: Date, endDate?: Date,
    dateCreated?: Timestamp) {
    super(id, medId, medName, route, dosage, quantity, quantityGiven, order, signature, dateCreated);
    this.startDate = startDate ? startDate : new Date();
    this.endDate = endDate ? endDate : new Date();
  }

  /**
   * Prepares the Order's format for Firebase by converting the order to a JSON object, 
   * with the JSON object's dates being Timestamps. Converts startDate and endDate from
   * Date to Timestamp objects.
   * 
   * @returns an Order represented as JSON.
   */
  toFirebase() {
    let obj = JSON.parse(JSON.stringify(Object.assign({}, this)))
    obj.startDate = Timestamp.fromDate(this.startDate);
    obj.endDate = Timestamp.fromDate(this.endDate);
    obj.dateCreated = this.dateCreated;

    return obj;
  }

  /**
   * @returns A date as string, formatted as 'DD/MM/YYYY @ HH:MM' (24-hour time).
   */
  getDateString(separator: string) {
    return this.startDate.toLocaleString([], hour24DateOptions).replace(',', ' ' + separator);
  }

  get startTimeString(): string {
    return this.startDate.toLocaleString([], hour24TimeOptions);
  }

  get endTimeString(): string {
    return this.endDate.toLocaleString([], hour24TimeOptions);
  }


}
