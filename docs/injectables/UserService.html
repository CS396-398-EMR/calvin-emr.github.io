import { Injectable } from '@angular/core';
import { AngularFirestore } from '@angular/fire/compat/firestore';
import { getFirestore, collection, query, where, getDocs, doc, updateDoc, arrayUnion } from '@angular/fire/firestore'
import { User } from '../type/user.type';
import firebase from 'firebase/compat/app';
import 'firebase/compat/firestore';
import 'firebase/compat/auth';
import 'firebase/compat/functions';
import { arrayRemove } from 'firebase/firestore';


@Injectable({
  providedIn: 'root'
})
export class UserService {

  userCollection: string = "users"
  currentUser: User = new User();

  constructor(private db: AngularFirestore) { }

  setUser(id: string) {
    this.db.collection(this.userCollection).doc(id).snapshotChanges().subscribe(
      (action) => {
        let obj = { id: action.payload.id, ...action.payload.data() as object };
        Object.assign(this.currentUser, obj)
      },
      (err) => {
        console.error("Error loading user", id, ":", err);
      }
    )
  }

  // Temporary function until Won-Seok understands how addPatientId() works
  tempAddPatientId(id: string, userId: string){
    const updateRef = this.db.collection(this.userCollection).doc(userId);
    updateRef.update({
      patientIds: arrayUnion(id)
    });
  }

  addPatientId(id: string) {
    const updateRef = this.db.collection(this.userCollection).doc(this.currentUser?.getId());
    updateRef.update({
      patientIds: arrayUnion(id)
    });
  }

  deletePatientId(id: string) {
    const updateRef = this.db.collection(this.userCollection).doc(this.currentUser?.getId());
    updateRef.update({
      patientIds: arrayRemove(id)
    })
  }

  // Read
  getUser(id: string) {
    return this.db
      .collection(this.userCollection)
      .doc(id)
      .snapshotChanges();
  }

  getCurrentUserRole() {
    return this.currentUser.role;
  }

  isCurrentUserAdmin() {
    return this.currentUser.role.toLocaleLowerCase() == "admin";
  }

  // Returns query snapshot of all "student" role user in the database
  async getAllUsers() {
    const db = getFirestore();

    // query(collectionReference, condition)
    const q = query(collection(db, "users"), where("role", "==", "student"));

    const querySnapshot = await getDocs(q);

    return querySnapshot
  }

  async setPatient(id: string, data: any){
    const db = getFirestore();
    
    const ref = doc(db, "users", id);

    await updateDoc(ref, data);
  }

  // Admin create user
  createUser(user: User) {
    console.log(user)
    const createUser = firebase.functions().httpsCallable('createUser');
    createUser({
      user: user,
    })
  }
}
