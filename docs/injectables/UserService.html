import { Injectable } from '@angular/core';
import { AngularFirestore } from '@angular/fire/compat/firestore';
import { getFirestore, collection, query, where, getDocs, arrayUnion } from '@angular/fire/firestore';
import { User } from '../models/user.model';
import firebase from 'firebase/compat/app';
import 'firebase/compat/firestore';
import 'firebase/compat/auth';
import 'firebase/compat/functions';
import { arrayRemove } from 'firebase/firestore';
import { ROLE_STUDENT, COLLECTION_USERS } from '../../global';
import { AuthService } from './auth.service';
import { from, Observable } from 'rxjs';

@Injectable({
  providedIn: 'root',
})
export class UserService {
  /**
   * Since _currentUser is a ReplaySubject, allow subscribers to cache the last
   * value emitted.
   * Set the currentUser$ Observable to only emit a new User if the current user's
   * information (e.g. first name, last name, email) has changed.
   *
   * @param db The Firestore instance used to call REST operations.
   */
  constructor(private db: AngularFirestore, private authService: AuthService) {}

  /**
   * Adds the provided patient ID to the provided user's document.
   *
   * @param patientId The patient ID to add to the patientIds array.
   * @param userId The ID of the user document to add the patient ID to.
   * @returns Promise
   */
  async addPatientIdToUser(patientId: string, userId: string) {
    const updateRef = this.db.collection(COLLECTION_USERS).doc(userId);
    return updateRef.update({
      patientIds: arrayUnion(patientId),
    });
  }

  /**
   * Iterates through all user documents that contain the patient ID
   * and deletes the patient ID from the document's 'patientIds' field.
   *
   * @param id The patient ID to remove from all user documents.
   * @returns Promise
   */
  async deletePatientIdFromAllUsers(id: string) {
    return await this.db
      .collection(COLLECTION_USERS)
      .ref.where('patientIds', 'array-contains', id)
      .get()
      .then((snapshot) => {
        snapshot.docs.forEach((doc) => {
          doc.ref.update({
            patientIds: arrayRemove(id),
          });
        });
      });
  }

  // Returns query snapshot of all "student" role user in the database
  async getAllStudents() {
    const db = getFirestore();
    const q = query(collection(db, COLLECTION_USERS), where('role', '==', ROLE_STUDENT));
    return await getDocs(q);
  }

  /**
   * Calls a Firebase cloud function to create a user auth account.
   *
   * @param user The user to create in Firestore.
   */
  createUserAccount(user: User) {
    const createUserAccount = firebase.functions().httpsCallable('createUserAccount');
    return from(createUserAccount({ user: user }));
  }

  /**
   * Creates a document for the provided user in Firestore.
   *
   * @param user The user to create a document for in Firestore.
   */
  createUserDocument(user: User) {
    this.cleanUserData(user);
    let obj = this.extractUserData(user);

    return from(this.db.collection(COLLECTION_USERS).doc(user.id).set(obj));
  }

  /**
   * Prepares the User object for Firestore before updating it
   * in the backend.
   *
   * @param user The User object to update in Firestore.
   */
  updateUser(user: User) {
    this.cleanUserData(user);
    let obj = this.extractUserData(user);
    return this.db.collection(COLLECTION_USERS).doc(user.id).set(obj);
  }

  /**
   * @param id The ID of the user to delete from Firestore.
   */
  deleteUser(id: string) {
    const deleteUser = firebase.functions().httpsCallable('deleteUser');
    return deleteUser({
      id: id,
    });
  }

  /**
   * Removes the ID parameter from the User object to prepare it for Firestore.
   *
   * @param user The User object to remove the ID from.
   * @returns A copy of the User object with the ID parameter removed.
   */
  extractUserData(user: User): object {
    // Convert User object to a JSON object
    let obj = JSON.parse(JSON.stringify(Object.assign({}, user)));
    // Do this since the ID will be the document name
    delete obj.id;

    return obj;
  }

  /**
   * Cleans the User object by ensuring no whitespaces are on either end
   * of the properties, and that there are no spaces in the ID.
   *
   * @param user The User object to clean the data for.
   */
  cleanUserData(user: User) {
    user.id = user.id.replace(/\s/g, '');
    user.firstName = user.firstName.trim();
    user.lastName = user.lastName.trim();
  }
}
