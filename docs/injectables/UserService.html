import { Injectable } from '@angular/core';
import { AngularFirestore } from '@angular/fire/compat/firestore';
import { getFirestore, collection, query, where, getDocs, doc, updateDoc, arrayUnion } from '@angular/fire/firestore'
import { User } from '../types/user.type';
import firebase from 'firebase/compat/app';
import 'firebase/compat/firestore';
import 'firebase/compat/auth';
import 'firebase/compat/functions';
import { arrayRemove, getDoc } from 'firebase/firestore';
import { adminRole, studentRole, usersCollectionName } from '../global';
import { BehaviorSubject, Observable, ReplaySubject, Subscription } from 'rxjs';
import { distinctUntilChanged } from 'rxjs/operators';


@Injectable({
  providedIn: 'root'
})
export class UserService {

  currentUser: User = new User();

  private _currentUser: ReplaySubject<User>;
  currentUser$: Observable<User>;
  currentUserSub?: Subscription;

  /**
   * Since _currentUser is a ReplaySubject, allow subscribers to cache the last
   * value emitted.
   * Set the currentUser$ Observable to only emit a new User if the current user's
   * information (e.g. first name, last name, email) has changed.
   * 
   * @param db The Firestore instance used to call REST operations.
   */
  constructor(private db: AngularFirestore) { 
    this._currentUser = new ReplaySubject<User>(1);
    this.currentUser$ = this._currentUser.asObservable().pipe(
      distinctUntilChanged(this.userCompare)
    );
  }

  /**
   * Called by AuthService when the user logs in (or if the user is already logged in).
   * Emits the user to all subscribers (e.g. guards, main-nav component) for the subscribers
   * to use.
   * 
   * TODO: The subscription in this function will persist for the lifetime of the application.
   * How can we unsubscribe from it? 
   * 
   * @param id The ID of the user to get from the "users" Firestore collection.
   */
  setUser(id: string) {
    this.currentUserSub = this.db.collection(usersCollectionName).doc(id).snapshotChanges().subscribe(
      (action) => {
        let obj = { id: action.payload.id, ...action.payload.data() as object };
        Object.assign(this.currentUser, obj)
        this.emitUser();
      },
      (err) => {
        console.error("Error loading user", id, ":", err);
      }
    )
  }
  
  unsetUser() {
    this.currentUserSub?.unsubscribe();
  }

  // Temporary function until Won-Seok understands how addPatientId() works
  tempAddPatientId(id: string, userId: string){
    const updateRef = this.db.collection(usersCollectionName).doc(userId);
    updateRef.update({
      patientIds: arrayUnion(id)
    });
  }

  addPatientId(id: string) {
    const updateRef = this.db.collection(usersCollectionName).doc(this.currentUser?.getId());
    updateRef.update({
      patientIds: arrayUnion(id)
    });
  }

  deletePatientId(id: string) {
    const updateRef = this.db.collection(usersCollectionName).doc(this.currentUser?.getId());
    updateRef.update({
      patientIds: arrayRemove(id)
    })
  }

  /**
   * Retrieves a user's data one time. This function allows us to get data without
   * subscribing to a data source, allowing for less need to clean up subscriptions.
   * 
   * @param id The ID of the user document to retrieve.
   * @returns Promise to load Patient data.
   */
   async getStaticUser(id: string): Promise<User | null> {
    const db = getFirestore();
    const docRef = doc(db, usersCollectionName, id);
    const docSnap = await getDoc(docRef);

    if (docSnap.exists()) {
      let user = new User();
      Object.assign(user, docSnap.data());

      // Format user data
      user.id = docSnap.id;

      return user;
    } else {
      return null;
    }
  }


  // Returns query snapshot of all "student" role user in the database
  async getAllUsers() {
    const db = getFirestore();

    // query(collectionReference, condition)
    const q = query(collection(db, usersCollectionName), where("role", "==", studentRole));

    const querySnapshot = await getDocs(q);

    return querySnapshot
  }

  async setPatient(id: string, data: any){
    const db = getFirestore();
    
    const ref = doc(db, usersCollectionName, id);

    await updateDoc(ref, data);
  }


  /**
   * @param user The User object to remove the ID from.
   * @returns A copy of the User object with the ID parameter removed.
   */
   extractUserData(user: User): object {
    // Convert User object to a JSON object
    let obj = JSON.parse(JSON.stringify(Object.assign({}, user)));
    // Do this since the ID will be the document name
    delete obj.id;

    return obj;
  }

  /**
   * Cleans the User object by ensuring no whitespaces are on either end
   * of the properties, and that there are no spaces in the ID.
   * 
   * @param user The User object to clean the data for.
   */
  cleanUserData(user: User) {
    user.id = user.id.replace(/\s/g, "");
    user.firstName = user.firstName.trim();
    user.lastName = user.lastName.trim();
  }

  /**
   * @param user The user to create in Firestore.
   */
  createUser(user: User) {
    console.log(user)
    const createUser = firebase.functions().httpsCallable('createUser');
    createUser({
      user: user,
    })
  }

  /**
   * Prepares the User object for Firestore before updating it
   * in the backend.
   * 
   * @param user The User object to update in Firestore.
   */
  updateUser(user: User) {
    this.cleanUserData(user);
    let obj = this.extractUserData(user);

    try {
      this.db.collection(usersCollectionName)
        .doc(user.id)
        .set(obj);
    } catch (e) {
      console.log(e);
    }
  }

  /**
   * @param id The ID of the user to delete from Firestore.
   */
  deleteUser(id: string) {
    const deleteUser = firebase.functions().httpsCallable('deleteUser');
    deleteUser({
      id: id,
    })
  }

  /**
   * Emit the current User object to all subscribers.
   */
  emitUser() {
    this._currentUser.next(this.currentUser);
  }

  /**
   * @returns Whether or not a user is signed in to the app.
   */
  get hasUser() {
    return this.currentUser != null;
  }

  get isCurrentUserAdmin() {
    return this.currentUser.role.toLocaleLowerCase() == adminRole;
  }

  /**
   * Compares two User objects and determines if they are the same regarding certain
   * properties.
   * 
   * @param a First User to compare properties of.
   * @param b Second User to compare properties of.
   * @returns If the two User objects are the same in ID, email, first name,
   * last name, and role.
   */
  private userCompare(a: User, b: User) {
    const idCheck = a.id == b.id;
    const emailCheck = a.email == b.email;
    const firstNameCheck = a.firstName == b.firstName;
    const lastNameCheck = a.lastName == b.lastName;
    const roleCheck = a.role == b.role;

    return idCheck && emailCheck && firstNameCheck && lastNameCheck && roleCheck;
  }
}
