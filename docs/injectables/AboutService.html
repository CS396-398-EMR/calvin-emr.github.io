import { Injectable } from '@angular/core';
import { collection, doc, getDoc, getDocs, getFirestore } from 'firebase/firestore';
import { getDownloadURL, getStorage, ref } from 'firebase/storage';
import { aboutUsPicturesPath, developersCollectionName, profilePicLoadingError } from '../global';
import { DeveloperProfile } from '../types/dev-profile.type';

@Injectable({
  providedIn: 'root'
})
export class AboutService {

  constructor() { }

  /**
   * Retrieves a developer's data one time. This function allows us to get data without
   * subscribing to a data source, allowing for less need to clean up subscriptions.
   * 
   * @param id The ID of the developer document to retrieve.
   * @returns Promise to load developer data.
   */
  async getDeveloper(id: string): Promise<DeveloperProfile | null> {
    // Set up and execute query
    const db = getFirestore();
    const docRef = doc(db, developersCollectionName, id);
    const docSnap = await getDoc(docRef);

    // Return result of query
    if(docSnap.exists()) {
      let dev = new DeveloperProfile();
      Object.assign(dev, docSnap.data());
      return dev;
    }
    else {
      return null;
    }
  }
  
  /**
   * Retrieves all developer data one time. This function allows us to get data without
   * subscribing to a data source, allowing for less need to clean up subscriptions.
   * 
   * @returns Promise to load all developer data.
   */
  async getDevelopers(): Promise<DeveloperProfile[]> {
    // Set up query
    const db = getFirestore();
    const collRef = collection(db, developersCollectionName);
    const collSnap = await getDocs(collRef);

    // Map each document to a DeveloperProfile object, and return the array
    let devs = collSnap.docs.map((doc) => {
      let dev = new DeveloperProfile();
      let obj = { id: doc.id, ...doc.data() as object };
      Object.assign(dev, obj)

      // Get image URL for dev profile
      this.setURL(dev);

      return dev;
    })

    return devs;
  }

  /**
   * Sets the imageURL property on the DeveloperProfile object passed to
   * the function. This function is specified to be async because it needs
   * to wait for the getDownloadURL Promise to set the passed 
   * DeveloperProfile object's URL before completing.
   * 
   * @param dev The DeveloperProfile to get the image URL for.
   */
  private async setURL(dev: DeveloperProfile) {
    const storage = getStorage();

    // Retrieve url from the specified path
    await getDownloadURL(ref(storage, `${aboutUsPicturesPath}/${dev.imageName}`))
      // If file exists set imageURL to returned url
      .then((url) => {
        dev.imageURL = url;
      })
      .catch((error) => {
        console.error(profilePicLoadingError, error);
      });
  }
}
