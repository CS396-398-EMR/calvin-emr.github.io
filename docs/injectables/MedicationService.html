import { Injectable } from '@angular/core';
import { AngularFirestore } from '@angular/fire/compat/firestore';
import { medicationsCollectionName } from '../global';
import { Medication } from '../types/medication.type';

@Injectable({
  providedIn: 'root'
})
export class MedicationService {

  constructor(private db: AngularFirestore) { }

  /**
   * 
   * @param med The Medication object to remove the ID from.
   * @returns A copy of the Medication object with the ID parameter removed.
   */
  extractMedData(med: Medication): object {
    // Convert Medication object to a JSON object
    let obj = JSON.parse(JSON.stringify(Object.assign({}, med)));
    // Do this since the ID will be the document name
    delete obj.id;

    return obj;
  }

  /**
   * Cleans the Medication object by ensuring no whitespaces are on either end
   * of the properties, and that there are no spaces in the ID.
   * 
   * @param med The Medication object to clean the data for.
   */
  cleanMedData(med: Medication) {
    med.id = med.id.replace(/\s/g, "");
    med.name = med.name.trim();
    med.routes.forEach((route) => route.trim());
  }

  /**
   * Gets a medication snapshot for a single med.
   * 
   * @param id The id of the medication to get.
   * @returns A snapshot of the medication.
   */
  getMedication(id: string) {
    return this.db.collection(medicationsCollectionName).doc(id).get()
  }

  /**
   * 
   * @returns A snapshot of the entire Medications collection.
   */
  getMedications() {
    return this.db.collection(medicationsCollectionName).snapshotChanges()
  }

  /**
   * Prepares the Medication object for Firestore before creating it
   * in the backend.
   * 
   * @param med Medication to create in Firestore.
   */
  createMedication(med: Medication) {
    this.cleanMedData(med);
    let obj = this.extractMedData(med);

    // Use "set" method, since we separate the ID from the data; ID
    // is used as the document ID
    this.db.collection(medicationsCollectionName)
      .doc(med.id)
      .set(obj)
      .catch(e => {
          console.log(e);
      });
  }

  /**
   * Prepares the Medication object for Firestore before updating it
   * in the backend.
   * 
   * @param med The Medication object to update in Firestore.
   */
  updateMedication(med: Medication) {
    this.cleanMedData(med);
    let obj = this.extractMedData(med);

    try {
      this.db.collection(medicationsCollectionName)
        .doc(med.id)
        .set(obj);
    } catch (e) {
      console.log(e);
    }
  }

  /**
   * 
   * @param id The ID of the medication to delete from Firestore.
   */
  deleteMedication(id: string) {
    try {
      this.db.collection(medicationsCollectionName)
        .doc(id)
        .delete();
    } catch (e) {
      console.log(e);
    }
  }
}
