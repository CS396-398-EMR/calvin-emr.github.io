import { Injectable, NgZone } from '@angular/core';
import { Router } from '@angular/router';

// New angular and firebase require compat layer to follow old tutorials
import { AngularFirestore } from '@angular/fire/compat/firestore'
import {
  getAuth,
  signInWithEmailAndPassword,
  createUserWithEmailAndPassword,
  signOut,
  onAuthStateChanged,
  setPersistence,
  browserSessionPersistence,
  getAdditionalUserInfo,
  UserCredential,
  browserLocalPersistence
} from 'firebase/auth'
import { UserService } from './user.service';


@Injectable({
  providedIn: 'root'
})

// code ref: https://www.positronx.io/protect-angular-2-routes-with-canactivate-guard-for-firebase-users/

export class AuthService {

  // ------ modified code below from https://angular.io/guide/router-tutorial-toh#milestone-5-route-guards -------
  isLoggedIn: boolean;

  // store the URL so we can redirect after logging in
  redirectUrl: string | null = null;

  constructor(private router: Router, private db: AngularFirestore, private userService: UserService) { 
    this.isLoggedIn = false;
  }

  // TODO: modify login() so it's dynamic with Firebase, etc.
  login(email: string, password: string) {
    const auth = getAuth();

    // Don't use this??
    const result = signInWithEmailAndPassword(auth, email, password);

    // // Set firebase to remember auth until browser close
    // setPersistence(auth, browserLocalPersistence)
    //   .then(() => {
    //     return result;
    //   })

    result.then((value: UserCredential) => {
      this.isLoggedIn = true;
      this.userService.setUser(value.user.uid);

      // If this is user's first sign-in, route to change-password component.
      // Otherwise, route to their patient list.
      if(auth.currentUser?.metadata.lastSignInTime) {
        this.router.navigateByUrl('/patients');
      }
      else {
        this.router.navigateByUrl('/change-password')
      }
    })
    .catch(err => {
      console.log('Something went wrong: ', err.message);
    });

    return result;
  }

  getUser(): any {
    const auth = getAuth();

    // Retrieve user's uid to use it for creating user document
    onAuthStateChanged(auth, (user) => {
      if (user) {
        this.isLoggedIn = true;
        return this.db
        .collection("users")
        .doc(user.uid)
        .get();
      } else {
        console.log("User not logged in!")
        return null;
      }
    });
  }

  logout(): void {
    const auth = getAuth();
    const result = signOut(auth);

    result.then(() => {
      this.isLoggedIn = false;
      this.router.navigate(['/']);
    });
  }

  emailSignup(email: string, password: string) {
    const auth = getAuth();
    const result = createUserWithEmailAndPassword(auth, email, password);

    result.then(value => {
      alert('Registration Successful');
      // See if two codes below could be separated
      this.login(email,password);
      this.router.navigateByUrl('/profile');
    })
      .catch(error => {
        alert('Something went wrong: ' + error);
      });
  }
