import { Injectable } from '@angular/core';
import { Router } from '@angular/router';

// New angular and firebase require compat layer to follow old tutorials
import { AngularFirestore } from '@angular/fire/compat/firestore';
import { getAuth, signInWithEmailAndPassword, signOut, onAuthStateChanged, UserCredential } from 'firebase/auth';
import { COLLECTION_USERS, PATH_LOGIN, ROLE_ADMIN } from '../../global';
import { Observable, ReplaySubject, Subscription } from 'rxjs';
import { User } from '../models/user.model';
import { distinctUntilChanged } from 'rxjs/operators';

@Injectable({
  providedIn: 'root',
})
export class AuthService {
  /**
   * Stores the current user for other services to use
   */
  currentUser: User = new User();

  /**
   * ReplaySubject that can emit a User object.
   */
  private _currentUser: ReplaySubject<User>;

  /**
   * Observable that other classes can subscribe to in order to get
   * the current user.
   */
  currentUser$: Observable<User>;
  currentUserSub?: Subscription;

  /**
   * ReplaySubject that emits a boolean value depending on whether
   * the user is logged in or out.
   */
  private _loggedIn = new ReplaySubject<boolean>(1);

  /**
   * The Observable that other components can subscribe to in order to
   * receive updates on the user's logged in/out status.
   */
  loggedIn$ = this._loggedIn.asObservable();

  constructor(private router: Router, private db: AngularFirestore) {
    this._currentUser = new ReplaySubject<User>(1);
    this.currentUser$ = this._currentUser.asObservable().pipe(distinctUntilChanged(this.userCompare));

    const auth = getAuth();

    // Code to run whenever the user logs in/out. This is automatically run if the browser remembers
    // that the user is logged in.
    onAuthStateChanged(auth, (user) => {
      if (user) {
        console.log('User is logged in.');
        this.setUser(user.uid);
        this.emitLoggedIn();

        // If this is user's first sign-in, route to change-password component.
        // Otherwise, route to their patient list.
        // TODO: this currently doesn't work because the user's lastSignInTime is
        //  filled out once the user signs in
        // if(auth.currentUser?.metadata.lastSignInTime) {
        //   this.router.navigateByUrl(patientsPath);
        // }
        // else {
        //   this.router.navigateByUrl(changePasswordPath)
        // }
      } else {
        console.log('User is not logged in.');
        this.unsetUser();
        this.emitLoggedOut();
        this.router.navigateByUrl(PATH_LOGIN);
      }
    });
  }

  /**
   *
   * @param email The email used to verify the user in Firebase Authentication.
   * @param password The password used to verify the user in Firebase Authentication.
   * @returns A UserCredential Promise.
   */
  async login(email: string, password: string): Promise<void | UserCredential> {
    const auth = getAuth();

    return signInWithEmailAndPassword(auth, email, password);
  }

  /**
   *
   * @returns Promise for the user's sign out.
   */
  logout(): Promise<void> {
    const auth = getAuth();
    return signOut(auth);
  }

  /**
   * Emits to all subscribers that the user is logged in.
   */
  emitLoggedIn() {
    this._loggedIn.next(true);
  }

  /**
   * Emits to all subscribers that the user is logged out.
   */
  emitLoggedOut() {
    this._loggedIn.next(false);
  }

  /**
   * Called by AuthService when the user logs in (or if the user is already logged in).
   * Emits the user to all subscribers (e.g. guards, main-nav component) for the subscribers
   * to use.
   *
   * TODO: The subscription in this function will persist for the lifetime of the application.
   * How can we unsubscribe from it?
   *
   * @param id The ID of the user to get from the "users" Firestore collection.
   */
  setUser(id: string) {
    this.currentUserSub = this.db
      .collection(COLLECTION_USERS)
      .doc(id)
      .snapshotChanges()
      .subscribe(
        (action) => {
          let obj = { id: action.payload.id, ...(action.payload.data() as object) };
          Object.assign(this.currentUser, obj);
          this.emitUser();
        },
        (err) => {
          console.error('Error loading user', id, ':', err);
        }
      );
  }

  unsetUser() {
    this.currentUserSub?.unsubscribe();
  }

  /**
   * Emit the current User object to all subscribers.
   */
  emitUser() {
    this._currentUser.next(this.currentUser);
  }

  /**
   * @returns Whether or not a user is signed in to the app.
   */
  get hasUser() {
    return this.currentUser != null;
  }

  get isCurrentUserAdmin() {
    return this.currentUser.role.toLocaleLowerCase() == ROLE_ADMIN;
  }

  /**
   * Compares two User objects and determines if they are the same regarding certain
   * properties.
   *
   * @param a First User to compare properties of.
   * @param b Second User to compare properties of.
   * @returns If the two User objects are the same in ID, email, first name,
   * last name, and role.
   */
  private userCompare(a: User, b: User) {
    const idCheck = a.id == b.id;
    const emailCheck = a.email == b.email;
    const firstNameCheck = a.firstName == b.firstName;
    const lastNameCheck = a.lastName == b.lastName;
    const roleCheck = a.role == b.role;

    return idCheck && emailCheck && firstNameCheck && lastNameCheck && roleCheck;
  }
}
