import { Injectable } from '@angular/core';
import { Router } from '@angular/router';

// New angular and firebase require compat layer to follow old tutorials
import { AngularFirestore } from '@angular/fire/compat/firestore'
import {
  getAuth,
  signInWithEmailAndPassword,
  signOut,
  onAuthStateChanged,
  UserCredential,
} from 'firebase/auth'
import { UserService } from './user.service';
import { changePasswordPath, loginPath, patientsPath } from '../global';
import { ReplaySubject } from 'rxjs';


@Injectable({
  providedIn: 'root'
})
export class AuthService {

  /**
   * ReplaySubject that emits a boolean value depending on whether
   * the user is logged in or out.
   */
  private _loggedIn = new ReplaySubject<boolean>(1);

  /**
   * The Observable that other components can subscribe to in order to
   * receive updates on the user's logged in/out status.
   */
  loggedIn$ = this._loggedIn.asObservable();

  constructor(private router: Router, private db: AngularFirestore, private userService: UserService) { 
    const auth = getAuth();

    // Code to run whenever the user logs in/out. This is automatically run if the browser remembers
    // that the user is logged in.
    onAuthStateChanged(auth, (user) => {
      if (user) {
        console.log("User is logged in.")
        this.userService.setUser(user.uid);
        this.emitLoggedIn();

        // If this is user's first sign-in, route to change-password component.
        // Otherwise, route to their patient list.
        // TODO: this currently doesn't work because the user's lastSignInTime is
        //  filled out once the user signs in
        // if(auth.currentUser?.metadata.lastSignInTime) {
        //   this.router.navigateByUrl(patientsPath);
        // }
        // else {
        //   this.router.navigateByUrl(changePasswordPath)
        // }
      } else {
        console.log("User is not logged in.")
        this.userService.unsetUser();
        this.emitLoggedOut();
        this.router.navigateByUrl(loginPath);
      }
    });
  }

  /**
   * 
   * @param email The email used to verify the user in Firebase Authentication.
   * @param password The password used to verify the user in Firebase Authentication.
   * @returns A UserCredential Promise.
   */
  async login(email: string, password: string): Promise<void | UserCredential> {
    const auth = getAuth();

    return signInWithEmailAndPassword(auth, email, password);
  }

  /**
   * 
   * @returns Promise for the user's sign out.
   */
  logout(): Promise<void> {
    const auth = getAuth();
    return signOut(auth);
  }

  /**
   * Emits to all subscribers that the user is logged in.
   */
  emitLoggedIn() {
    this._loggedIn.next(true);
  }

  /**
   * Emits to all subscribers that the user is logged out.
   */
  emitLoggedOut() {
    this._loggedIn.next(false);
  }
}
