import { Component, Input, OnInit } from '@angular/core';
import { UntypedFormGroup } from '@angular/forms';

@Component({
  selector: 'app-base',
  template: ` <p>assessment-form works!</p> `,
  styles: [],
})
export class AssessmentForm implements OnInit {
  // Assume formGroup will be non-null; it must be passed to any extensions of AssessmentForm
  @Input() formGroup!: UntypedFormGroup;

  // Boolean to store value of "All Within Normal Limits" checkbox
  allWNL: boolean = false;

  /**
   * controlList
   *
   * List to store data about each control in a sub-form
   *
   * control: a string naming the specific control
   * default: the default value of the control should "WNL" be true; should match the
   *  'value' property of the default FormOption object
   * wnl: whether or not the control is "within normal limits"
   */
  controlList: { control: string; default: any; wnl: boolean }[] = [];

  constructor() {}

  ngOnInit(): void {}

  /**
   * updateAllWNL
   *
   * Sets allWNL to true if all wnl properties in the controlList are true; otherwise,
   * allWNL is set to false.
   */
  updateAllWNL() {
    this.allWNL = this.controlList != null && this.controlList.every((t) => t.wnl);
  }

  /**
   * someWNL
   *
   * @returns boolean indicating whether or not the "WNL" options in
   * the controlList are all true)
   */
  someWNL(): boolean {
    return this.controlList.filter((t) => t.wnl).length > 0 && !this.allWNL;
  }

  /**
   * setAllWNL
   *
   * @param checked boolean that represents whether the "All WNL" box is checked or not
   *
   * Sets all "wnl" properties in the controlList to the value of the "All WNL" box
   */
  setAllWNL(checked: boolean) {
    this.allWNL = checked;
    this.controlList.forEach((t) => {
      t.wnl = checked;
    });
  }
}
