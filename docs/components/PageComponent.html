import { Component, OnInit } from '@angular/core';
import { ProgressSpinnerMode } from '@angular/material/progress-spinner';
import { Observable, Subscription } from 'rxjs';
import { UserService } from '../services/user.service';
import { User } from '../types/user.type';
 
/**
 * Component that provides the basic properties needed for pages.
 */
@Component({
  selector: 'app-page',
  template: ``,
  styles: [
  ]
})
export class PageComponent implements OnInit {
  // Loading indicator values

  /**
   * Flag to indicate whether the page is loading or not.
   */
  isLoading = false;

  /**
   * The mode of the ProgressSpinner. Defaults to 'indeterminate'.
   */
  mode: ProgressSpinnerMode = 'indeterminate';

  /**
   * The value of the ProgressSpinner that indicates how far the page has loaded.
   * Used only with the 'determinate' mode.
   */
  value = 50;
 
  currentUser: User = new User();
  currentUserSub: Subscription;

  /**
   * Sets isLoading to true, indicating that the progress spinner 
   * should appear. Subclasses should ensure that they are not
   * rendered until isLoading is set to false (in the subclass), 
   * which tells the progress spinner to disappear.
   * 
   * @param userService The service used to retrieve user data.
   */
  constructor(public userService: UserService) {
    this.startLoading();
    this.currentUserSub = userService.currentUser$.subscribe((user: User) => {
      this.currentUser = user;
    })
  }
 
  ngOnInit(): void {
    
  }

  ngOnDestroy() {
    this.currentUserSub.unsubscribe();
  }

  /**
   * Sets the isLoading property to true.
   */
  startLoading(): void {
    this.isLoading = true;
  }

  /**
   * Sets the isLoading property to false.
   */
  stopLoading(): void {
    this.isLoading = false;
  }

  /**
   * @returns whether the user has the admin role or not.
   */
  get isAdmin(): boolean {
    return this.userService.isCurrentUserAdmin;
  }

}
