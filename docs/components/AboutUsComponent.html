import { Component, OnInit } from '@angular/core';
import { AboutService } from 'src/app/shared/services/about.service';
import { Developer } from 'src/app/shared/models/developer.model';

@Component({
  selector: 'app-about-us',
  templateUrl: './about-us.component.html',
  styleUrls: ['./about-us.component.scss'],
})
export class AboutUsComponent implements OnInit {
  developers: Developer[] = [];
  bannerImageName: string = '../../../assets/images/EHR_students_photo_2022-04-06.jpg';

  /**
   * Get all the developers one time from the "developers" Firestore collection.
   *
   * @param aboutService The service used to get developer information.
   */
  constructor(private aboutService: AboutService) {
    this.aboutService.getDevelopers().then((devs: Developer[]) => {
      this.developers = devs.sort((a, b) => a.lastName.localeCompare(b.lastName));
    });
  }

  gridColumns = 3;

  ngOnInit(): void {}

  getDevDegreeInfo(dev: Developer): string[] {
    return dev.getDegreeInfoArray();
  }
}
