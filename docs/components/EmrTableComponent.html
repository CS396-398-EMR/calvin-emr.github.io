Specifies the table's buttons in the "actions" column.

Actions should be handled by the parent component. The table displays the buttons labelled with each action's name and icon. When a button is clicked, the label and the data from the row that the button is associated with is sent up to the parent component, (see the emitRowAction function). The parent component should implement a handler to run logic based on what action was passed to it.

The parent component must handle the button clicks, since the emr-table has no access to the data that the parent manages. If the buttons on the emr-table were to execute functions instead, there would be no data to execute it on (since the parent components) store the data.

import { Component, EventEmitter, Input, OnInit, Output, ViewChild } from '@angular/core';
import { MatTableDataSource } from '@angular/material/table';
import { MatPaginator } from '@angular/material/paginator';
import { Router } from '@angular/router';
import { TableAction, TableEvent } from 'src/app/type/table-action.enum';
import { TableColumn } from 'src/app/type/table-column.type';

@Component({
  selector: 'app-emr-table',
  templateUrl: './emr-table.component.html',
  styleUrls: ['./emr-table.component.scss']
})
export class EmrTableComponent implements OnInit {

  /**
   * The data source used in the mat-table.
   */
  tableDataSource = new MatTableDataSource(([] as any[]));

  /**
   * The columns that are displayed in the table. This is different from the input
   * tableColumns, since this array may include columns that are added on to the
   * tableColumns array (e.g. the "actions" column).
   */
  displayedColumns: string[] = [];

  /**
   * A TableAction used to store the "Navigate" label and "navigate_next" icon.
   */
  navAction: TableAction = new TableAction("Navigate", "navigate_next");

  /**
   * Specifies the columns that will show up in the table.
   */
  @Input() tableColumns: TableColumn[] = [];

  /**
   * Specifies the table's buttons in the "actions" column.
   * 
   * Actions should be handled by the parent component. The table displays the buttons
   * labelled with each action's name and icon. When a button is clicked, the label and
   * the data from the row that the button is associated with is sent up to the parent component, 
   * (see the emitRowAction function). The parent component should implement a handler to
   * run logic based on what action was passed to it.
   * 
   * The parent component must handle the button clicks, since the emr-table has no access
   * to the data that the parent manages. If the buttons on the emr-table were to execute
   * functions instead, there would be no data to execute it on (since the parent components)
   * store the data.
   */
  @Input() tableActions: TableAction[] = [];

  /**
   * Sets the tableDataSource property to the data input from the parent component.
   */
  @Input() set tableData(data: any[]) {
    this.setTableDataSource(data);
  }

  // Cosmetic inputs

  /**
   * Flag indicating whether the table displays checkboxes in the leftmost column.
   */
  @Input() checked: boolean = false;

  /**
   * Flag indicating whether the table displays numbers in the leftmost column.
   */
  @Input() numbered: boolean = false;

  /**
   * Flag indicating whether the table is paginated.
   */
  @Input() paginated: boolean = false;

  /**
   * Table type is a specifier that enables the emr-table to render
   * certain aspects of the table depending on the value.
   * 
   * patient-list: make the table rows clickable
   * 
   * vitals: numbers outside of specified range are red
   */
  @Input() tableType: string = "none"

  /**
   * Emits data (e.g. action clicked) to the parent component.
   */
  @Output() rowAction: EventEmitter<any> = new EventEmitter<any>();

  /**
   * Emits when a checkbox has been checked.
   */
  @Output() checkboxAction: EventEmitter<any> = new EventEmitter<any>();

  /**
   * The MatPaginator component used to configure the table's paginator.
   */
  @ViewChild(MatPaginator) paginator!: MatPaginator;

  /**
   * @param router Allows table to navigate to other routes if table is specified to be "patient-list" type.
   */
  constructor(private router: Router) {}

  /**
   * Determine which columns to show in the rendered table.
   * 
   * The additional columns that can be added are: the checkbox column, numbered column,
   * and actions column.
   */
  ngOnInit(): void {
    this.displayedColumns = this.tableColumns.map((tableColumn: TableColumn) => tableColumn.name);

    // Display a "No." column if the table is specified to be numbered
    if (this.checked) {
      this.displayedColumns = ["check", ...this.displayedColumns]
    }

    if (this.numbered) {
      this.displayedColumns = ["index", ...this.displayedColumns];
    }

    // Display an "Actions" column if the table is passed a non-empty array of TableActions
    if (this.tableActions.length > 0) {
      this.displayedColumns = [...this.displayedColumns, "actions"]
    } 
  }

  /**
   * Needed to re-render table if actions are passed based on user's role.
   */
  ngOnChanges() {
    // Display an "Actions" column if the table is passed a non-empty array of TableActions
    if (this.tableActions.length > 0) {
      // Check if table already has an "actions column"
      if(this.displayedColumns.indexOf("actions") == -1) {
        this.displayedColumns = [...this.displayedColumns, "actions"];
      }
    }
    else {
      this.displayedColumns.pop();    // remove the "actions" from the columns array
    }
  }

  /**
   * Sets the data source's paginator to the created paginator component.
   */
  ngAfterViewInit() {
    this.tableDataSource.paginator = this.paginator;
  }

  /**
   * @param data Data to pass to the tableDataSource property.
   */
  setTableDataSource(data: any) {
    this.tableDataSource = new MatTableDataSource<any>(data);
  }

  // Used to stop emitRowAction from affecting columns that need another click function
  stopPropagate(e: Event) {
    e.stopPropagation();
  }

  /**
   * @param data The data to emit up to the parent component.
   */
  emitCheckboxAction(data: any) {
    this.checkboxAction.emit(data)
  }


  /**
   * Emits a TableAction object containing the action and the data to act on to the 
   * parent component. When a button is clicked in the "Actions" column, the type 
   * of action is sent to the parent component for it to handle - the emr-table
   * does not carry out any actions itself.
   * 
   * @param action The action to emit up to the parent component.
   * @param data The data to emit up to the parent component.
   */
  emitRowAction(action: TableAction, data: any) {
    this.rowAction.emit(new TableEvent(action, data));
  }
}
<!--
  Note: the matColumnDef strings must match the strings contained in the
        ProviderChartsTab component's columns variable. There must be the
        same amount of strings, and the strings must match. If not, the entire
        table will not show up.
-->
<table mat-table [dataSource]="tableDataSource" *ngIf="tableDataSource.data.length > 0"
  class="mat-elevation-z2">

  <ng-container matColumnDef="check">
    <th mat-header-cell *matHeaderCellDef>Selected</th>
    <td mat-cell *matCellDef="let element" (click)="stopPropagate($event)">
      <mat-checkbox
      (change)="emitCheckboxAction(element.user.id)">
        
      </mat-checkbox>
    </td>
  </ng-container>

  <ng-container matColumnDef="index">
    <th mat-header-cell *matHeaderCellDef>No.</th>
    <td mat-cell *matCellDef="let element; let i = index">
      {{ i + 1 }}
    </td>
  </ng-container>

  <ng-container *ngFor="let column of tableColumns" [matColumnDef]="column.name">
    <th mat-header-cell *matHeaderCellDef>{{ column.name }}</th>
    <td mat-cell *matCellDef="let element">

      <!-- tableType == 'vitals' (display values outside normal range in red) -->
      <!-- For more information on how this is set up, reference vitals-tab.component.ts -->
      <span *ngIf="tableType == 'vitals'; else normal_text">
        <!-- If element is deleted, put strikethrough through it -->
        <s *ngIf="element.deleted; else show_not_deleted_vitals">
          <span style="color:#aaa">{{ column.cell(element).vital }}</span>
        </s>
        <!-- If not, display text normally, making it red if value is outside normal range -->
        <ng-template #show_not_deleted_vitals>
          <span *ngIf="column.cell(element).normal; else outside_normal_range">{{ column.cell(element).vital }}</span>
          <ng-template #outside_normal_range>
            <span style="color: red;">{{ column.cell(element).vital }}</span>
          </ng-template>
        </ng-template>
      </span>

      <!-- tableType == 'none' -->
      <!-- If element is deleted, put strikethrough through it -->
      <ng-template #normal_text>
        <s *ngIf="element.deleted; else show_not_deleted_items">
          <span style="color:#aaa">{{ column.cell(element) }}</span>
        </s>
        <!-- If not, display text normally -->
        <ng-template #show_not_deleted_items>
          {{ column.cell(element) }}
        </ng-template>
      </ng-template>
    </td>
  </ng-container>

  <!-- Actions -->
  <ng-container *ngIf="tableActions.length > 0" matColumnDef="actions">
    <th mat-header-cell *matHeaderCellDef>Actions</th>
    <td mat-cell *matCellDef="let element">
      <button mat-icon-button class="action-menu-btn" [matMenuTriggerFor]="menu" aria-label="Show actions">
        <mat-icon>more_vert</mat-icon>
      </button>
      <mat-menu #menu="matMenu">
        <!-- The ngFor below renders one button for each TableAction object in the
              tableActions array. Note that each button comes with a label; when a button
              is pressed, it's label is passed back to the parent so that the parent can
              take action depending on the button pressed. -->
        <button mat-menu-item *ngFor="let tableAction of tableActions"
          (click)="emitRowAction(tableAction, element)">
          <!-- Handles the action of switching "Delete" and "Undo Delete" actions -->
          <div *ngIf="!element.deleted || tableAction.name != 'Delete'; else undoDelete">
            <mat-icon>{{tableAction.icon}}</mat-icon>
            <span>{{tableAction.name}}</span>
          </div>
          <ng-template #undoDelete>
            <mat-icon>{{tableAction.altIcon}}</mat-icon>
            <span>{{tableAction.altName}}</span>
          </ng-template>
        </button>
      </mat-menu>
    </td>
  </ng-container>

  <!-- Column header row -->
  <tr mat-header-row *matHeaderRowDef="displayedColumns"></tr>

  <!-- tableType == 'patient-list' (make rows clickable) -->
  <ng-container *ngIf="tableType == 'patient-list'">
    <!-- We assume that the onRowClick for the table takes the patient ID  as a parameter
          (this is primarily for navigation/routing) -->
    <tr id="patient-list" mat-row *matRowDef="let row; columns: displayedColumns" (click)="emitRowAction(navAction, row.id)">
    </tr>
  </ng-container>
  <!-- tableType == 'student-list' (make rows clickable, but slight change to the emitRowAction 'data' argument)-->
  <ng-container *ngIf="tableType == 'student-list'">
    <tr id="student-list" mat-row *matRowDef="let row; columns: displayedColumns" (click)="emitRowAction(navAction, row.user.id)"></tr>
  </ng-container>
  <ng-container *ngIf="tableType != 'patient-list' && tableType != 'student-list'">
    <tr mat-row *matRowDef="let row; columns: displayedColumns"></tr>
  </ng-container>
</table>
