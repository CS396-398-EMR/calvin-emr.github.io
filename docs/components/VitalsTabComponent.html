import { Component, OnInit } from '@angular/core';
import { VITALS_FIELD } from 'src/app/global';
import { AuthService } from 'src/app/shared/services/auth.service';
import { PatientDashboardTabComponent } from 'src/app/patient-dashboard/dashboard-tab/dashboard-tab.component';
import { Vital } from 'src/app/shared/models/vitals.model';
import { PatientDashboardStore } from 'src/app/patient-dashboard/state/patient-dashboard.store';
import { Patient } from 'src/app/shared/models/patient.model';

@Component({
  selector: 'app-vitals-tab',
  templateUrl: './vitals-tab.component.html',
  styleUrls: ['./vitals-tab.component.scss'],
})
export class VitalsTabComponent extends PatientDashboardTabComponent implements OnInit {
  /**
   * Vitals that can be viewed in the Vitals chart.
   */
  viewableVitals: string[] = [
    'Heart Rate',
    'Oxygen Saturation',
    'Respirations',
    'Temperature',
    'Systolic BP',
    'Diastolic BP',
  ];

  vitals: Vital[] = [];

  constructor(patientStore: PatientDashboardStore, authService: AuthService) {
    super(patientStore, authService);
    this.patient$.subscribe((patient: Patient) => {
      this.vitals = patient.vitals;
    });
  }

  ngOnInit(): void {}

  /**
   *
   * @param vital The Vital object to add to the patient's vitals array.
   */
  addVital(vital: Vital) {
    this.vitals.push(vital);
    this.updatePatientField(VITALS_FIELD, this.vitals);
  }

  editVital(vital: Vital) {
    console.log('Edit');
    this.updatePatientField(VITALS_FIELD, this.vitals);
  }

  /**
   * Deletes a Vital from the local vitals array if the user has the admin role.
   * If the user is not an admin, the function toggles the 'deleted' property of
   * the Vital object.
   *
   * @param vital The Vital to delete.
   */
  deleteVital(vital: Vital): void {
    if (this.isAdmin) {
      this.vitals = this.vitals.filter((v) => v != vital);
    } else {
      vital.toggleDelete();
    }
    this.updatePatientField(VITALS_FIELD, this.vitals);
  }
}
