import { Component, Input, OnInit } from '@angular/core';
import { UserService } from 'src/app/services/user.service';
import { TabComponent } from 'src/app/shared/tab.component';
import { ActionName, TableAction, TableEvent } from 'src/app/type/table-action.enum';
import { TableColumn } from 'src/app/type/table-column.type';
import { Vital } from 'src/app/type/vitals.type';

@Component({
  selector: 'app-vitals-tab',
  templateUrl: './vitals-tab.component.html',
  styleUrls: ['./vitals-tab.component.scss']
})
export class VitalsTabComponent extends TabComponent implements OnInit {

  // Vitals table configurations
  // The TableColumn configurations for Vitals is different - the "cell" function returns an object
  // containing the vital value and a boolean indicating whether that vital is within a normal range
  // or not. The emr-table component uses this information to decide whether to display the vital
  // in red or not. 
  columns: TableColumn[] = [
    new TableColumn("Day/Time", "date",
      (element: Vital) => {
        return {vital: new Date(element.date.seconds * 1000), normal: true}
      }
    ),
    new TableColumn("Heart Rate (bpm)", "heartRate",
      (element: Vital) => { return {vital: element.heartRate, normal: element.isHeartRateWithinNormalRange()} }),
    new TableColumn("Oxygen saturation (%)", "oxySat", 
      (element: Vital) => { return {vital: element.oxySat, normal: element.isOxySatWithinNormalRange()} }),
    new TableColumn("Respirations", "respirations",
      (element: Vital) => { return {vital: element.respirations, normal: element.isRespirationsWithinNormalRange()} }),
    new TableColumn("Temperature (°C)", "temperature",
      (element: Vital) => { return {vital: element.temp, normal: element.isTempWithinNormalRange()} }),    
    new TableColumn("Systolic", "systolic",
      (element: Vital) => { return {vital: element.systolic, normal: element.isSystolicWithinNormalRange()} }),    
    new TableColumn("Diastolic", "diastolic",
      (element: Vital) => { return {vital: element.diastolic, normal: element.isDiastolicWithinNormalRange()} }),
    ];

  actions: TableAction[] = [
    new TableAction("Edit", "edit"),
    new TableAction("Delete", "delete", "Undo Delete", "undo"),
  ];
  
  viewableVitals: string[] = ["Heart Rate", "Oxygen Saturation", "Respirations", "Temperature", "Systolic BP", "Diastolic BP"];

  @Input() vitals: Vital[] = [];

  constructor(private userService: UserService) {
    super();
  }

  ngOnInit(): void {
  }

  addVital(vital: Vital) {
    this.vitals.push(vital);
    this.updateVitals();
  }

  editVital() {
    console.log("Edit")
  }

  onActionClick(event: TableEvent) {
    if(event.action.name == ActionName.Edit) {
      // TODO: implement dialog to edit
    }
    else if(event.action.name == ActionName.Delete) {
      if(this.isAdmin) {
        this.vitals = this.vitals.filter(vital => vital != event.data);
      }
      else {
        this.toggleDelete(event.data as Vital);
      }
    }

    this.updateVitals();
  }

  updateVitals(): void {
    // Merge vitals array with empty array - this triggers the
    //  ngOnChanges() in the emr-table component, since pushing to an
    //  array is not seen as a change by onChanges()
    this.vitals = ([] as any[]).concat(this.vitals);
    this.updatePatient("vitals", this.vitals);
  }

  /**
   * isAdmin
   * 
   * @returns whether or not current user has admin role
   */
   get isAdmin(): boolean {
    return this.userService.getCurrentUserRole() == "admin";
  }
}

<div class="vitals-container">
