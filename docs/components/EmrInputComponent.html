Description

Customizable input used throughout the app (but mostly for the Nursing Assessments tab). Input can be rendered as a text, textarea, numerical, select, or date input. This component makes it simple to implement an input paired with a "Within Normal Limits" checkbox, which when checked, disables the input and fills it with a provided default value.

To see examples where this component is used, visit the any of the classes that extend the AssessmentForm class (e.g. Neurological, Gastrointestinal, Cardiovascular).

Note: WNL = "Within Normal Limits"

import { Component, EventEmitter, Input, OnInit, Output } from '@angular/core';
import { UntypedFormControl } from '@angular/forms';
import { FormOption } from 'src/app/shared/models/form-option.model';

/**
 * Customizable input used throughout the app (but mostly for the Nursing Assessments tab).
 * Input can be rendered as a text, textarea, numerical, select, or date input.
 * This component makes it simple to implement an input paired with a "Within Normal Limits" checkbox,
 * which when checked, disables the input and fills it with a provided default value.
 *
 * To see examples where this component is used, visit the any of the classes that extend the
 * AssessmentForm class (e.g. Neurological, Gastrointestinal, Cardiovascular).
 *
 * Note: WNL = "Within Normal Limits"
 */
@Component({
  selector: 'emr-input',
  templateUrl: './emr-input.component.html',
  styleUrls: ['./emr-input.component.scss'],
})
export class EmrInputComponent implements OnInit {
  /**
   * The type of input to display.
   *
   * Available options: "text", "number", "select", "date"
   */
  @Input() type: string = '';

  /**
   * Controls whether or not to display the "WNL" checkbox next to the input.
   */
  @Input() hasWNL: boolean = false;

  /**
   * Controls the width of the p-input element that is displayed.
   */
  @Input() width: string = '100%';

  /**
   * Label displayed above the input field that describes what the input is for.
   */
  @Input() label: string = '';

  /**
   * Text displayed within the input field when nothing has been entered.
   */
  @Input() placeholder: string = '';

  /**
   * Hint that appears under the input box.
   */
  @Input() hint?: string;

  /**
   * The FormControl to get/set the value for.
   */
  @Input() control: UntypedFormControl = new UntypedFormControl();

  /**
   * The value that should be automatically entered into the input field
   * when the "Within Normal Limits" checkbox is checked.
   */
  @Input() default: any;

  /**
   * Used to toggle whether the form control is within normal limits or not.
   */
  @Input() wnl: boolean = false;

  /**
   * The minimum value that can be entered in the input.
   * Only used with the "number" type.
   */
  @Input() min: number = 0;

  /**
   * The maximum value that can be entered in the input.
   * Only used with the "number" type.
   */
  @Input() max: number = 9999999999;

  /**
   * The options to display in the list the user can select from.
   * Only used with the "select" type.
   */
  @Input() options: FormOption[] = [];

  /**
   * The text to display in the p-dropdown if there are no options.
   * Only used with the "select" type.
   */
  @Input() emptyMessage: string = 'No data';

  /**
   * Whether the input is a required input. This is used primarily to
   * display a '*' character next to the label above the input field
   * to indicate it is required.
   */
  @Input() required: boolean = false;

  /**
   * Whether the component should have a blank space below the input field.
   * This is used primarily to align the input field with other input fields
   * that have hints displayed on the bottom.
   */
  @Input() bottomSpace: boolean = false;

  /**
   * Emits when the "WNL" checkbox is changed.
   */
  @Output() wnlChange = new EventEmitter<boolean>();

  constructor() {}

  ngOnInit(): void {}

  /**
   * Run when the "Set All WNL" checkbox has been set to true, so that
   * the input's value can be set to the default value.
   */
  ngOnChanges(): void {
    if (this.hasWNL) {
      this.toggleWNL();
    }
  }

  /**
   * Toggles the wnl property. Enables/disables input and autofills/clears
   * the default value depending on the value of WNL checkbox.
   */
  toggleWNL() {
    if (this.wnl) {
      this.toDefaultValue();
    } else {
      this.clearControl();
    }
  }

  /**
   * Disables the FormControl and sets its value to be the default value.
   * Emits the value of the wnl property for the parent to handle the
   * "Set All WNL" checkbox display.
   */
  toDefaultValue() {
    // use {onlySelf: true} to restrict validity checking to the current control,
    // rather than the entire form
    this.control?.setValue(this.default, { onlySelf: true });
    this.control?.disable({ onlySelf: true });
    this.wnlChange.emit(this.wnl);
  }

  /**
   * Enables the FormControl and clears its value.
   * Emits the value of the wnl property for the parent to handle the
   * "Set All WNL" checkbox display.
   */
  clearControl() {
    // Enable control and clear it of any value
    this.control?.enable({ onlySelf: true });
    this.control?.setValue('', { onlySelf: true });
    this.wnlChange.emit(this.wnl);
  }

  /**
   * Event handler that is called whenever the input has been changed,
   * if the input is the "select" type. This is used to toggle the
   * "WNL" checkbox whenever the default/WNL option has been selected.
   *
   * @param event The value that the select input has been changed to.
   */
  selectionChanged(event: any) {
    if (event.value === this.default) {
      this.wnl = !this.wnl;
      this.toDefaultValue();
    }
  }

  get isTextType() {
    return this.type.toLocaleLowerCase() === 'text';
  }

  get isTextAreaType() {
    return this.type.toLocaleLowerCase() === 'textarea';
  }

  get isNumberType() {
    return this.type.toLocaleLowerCase() === 'number';
  }

  get isSelectType() {
    return this.type.toLocaleLowerCase() === 'select';
  }

  get isDateType() {
    return this.type.toLocaleLowerCase() === 'date';
  }

  get hasHint() {
    return this.hint !== undefined && this.hint !== null && this.hint !== '';
  }

  get hasBottomSpace() {
    return this.bottomSpace;
  }
}

  <div class="input-container">
    <label [for]="label">{{ required ? label + ' *' : label }}</label>
    <input
      *ngIf="isTextType"
      pInputText
      [id]="label"
      class="input-field"
      [style.width]="width"
      aria-describedby="input-hint"
      [placeholder]="placeholder"
      [formControl]="control"
    />

    <textarea
      *ngIf="isTextAreaType"
      pInputTextarea
      class="input-field"
      rows="8"
      cols="30"
      [placeholder]="placeholder"
      [formControl]="control"
    ></textarea>

    <p-inputNumber
      *ngIf="isNumberType"
      [id]="label"
      class="input-field"
      [style.width]="width"
      aria-describedby="input-hint"
      [placeholder]="placeholder"
      [min]="min"
      [max]="max"
      [formControl]="control"
    ></p-inputNumber>

    <p-dropdown
      *ngIf="isSelectType"
      [id]="label"
      class="input-field dropdown-field"
      [style.width]="width"
      aria-describedby="input-hint"
      [placeholder]="placeholder"
      [emptyMessage]="emptyMessage"
      [autoDisplayFirst]="false"
      [options]="options"
      [formControl]="control"
      (onChange)="selectionChanged($event)"
    >
    </p-dropdown>

    <p-calendar
      *ngIf="isDateType"
      [id]="label"
      class="input-field calendar-field"
      [style.width]="width"
      [formControl]="control"
      [showTime]="true"
      hourFormat="24"
      [placeholder]="placeholder"
      [showIcon]="true"
    ></p-calendar>

    <span *ngIf="hasHint" style="font-size: 0.8em" class="input-hint">{{ hint }}</span>
    <span *ngIf="hasBottomSpace" style="font-size: 0.8em" class="input-hint">&nbsp;</span>
  </div>
  <div *ngIf="hasWNL" class="wnl-checkbox">
    <p-checkbox inputId="wnl" [(ngModel)]="wnl" [binary]="true" label="WNL" (ngModelChange)="toggleWNL()"> </p-checkbox>
  </div>
</div>
