import { Component, OnInit } from '@angular/core';
import { User } from '../../types/user.type';
import { Patient } from 'src/app/types/patient.type';
import { UserService } from 'src/app/services/user.service';
import { PageComponent } from 'src/app/shared/page.component';
import { PatientService } from 'src/app/services/patient.service';
import { TableColumn } from 'src/app/types/table-column.type';
import { Router } from '@angular/router';
import { ActionName, TableEvent } from 'src/app/types/table-action.enum';
import { FileUploadDialogComponent } from 'src/app/shared/file-upload-dialog/file-upload-dialog.component';
import { HistoryAndPhysical } from 'src/app/types/hist-and-phys.type';
import { MatDialog } from '@angular/material/dialog';
import { PatientListDialogComponent } from 'src/app/component/patient-list-dialog/patient-list-dialog.component';
import { ConfirmationDialogComponent } from 'src/app/shared/confirmation-dialog/confirmation-dialog.component';
import { patientsCollectionName } from 'src/app/global';

@Component({
  selector: 'app-patient-list',
  templateUrl: './patient-list.component.html',
  styleUrls: ['./patient-list.component.scss']
})
export class PatientListComponent extends PageComponent implements OnInit {

  patients: Patient[] = [];
  students: User[] = [];
  histAndPhysForm = new HistoryAndPhysical();

  // Variable used for accepting return value from modal
  selectedPatient: string = "";

  // Patient table configurations
  public columns: TableColumn[] = [
    new TableColumn("MRN", "mrn",
      (element: Patient) => element.getMRN()),
    new TableColumn("ID", "id",
      (element: Patient) => element.getId()),
    new TableColumn("Patient Name", "patientName",
      (element: Patient) => element.getLastFirstMiddleName())
  ];

  // Used for importing users
  importedStudents: string[] = [];
  importErrors: string[] = [];

  // Used for importing history and physical form
  importedPatientHistory: string[] = [];

  constructor(private router: Router,
              userService: UserService, 
              private patientService: PatientService,
              private dialog: MatDialog) { 
    super(userService);

    // Get the user's patients each time the user's data changes.
    this.currentUserSub = this.userService.currentUser$.subscribe(() => {
      this.startLoading();
      this.getPatients();
    })
  }

  ngOnInit(): void {
    super.ngOnInit();
  }

  /**
   * Unsubscribes from the currentUser$ Observable.
   */
  ngOnDestroy(): void {
    super.ngOnDestroy();
    this.currentUserSub.unsubscribe();
  }

  /**
   * Get each patient in the user's patient list and add 
   * resulting data to the "patients" array.
   */
  getPatients() {
    this.patients = [];
    this.runLoadingCheck();
    this.currentUser.patientIds.forEach((id) => {
      this.patientService.getStaticPatient(id, patientsCollectionName).then((res: Patient | null) => {
        if(res != null) {
          this.patients.push(res);
        }
      })
      .catch(err => {
        console.error("Error in getting patient:", id , err);
        this.stopLoading();
      })
      .finally(() => {
        this.runLoadingCheck();
      })
    })
  }

  /**
   * Route to the patient page with the clicked row's patient id
   * 
   * @param {TableEvent} event 
   */
  goToPatient(event: TableEvent) {
    if (event.action.name == ActionName.Navigate) {
      let patientId = event.data  // id is passed from emr-table

      // allows patient page to show appropriate patient using patientId in URL
      this.router.navigate(['/patient', "patients", patientId]);
    }
    else {
      console.error("Error in navigating to patient: event is not category 'Navigate'")
    }
  }

  /**
   * Checks if the loading is done and subsequently stops the loading
   * if it is done.
   */
  runLoadingCheck() {
    if(this.patients.length >= this.numPatients) {
      this.stopLoading(); 
    }
  }

  /**
   * @returns The number of patients assigned to the current user.
   */
  get numPatients() {
    return this.currentUser.patientIds.length;
  }

  // ----  operations for uploading History and Physical Form ----

  createPatientHistory(patientHistoryList: string) {
    console.log(patientHistoryList);
    console.log("Create Patient History Test 1 - Load Data...Passed!");
    // Ignore first row, since first row is column headers
    let newPatientHistory = patientHistoryList.split('\r').slice(1);
    // element 0 is the MRN, and element 1 is the date of when the form was created
    // start at element 0 and remove the MRN and Date from the remaining array
    let mrn = newPatientHistory.splice(0, 2)[0]
      .split(',')
      .filter(x => x != "")
      .map(x => x.trim())[1]; // the 0th element is key "mrn" and element 1 is the value to "mrn"

    this.isLoading = true;
    console.log("Create Patient History Test 2 - Assign MRN and remove date...Passed!");

    // now, newPatientHistory should only have keys starting from "allergies"
    // get values from "values" column in .csv
    let histAndPhysVal: string[] = newPatientHistory.map(
      (string) => string.split(",")
        .map(x => x.trim())[1] // the value is element in position 1
    );

    let histAndPhysForm = new HistoryAndPhysical();
    console.log("Create Patient History Test 3 - created histAndPhysForm...Passed!");

    // assign values from .csv to attributes in the histAndPhysForm Object
    Object.keys(histAndPhysForm).forEach((key: string, index: number) => {
      histAndPhysForm[key] = histAndPhysVal[index];
    })
    console.log(histAndPhysForm);
    console.log("Create Patient History Test 4 - Mapped key-value pairs...Passed!");
    console.log(mrn)
    console.log("Create Patient History Test 5 - Retrieved MRN...Passed!")
    // turns the histAndPhysForm object into something that Firebase can understand when pushing to firebase
    let obj = JSON.parse(JSON.stringify(Object.assign({}, histAndPhysForm)))
    this.patientService.setPatient(mrn, { "historyAndPhysical": obj }, "patients"); // upload to firebase
    console.log("Create Patient History Test 6 - Pushing to firebase...Passed!");
    alert("Your history and physical form has been uploaded!");
    // Try to add data; if data is not valid, add to errors array
    this.stopLoading();

    // Display errors, if any
    if (this.importErrors.length > 0) {
      let errorString = this.importErrors.join("\n")
      console.error("Could not import the following data:", errorString,
        "\nVerify that your file is correctly formatted and that no data is missing.")
      // TODO: make this a dialog box
    }
  }

  /**
   * Open up the dialog for uploading history and physical form
   */
  openUploadPatientHistoryDialog() {
    this.importErrors = [];
    const dialogRef = this.dialog.open(FileUploadDialogComponent, {
      width: '500px',
      data: { title: "Import History and Physical", fileResult: this.importedPatientHistory }
    })

    // result is string, since fileReader returns string after parsing file
    dialogRef.afterClosed().subscribe((result: string) => {
      this.createPatientHistory(result);
    })
  }

  /**
   * Open up the dialog for deleting patients.
   */
  openDeletePatientsDialog() {
    // Open the dialog to select patients to delete
    const deletePatientsDialogRef = this.dialog.open(PatientListDialogComponent, {
      width: '500px',
      data: {
        title: "Delete Patients", 
        prompt: "Select patients you wish to delete:",
        buttonLabel: "Delete Patients",
        patients: this.patients,
        type: "student"}
    });

    // If the user selected patients to delete, open a dialog to confirm deletion
    deletePatientsDialogRef.afterClosed().subscribe((patientIds: string[]) => {
      if(patientIds && patientIds.length > 0) {
        const confirmDeleteDialogRef = this.dialog.open(ConfirmationDialogComponent, {
          width: '400px',
          data: { 
            title: "Delete Patients", 
            body: "Are you sure you want to delete the following patients?",
            list: patientIds
          }
        })

        // If user confirms they want to delete patients, delete patients from database
        //  and the user's patientId array
        confirmDeleteDialogRef.afterClosed().subscribe((deletePatients: boolean) => {
          if(deletePatients) {
            patientIds.forEach((id) => {
              this.userService.deletePatientId(id);
              this.patientService.deletePatient(id);
            })
          }
        })
      }
    })
  }
}
