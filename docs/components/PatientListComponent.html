import { Component, OnInit, ViewChild } from '@angular/core';
import { Patient } from 'src/app/shared/models/patient.model';
import { UserService } from 'src/app/shared/services/user.service';
import { PageComponent } from 'src/app/shared/page.component';
import { PatientService } from 'src/app/shared/services/patient.service';
import { Table } from 'primeng/table';
import { Router } from '@angular/router';
import { COLLECTION_PATIENTS, ERROR, PATH_CREATE_PATIENT, SUCCESS } from 'src/app/global';
import { ToastService } from 'src/app/shared/services/toast.service';
import { AuthService } from 'src/app/shared/services/auth.service';

@Component({
  selector: 'app-patient-list',
  templateUrl: './patient-list.component.html',
  styleUrls: ['./patient-list.component.scss'],
})
export class PatientListComponent extends PageComponent implements OnInit {
  /**
   * Array of all patients held by the current user.
   */
  patients: Patient[] = [];

  // Table configuration

  /**
   * The PrimeNG table HTML element.
   */
  @ViewChild('patientTable') patientTable: Table | undefined;

  /**
   * The fields to filter the table by when using the search bar.
   */
  tableFilterFields: string[] = ['mrn', 'id', 'firstName', 'lastName'];

  // Variable used for accepting return value from modal
  selectedPatients: Patient[] = [];

  // Used for importing patients
  importErrors: string[] = [];

  showImportPatientsDialog: boolean = false;
  showDeletePatientsDialog: boolean = false;
  confirmDeletePatients: boolean = false;

  /**
   * Path local constant to direct to "Create Patient" page.
   */
  createPatientPath: string = `/${PATH_CREATE_PATIENT}`;

  constructor(
    authService: AuthService,
    private router: Router,
    private userService: UserService,
    private patientService: PatientService,
    private toastService: ToastService
  ) {
    super(authService);

    // Get the user's patients each time the user's data changes.
    this.currentUserSub = this.authService.currentUser$.subscribe(() => {
      this.startLoading();
      this.getPatients();
    });
  }

  ngOnInit(): void {
    super.ngOnInit();
  }

  /**
   * Unsubscribes from the currentUser$ Observable.
   */
  ngOnDestroy(): void {
    super.ngOnDestroy();
    this.currentUserSub.unsubscribe();
  }

  /**
   * Get each patient in the user's patient list and add
   * resulting data to the "patients" array.
   */
  getPatients() {
    this.patients = [];
    this.startLoading();
    this.currentUser.patientIds.forEach((id) => {
      this.patientService
        .getStaticPatient(id, COLLECTION_PATIENTS)
        .then((res: Patient | null) => {
          if (res != null) {
            this.patients.push(res);
          }
        })
        .catch((err) => {
          this.toastService.addOne({
            severity: ERROR,
            summary: 'Error',
            detail: `Unable to retrieve patient: ${id}!`,
          });
          this.stopLoading();
        })
        .finally(() => {
          this.stopLoading();
        });
    });
  }

  /**
   * Route to the patient page with the clicked row's patient id.
   *
   * @param {string} id The ID of the patient to route to.
   */
  goToPatient(id: string) {
    this.router.navigate(['patients', id]);
  }

  /**
   * Filters the data in the table based on the text in the search field
   * HTML input element. The data is filtered based on whether the any
   * of the fields in the tableFilterFields array contain the text.
   *
   * @param target The search field HTML input element.
   */
  filterTable(target: EventTarget | null) {
    if (target) {
      this.patientTable?.filterGlobal((target as HTMLInputElement).value, 'contains');
    }
  }

  /**
   * @returns The number of patients assigned to the current user.
   */
  get numPatients() {
    return this.currentUser.patientIds.length;
  }

  openImportPatientsDialog() {
    this.showImportPatientsDialog = true;
  }

  /**
   * Open up the dialog to confirm deleting patients.
   */
  openDeletePatientsDialog() {
    this.showDeletePatientsDialog = true;
  }

  onCloseDeletePatientsDialog(didConfirm: boolean) {
    if (didConfirm) {
      this.deletePatients();
    } else {
      this.confirmDeletePatients = false;
    }
  }

  deletePatients() {
    this.startLoading();
    this.toastService.clear();
    this.selectedPatients.forEach((patient) => {
      this.userService.deletePatientIdFromAllUsers(patient.id);

      this.patientService
        .deletePatient(patient.id)
        .catch((error) => {
          this.toastService.addOne({
            severity: ERROR,
            summary: 'Error',
            detail: `Unable to delete patient ${patient.id}!`,
          });
        })
        .then(() => {
          this.patients = this.patients.filter((p) => p.id !== patient.id);
          this.toastService.addOne({
            severity: SUCCESS,
            summary: 'Success',
            detail: `Deleted patient ${patient.id}!`,
          });
        })
        .finally(() => {
          this.stopLoading();
        });
    });
  }

  get didSelectPatients(): boolean {
    return this.selectedPatients.length > 0;
  }
}
