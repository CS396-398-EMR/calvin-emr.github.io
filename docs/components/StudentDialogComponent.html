import { Component, Inject, OnInit } from '@angular/core';
import { UntypedFormBuilder, UntypedFormControl, Validators } from '@angular/forms';
import { DynamicDialogRef, DynamicDialogConfig } from 'primeng/dynamicdialog';
import { uniqueEntryValidator } from 'src/app/shared/directives/unique-entry.directive';
import { ActionName, TableEvent } from 'src/app/shared/models/table-action.enum';
import { User } from 'src/app/shared/models/user.model';

@Component({
  selector: 'app-student-dialog',
  templateUrl: './student-dialog.component.html',
  styleUrls: ['./student-dialog.component.scss'],
})
export class StudentDialogComponent implements OnInit {
  /**
   * Form group for the dialog's form for student info.
   */
  form = this.fb.group({
    id: ['', Validators.required],
    email: ['', [Validators.required, Validators.email]],
    firstName: ['', Validators.required],
    lastName: ['', Validators.required],
  });

  /**
   * Determines what text to display on the dialog.
   * Options: "Add", "Edit", "Remove"
   */
  type: string = '';

  /**
   * Title to display at the top of the dialog box.
   */
  title: string = '';

  /**
   * The text label of the form's submit button.
   */
  btnLabel: string = '';

  /**
   * The student object being examined in the dialog.
   */
  student: User = new User();

  /**
   * List of existing student IDs passed from parent (student-list) to validate ID input.
   */
  ids: string[] = [];

  /**
   * Assigns the dialog's properties values based on the action passed to it.
   */
  constructor(public ref: DynamicDialogRef, public config: DynamicDialogConfig, private fb: UntypedFormBuilder) {
    // Get the User object passed into the dialog
    let obj = config.data as { student: User; ids: string[] };
    Object.assign(this.student, obj.student);

    switch (config.data.action.name) {
      // Create
      case ActionName.Create: {
        this.type = ActionName.Create;
        this.title = 'New Student';
        this.btnLabel = 'Add';
        this.ids = obj.ids;

        // Add validator to student ID to ensure it is unique
        this.idFormControl.addValidators([uniqueEntryValidator(this.ids)]);
        break;
      }
      // Edit
      case ActionName.Update: {
        this.type = ActionName.Update;
        this.title = 'Edit Student Information';
        this.btnLabel = 'Update';

        // Fill the form fields
        this.idFormControl.setValue(this.student.id);
        this.firstNameFormControl.setValue(this.student.firstName);
        this.lastNameFormControl.setValue(this.student.lastName);
        this.emailFormControl.setValue(this.student.email);

        break;
      }
      // Delete
      case ActionName.Delete: {
        this.type = ActionName.Delete;
        this.title = 'Remove Student from Class';
        this.btnLabel = 'Remove';
        break;
      }
      default: {
        this.title = 'Error';
      }
    }
  }

  ngOnInit(): void {}

  onSubmit() {}

  get isCreateType(): boolean {
    return this.type == ActionName.Create;
  }

  get isEditType(): boolean {
    return this.type == ActionName.Update;
  }

  get hasIdError(): boolean {
    return this.idFormControl.invalid && (this.idFormControl.dirty || this.idFormControl.touched);
  }

  get hasIdRequiredError(): boolean {
    return this.idFormControl.errors?.required;
  }

  get hasIdUniqueEntryError(): boolean {
    return this.idFormControl.errors?.uniqueEntry;
  }

  get isSubmitDisabled(): boolean {
    return !this.form.valid && this.type != ActionName.Delete;
  }

  get idFormControl(): UntypedFormControl {
    return this.form.controls.id as UntypedFormControl;
  }

  get emailFormControl(): UntypedFormControl {
    return this.form.controls.email as UntypedFormControl;
  }

  get firstNameFormControl(): UntypedFormControl {
    return this.form.controls.firstName as UntypedFormControl;
  }

  get lastNameFormControl(): UntypedFormControl {
    return this.form.controls.lastName as UntypedFormControl;
  }
}
