import { ChangeDetectorRef, Component, OnInit, ViewChild } from '@angular/core';
import { UntypedFormBuilder, UntypedFormControl, UntypedFormGroup } from '@angular/forms';
import { NURSING_ASSESSMENTS_FIELD } from 'src/app/global';
import { AuthService } from 'src/app/shared/services/auth.service';
import { PatientDashboardTabComponent } from 'src/app/patient-dashboard/dashboard-tab/dashboard-tab.component';
import { Assessment } from 'src/app/shared/models/assessment.model';
import { formToAssessment } from './assessment-converter.helper';
import { PatientDashboardStore } from 'src/app/patient-dashboard/state/patient-dashboard.store';
import { Patient } from 'src/app/shared/models/patient.model';
import { CardiovascularFormComponent } from './cardiovascular-form/cardiovascular-form.component';
import { RespiratoryFormComponent } from './respiratory-form/respiratory-form.component';
import { NeurologicalFormComponent } from './neurological-form/neurological-form.component';
import { MusculoskeletalFormComponent } from './musculoskeletal-form/musculoskeletal-form.component';
import { GastrointestinalFormComponent } from './gastrointestinal-form/gastrointestinal-form.component';
import { GenitourinaryFormComponent } from './genitourinary-form/genitourinary-form.component';
import { PsychosocialFormComponent } from './psychosocial-form/psychosocial-form.component';
import { IntegumentaryFormComponent } from './integumentary-form/integumentary-form.component';
import { initialFormState } from './assessments-tab.metadata';

export interface FormControlData {
  control: UntypedFormControl;
  default: string;
  wnl: any;
}

@Component({
  selector: 'app-assessments-tab',
  templateUrl: './assessments-tab.component.html',
  styleUrls: ['./assessments-tab.component.scss'],
})
export class AssessmentsTabComponent extends PatientDashboardTabComponent implements OnInit {
  assessments: Assessment[] = [];

  /**
   * Overarching nursing assessment form
   *
   * Notes:
   *  o emr-checkbox-group elements should not include a "WNL" option; the "WNL" option will be
   *    implemented on its own in the component.
   */
  form = this.fb.group({
    // Date
    date: [''],

    // Cardiovascular
    cardiovascular: this.fb.group({
      heartTones: this.fb.group({
        pulsePattern: false, // is pulse pattern irregular?
        heartTones: [''], // murmur, gallup
        rhythm: [''], // NSR, atrial fibrillation, PVC's, PAC's, SVT, Atrial flutter, other
        otherRhythm: [''], // for nurse to fill out if rhythm is "Other"
      }),
      edema: this.fb.group({
        jvd: false, // presence of JVD
        gnpe: false, // generalized non-pitting edema
        pittingEdema: this.fb.group({
          rightHand: [''],
          rightFoot: [''],
          leftFoot: [''],
          leftHand: [''],
        }),
      }),
      generalizedPerfusion: [''], // cold, pale, cyanotic, palpitations
      capRefill: false, // is cap refill > 3 seconds?
      pulses: this.fb.group({
        rue: [''], // +3 strong, +2 normal, +1 weak, thready, 0 absent
        rle: [''],
        lue: [''],
        lle: [''],
      }),
    }),

    // Respiratory
    respiratory: this.fb.group({
      respiratoryPattern: this.fb.group({
        irregularRhythm: false,
        quality: [''],
      }),
      lungSounds: this.fb.group({
        rul: [''], // wnl, coarse rhonchi, crackles/rales, wheezes expiratory, wheezes inspiratory, other
        otherRul: [''], // for nurse to fill out if lung sound is "Other"
        rll: [''],
        otherRll: [''],
        rml: [''],
        otherRml: [''],
        lul: [''],
        otherLul: [''],
        lll: [''],
        otherLll: [''],
      }),
      cough: false,
      coughDescription: [''], // productive, nonproductive, dry
      productiveDescription: [[]],
    }),

    // Neurological
    neurological: this.fb.group({
      loc: [''], // lethargic, somnolent
      orientation: [[]],
      extremityMovement: [''],
      handGrips: [''], // moderate, weak
      lePushPull: [''], // moderate, weak
      pupillaryResponse: [''], // sluggish, absent
      pupillaryDilation: this.fb.group({
        left: [''], // 2mm, 4mm, 6mm, 8mm
        right: [''],
      }),
      gcss: [''], // AKA Glasgow Coma Scale score
      sensation: this.fb.group({
        rue: [''], // numbness, tingling, full sensation
        rle: [''],
        lle: [''],
        lue: [''],
      }),
      tremors: [[]],
    }),

    // Musculoskeletal
    musculoskeletal: this.fb.group({
      // Refer to the "Johns Hopkins Fall Risk Assessment Tool"
      fallRisk: this.fb.group({
        age: [''],
        fallHistory: false,
        // ebu = elimination, bowel and urine
        ebu: [''],
        medications: [''],
        // pce = patient care equipment
        pce: [''], // one present, two present, or 3 or more present
        mobility: [[]],
        cognition: [[]],
      }),
      fallScore: [''],
      mobilityAndTransfers: [''],
      assistiveDevices: [[]],
    }),

    // Gastrointestinal
    gastrointestinal: this.fb.group({
      bowelSounds: this.fb.group({
        ruq: [''], // hypoactive, hyperactive, absent
        rlq: [''],
        luq: [''],
        llq: [''],
      }),
      dateLastBM: [''],
      nausea: false, // yes (true), no (false)
      emesis: [''], // small, medium, large
      incontinence: false,
      ostomy: this.fb.group({
        stomaSite: [''], // pink, beefy red, pale, purple
        output: [''], // loose, watery, formed
      }),
      hemorrhoids: false, // present (true), absent (false)
    }),

    // Genitourinary
    genitourinary: this.fb.group({
      urineColor: [''],
      urineCharacteristics: [[]],
      urinaryPattern: [[]],
      foley: this.fb.group({
        justification: [''],
        securement: [''],
      }),
    }),

    // Psychosocial
    psychosocial: this.fb.group({
      mood: [[]],
      suicidal: false,
    }),

    // Integumentary (wounds)
    integumentary: this.fb.group({
      skinAppearance: [[]],
      wounds: this.fb.group({
        location: [''],
        staging: [''],
        care: [[]],
      }),
    }),
  });

  // ViewChildren necessary to reset each form
  @ViewChild('cardiovascularForm') cardiovascularForm!: CardiovascularFormComponent;
  @ViewChild('respiratoryForm') respiratoryForm!: RespiratoryFormComponent;
  @ViewChild('neurologicalForm') neurologicalForm!: NeurologicalFormComponent;
  @ViewChild('musculoskeletalForm') musculoskeletalForm!: MusculoskeletalFormComponent;
  @ViewChild('gastrointestinalForm') gastrointestinalForm!: GastrointestinalFormComponent;
  @ViewChild('genitourinaryForm') genitourinaryForm!: GenitourinaryFormComponent;
  @ViewChild('psychosocialForm') psychosocialForm!: PsychosocialFormComponent;
  @ViewChild('integumentaryForm') integumentaryForm!: IntegumentaryFormComponent;

  constructor(
    patientStore: PatientDashboardStore,
    authService: AuthService,
    private fb: UntypedFormBuilder,
    private ref: ChangeDetectorRef
  ) {
    super(patientStore, authService);
    this.patient$.subscribe((patient: Patient) => {
      this.assessments = patient.assessments;
    });
  }

  ngOnInit(): void {}

  onSubmit(): void {
    // Enable the form to get values of previously-disabled controls
    this.form.enable();

    // Update assessments field in current patient
    let assessment = formToAssessment(this.form);
    this.assessments.push(assessment);
    this.assessments = ([] as any[]).concat(this.assessments);
    this.updatePatientField(NURSING_ASSESSMENTS_FIELD, this.assessments);

    // Reset
    this.resetAllForms();
    this.form.reset(initialFormState);
  }

  /**
   * Sets all sub-forms back to initial state (namely the WNL values).
   */
  resetAllForms() {
    this.cardiovascularForm.reset();
    this.respiratoryForm.reset();
    this.neurologicalForm.reset();
    this.musculoskeletalForm.reset();
    this.gastrointestinalForm.reset();
    this.genitourinaryForm.reset();
    this.psychosocialForm.reset();
    this.integumentaryForm.reset();

    // Eliminates the "Changed After Checked" error since we could be
    //  changing the WNL checkboxes from 'true' to 'false'
    this.ref.detectChanges();
  }

  get dateFormControl(): UntypedFormControl {
    return this.form.controls.date as UntypedFormControl;
  }

  get cardiovascularFormGroup(): UntypedFormGroup {
    return this.form.controls.cardiovascular as UntypedFormGroup;
  }

  get respiratoryFormGroup(): UntypedFormGroup {
    return this.form.controls.respiratory as UntypedFormGroup;
  }

  get neurologicalFormGroup(): UntypedFormGroup {
    return this.form.controls.neurological as UntypedFormGroup;
  }

  get musculoskeletalFormGroup(): UntypedFormGroup {
    return this.form.controls.musculoskeletal as UntypedFormGroup;
  }

  get gastrointestinalFormGroup(): UntypedFormGroup {
    return this.form.controls.gastrointestinal as UntypedFormGroup;
  }

  get genitourinaryFormGroup(): UntypedFormGroup {
    return this.form.controls.genitourinary as UntypedFormGroup;
  }

  get psychosocialFormGroup(): UntypedFormGroup {
    return this.form.controls.psychosocial as UntypedFormGroup;
  }

  get integumentaryFormGroup(): UntypedFormGroup {
    return this.form.controls.integumentary as UntypedFormGroup;
  }

  get drainsAndTubesFormGroup(): UntypedFormGroup {
    return this.form.controls.drainsAndTubes as UntypedFormGroup;
  }
}
