import { Component, Input, OnInit } from '@angular/core';
import { FormBuilder, FormControl, FormGroup } from '@angular/forms';
import { nursingAssessmentsField } from 'src/app/global';
import { UserService } from 'src/app/services/user.service';
import { TabComponent } from 'src/app/shared/tab.component';
import { Assessment } from 'src/app/types/assessment.type';
import { formToAssessment } from './assessment-converter.helper';

export interface FormControlData {
  control: FormControl, 
  default: string, 
  wnl: any
}

@Component({
  selector: 'app-assessments-tab',
  templateUrl: './assessments-tab.component.html',
  styleUrls: ['./assessments-tab.component.scss']
})
export class AssessmentsTabComponent extends TabComponent implements OnInit {

   * Overarching nursing assessment form

        otherRhythm: [''],  // for nurse to fill out if rhythm is "Other"

        jvd: false,  // presence of JVD
        gnpe: false, // generalized non-pitting edema

        lue: [''],
        lle: [''],

    }),

      respiratoryPattern: this.fb.group({
        irregularRhythm: false,
        quality: [''],

        otherRll: [''],
        rml: [''],
        otherRml: [''],
        lul: [''],
        otherLul: [''],
        lll: [''],
        otherLll: [''],
      }),
      cough: false,
      coughDescription: [''],  // productive, nonproductive, dry
      productiveDescription: this.fb.group({
        scant: false,

        large: false,
        purulent: false,

        black: false,
        green: false,
        yellow: false,
        thick: false,
        thin: false,
        tenacious: false,
      }),
    }),

    // Neurological
    neurological: this.fb.group({
      loc: [''],  // lethargic, somnolent
      orientation: this.fb.group({  // Not to person/place/time/situation
        person: false,
        place: false,
        time: false,
        situation: false,
      }),
      extremityMovement: [''],
      handGrips: [''],  // moderate, weak
      lePushPull: [''], // moderate, weak
      pupillaryResponse: [''],  // sluggish, absent
      pupillaryDilation: this.fb.group({
        left: [''],  // 2mm, 4mm, 6mm, 8mm
        right: [''],
      }),
      gcss: [''],  // AKA Glasgow Coma Scale score
      sensation: this.fb.group({
        rue: [''],  // numbness, tingling, full sensation
        rle: [''],
        lle: [''],

        rue: false,
        rle: false,

        lue: false

    }),

    musculoskeletal: this.fb.group({
      // Refer to the "Johns Hopkins Fall Risk Assessment Tool"
      fallRisk: this.fb.group({
        age: [''],
        fallHistory: false,
        // ebu = elimination, bowel and urine
        ebu: [''],
        medications: [''],
        // pce = patient care equipment
        pce: [''],  // one present, two present, or 3 or more present
        mobility: this.fb.group({
          assistance: false,  // "requires assistance..."
          gait: false,  // "unsteady gait"
          impairment: false,  // "visual or auditory impairment affecting mobility"
        }),
        cognition: this.fb.group({
          awareness: false, // "altered awareness of immediate physical environment"
          impulsive: false, // "impulsive"
          understanding: false, // "lack of understanding of one's physical and cognitive limitations"
        }),
      }),
      fallScore: [''],
      mobilityAndTransfers: [''],
      assistiveDevices: this.fb.group({

        cane: false,

      bowelSounds: this.fb.group({
        ruq: [''], // hypoactive, hyperactive, absent
        rlq: [''],
        luq: [''],

  constructor(private fb: FormBuilder, private userService: UserService) { 
    super();
  }

  ngOnInit(): void {
  }

  onSubmit(): void {
    this.form.enable();

    let assessment = formToAssessment(this.form)
    this.assessments.push(assessment);
  
    this.assessments = ([] as any[]).concat(this.assessments);
    this.updatePatient(nursingAssessmentsField, this.assessments)

    this.form.reset();
    alert("Assessment submitted!")
  }

  /**
   * @returns Whether or not current user has admin role.
   */
  get isAdmin(): boolean {
    return this.userService.isCurrentUserAdmin;
  }

  get dateFormControl(): FormControl {
    return this.form.controls.date as FormControl;
  }

  get cardiovascularFormGroup(): FormGroup {
    return this.form.controls.cardiovascular as FormGroup;
  }

  get respiratoryFormGroup(): FormGroup {
    return this.form.controls.respiratory as FormGroup;
  }

  get neurologicalFormGroup(): FormGroup {
    return this.form.controls.neurological as FormGroup;
  }

  get musculoskeletalFormGroup(): FormGroup {
    return this.form.controls.musculoskeletal as FormGroup;
  }

  get gastrointestinalFormGroup(): FormGroup {
    return this.form.controls.gastrointestinal as FormGroup;
  }
  
  get genitourinaryFormGroup(): FormGroup {
    return this.form.controls.genitourinary as FormGroup;
  }

  get psychosocialFormGroup(): FormGroup {
    return this.form.controls.psychosocial as FormGroup;
  }

  get integumentaryFormGroup(): FormGroup {
    return this.form.controls.integumentary as FormGroup;
  }

  get drainsAndTubesFormGroup(): FormGroup {
    return this.form.controls.drainsAndTubes as FormGroup;
  }

}
