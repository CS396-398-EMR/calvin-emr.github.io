import { Component, OnInit } from '@angular/core';
import { ActivatedRoute, Router } from '@angular/router';

export interface MedicalConditionForm {
  name: string;
  altName: string;
}

export interface InterdisciplinaryTeamForm {
  specialization: string;
  name: string;
  title: string;
}

@Component({
  selector: 'app-create-patient',
  templateUrl: './create-patient.component.html',
  styleUrls: ['./create-patient.component.scss'],
})
export class CreatePatientComponent implements OnInit {
  prefillForm: boolean = false;
  type: string = '';
  pid: string | null;

  constructor(private router: Router, private route: ActivatedRoute) {
    // Get type from the router URL ('create-patient' or 'edit-patient/:id')
    this.type = (this.route.snapshot.routeConfig?.path as string).split('/')[0];

    // Get patient ID (if it exists in path)
    this.pid = this.route.snapshot.paramMap.get('id');
    if (!this.isValidPath) {
      this.router.navigate(['/page-not-found']);
    } else {
      this.prefillForm = true;
    }
  }

  ngOnInit(): void {}

  get isValidPath() {
    return this.isCreatePatientPage || (this.isEditPatientPage && this.pid !== null && this.pid !== '');
  }

  get isCreatePatientPage() {
    return this.type.toLocaleLowerCase() === 'create-patient';
  }

  get isEditPatientPage() {
    return this.type.toLocaleLowerCase() === 'edit-patient';
  }
}
