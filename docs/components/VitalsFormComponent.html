import { Component, EventEmitter, OnInit, Output } from '@angular/core';
import { UntypedFormBuilder, UntypedFormControl } from '@angular/forms';
import { Timestamp } from 'firebase/firestore';
import { Vital } from 'src/app/shared/models/vitals.model';

@Component({
  selector: 'app-vitals-form',
  templateUrl: './vitals-form.component.html',
  styleUrls: ['./vitals-form.component.scss'],
})
export class VitalsFormComponent implements OnInit {
  vitalsForm = this.fb.group({
    heartRate: [],
    oxySat: [],
    oxygen: [],
    respirations: [],
    temperature: [],
    systolic: [],
    diastolic: [],
  });

  @Output() newVitalEvent = new EventEmitter<Vital>();

  constructor(private fb: UntypedFormBuilder) {}

  ngOnInit(): void {}

  addNewVital(): void {
    // We assume nothing from the vitals array can be truly deleted, so use
    //  array length as vitals ID
    let newVital = new Vital(
      Timestamp.now(),
      this.vitalsForm.value.heartRate,
      this.vitalsForm.value.oxySat,
      this.vitalsForm.value.oxygen,
      this.vitalsForm.value.respirations,
      this.vitalsForm.value.temperature,
      this.vitalsForm.value.systolic,
      this.vitalsForm.value.diastolic,
      false
    );

    this.vitalsForm.reset();

    // Emit the new vital to vitals-tab, which calls addVital(newVital)
    this.newVitalEvent.emit(newVital);
  }

  get heartRateFormControl(): UntypedFormControl {
    return this.vitalsForm.controls.heartRate as UntypedFormControl;
  }

  get oxySatFormControl(): UntypedFormControl {
    return this.vitalsForm.controls.oxySat as UntypedFormControl;
  }

  get oxygenFormControl(): UntypedFormControl {
    return this.vitalsForm.controls.oxygen as UntypedFormControl;
  }

  get respirationsFormControl(): UntypedFormControl {
    return this.vitalsForm.controls.respirations as UntypedFormControl;
  }

  get temperatureFormControl(): UntypedFormControl {
    return this.vitalsForm.controls.temperature as UntypedFormControl;
  }

  get systolicFormControl(): UntypedFormControl {
    return this.vitalsForm.controls.systolic as UntypedFormControl;
  }

  get diastolicFormControl(): UntypedFormControl {
    return this.vitalsForm.controls.diastolic as UntypedFormControl;
  }
}
