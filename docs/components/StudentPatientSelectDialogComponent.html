import { Component, OnInit, Inject } from '@angular/core';
import { MatDialog, MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { Patient } from 'src/app/type/patient.type';
import { Router } from '@angular/router';

@Component({
  selector: 'app-student-patient-select-dialog',
  templateUrl: './student-patient-select-dialog.component.html',
  styleUrls: ['./student-patient-select-dialog.component.scss']
})
export class StudentPatientSelectDialogComponent implements OnInit {

  displayedColumns: string[] = ["id", "name"];
  dataSource: any[] = [];
  selectedPatient: Patient = new Patient();
  selectedRow:number = -1;

  constructor(
      private router: Router,
      public dialogRef: MatDialogRef<StudentPatientSelectDialogComponent>,
      @Inject(MAT_DIALOG_DATA) public data: any
    ){ }

  ngOnInit(): void {
    this.dataSource = this.data;
  }

  /**
   *  Routes to the patient page and pass down the appropriate UID value
   * bound to the row and close the dialog
   * 
   * @param {Patient} row   Patient data of the selected row 
   * @param {number} index  Index number of selected row
   */
  selectPatient(row: any, index: number): void {
    // Information in clicked row gets stored in 'selectedPatient'
    this.selectedRow = index
    
    this.router.navigate(['/patient', "studentPatients", row.uid]);
    this.dialogRef.close();
  }

}
