import { Component, OnInit } from '@angular/core';
import { UntypedFormBuilder, UntypedFormControl } from '@angular/forms';
import { DynamicDialogConfig, DynamicDialogRef } from 'primeng/dynamicdialog';
import { Subscription } from 'rxjs';
import { Patient } from 'src/app/shared/models/patient.model';
import { ProviderOrder } from 'src/app/shared/models/provider-order.model';

@Component({
  selector: 'app-administer-med-dialog',
  templateUrl: './administer-med-dialog.component.html',
  styleUrls: ['./administer-med-dialog.component.scss'],
})
export class AdministerMedDialogComponent implements OnInit {
  // Variables to store data
  patient!: Patient;
  order!: ProviderOrder;
  orderAdministered: boolean = false;

  medicationIdFormControl: UntypedFormControl = new UntypedFormControl('');

  medicationIdSubscription: Subscription;

  // Variables to handle error messages
  submitted: boolean = false;
  error: string = '';

  constructor(public ref: DynamicDialogRef, public config: DynamicDialogConfig, private fb: UntypedFormBuilder) {
    // Set submitted to false whenever the form's value changes;
    // this gets rid of any error messages when the user starts typing
    this.medicationIdSubscription = this.medicationIdFormControl.valueChanges.subscribe(() => {
      this.submitted = false;
    });

    this.patient = config.data.patient;
    this.order = config.data.order;
  }

  ngOnInit(): void {}

  /**
   * Closes the dialog and sends medication data to the parent component.
   */
  onCloseClick(): void {
    // Pass the med back in case the user partially administered the medication
    this.medicationIdFormControl.reset();
    this.ref.close({ order: this.order, administered: this.orderAdministered });
  }

  /**
   * Submit the form and checks to see correct medication is administered
   */
  onSubmit(): void {
    if (this.correctMedId) {
      this.order.quantityGiven += 1;
      this.medicationIdFormControl.reset();
      this.orderAdministered = true;
    } else {
      this.submitted = true;
      this.error = this.getError();
    }
  }

  /**
   * Returns various error messages to be displayed on client side.
   * The error messages are checked below in descending precedence.
   */
  getError(): string {
    if (!this.correctMedId) {
      return 'This is the incorrect medication to administer!';
    }
    return 'Something went wrong - try again.';
  }

  get correctMedId(): boolean {
    let val = this.medicationIdFormControl.value as string;
    return val != null && val.trim() === this.order.medId;
  }

  /**
   * Returns whether the administered quantity is greater than the
   * provider order quantity.
   */
  get isOverdosed(): boolean {
    return this.order.quantityGiven > this.order.quantity;
  }
}
