import { Component, OnInit } from '@angular/core';
import { Cardiovascular } from 'src/app/shared/models/assessment.model';
import { AssessmentInfoComponent } from '../assessment-info/assessment-info.component';

@Component({
  selector: 'app-cardiovascular-info',
  templateUrl: './cardiovascular-info.component.html',
  styleUrls: ['./cardiovascular-info.component.scss'],
})
export class CardiovascularInfoComponent extends AssessmentInfoComponent<Cardiovascular> implements OnInit {
  constructor() {
    super();
  }

  ngOnInit(): void {}

  get isIrregularPulse() {
    return this.data.pulsePattern ? 'Yes' : 'No';
  }

  get isJVDPresent() {
    return this.data.jvd ? 'Yes' : 'No';
  }

  get hasGNPE() {
    return this.data.gnpe ? 'Yes' : 'No';
  }

  get isNormalCapRefill() {
    return this.data.capRefill ? 'Yes' : 'No';
  }
}
