import { Component, OnInit, Output, EventEmitter, Input } from '@angular/core';
import { UntypedFormBuilder, UntypedFormControl, Validators } from '@angular/forms';
import { ProgressNote } from '../../../shared/models/progress-note.model';
import { Timestamp } from '@firebase/firestore';

@Component({
  selector: 'app-note-form',
  templateUrl: './note-form.component.html',
  styleUrls: ['./note-form.component.scss'],
})
export class NoteFormComponent implements OnInit {
  form = this.fb.group({
    title: ['', Validators.required],
    note: ['', Validators.required],
    date: ['', Validators.required],
    signature: ['', Validators.required],
  });

  @Input() note?: ProgressNote;

  constructor(private fb: UntypedFormBuilder) {}

  ngOnInit(): void {
    this.prefillForm();
  }

  /**
   * Used to emit new note objects up to the parent component.
   */
  @Output() newNoteEvent = new EventEmitter<ProgressNote>();

  /**
   * If the form is specified to be prefilled (for example, when a note is being edited),
   * set all values in the form to match those of the provided ProgressNote.
   * ngOnInit must be used to set the form; otherwise, the form's value will be set before
   * the Note object is passed to the NoteForm component.
   */
  prefillForm() {
    // For pre-filling form in OrderEditDialogComponent
    if (this.note !== undefined && this.note !== null) {
      const date = this.note.simDateCreated.toDate();
      this.dateFormControl.setValue(date);

      this.form.patchValue({
        title: this.note.title,
        note: this.note.content,
        signature: this.note.signature,
      });
    }
  }

  /**
   * Retrieves and formats data from the form to put into a ProgressNote object.
   *
   * @returns A new ProgressNote with values taken from the form.
   */
  getNoteFromForm(): ProgressNote {
    let id = '' + Date.now();
    const title = this.titleFormControl.value.trim();
    const simDateCreated = Timestamp.fromDate(new Date(this.dateFormControl.value));
    const signature = this.signatureFormControl.value.trim();
    let dateCreated = Timestamp.now();

    if (this.isPrefilled && this.note) {
      id = this.note.id;
      dateCreated = this.note?.dateCreated;
    }

    return new ProgressNote(id, simDateCreated, title, this.noteFormControl.value, signature, dateCreated);
  }

  /**
   * Creates a new ProgressNote from the form and emits the instance to the parent.
   */
  onSubmit() {
    // We assume nothing from the providerOrders array can be truly deleted, so use
    //  array length as provider order ID
    let note = this.getNoteFromForm();

    this.form.reset();

    // Emit the new provider order to progress-notes-tab
    this.newNoteEvent.emit(note);
  }

  get isPrefilled() {
    return this.note !== undefined && this.note !== null;
  }

  get titleFormControl(): UntypedFormControl {
    return this.form.controls.title as UntypedFormControl;
  }

  get noteFormControl(): UntypedFormControl {
    return this.form.controls.note as UntypedFormControl;
  }

  get dateFormControl(): UntypedFormControl {
    return this.form.controls.date as UntypedFormControl;
  }

  get signatureFormControl(): UntypedFormControl {
    return this.form.controls.signature as UntypedFormControl;
  }
}
