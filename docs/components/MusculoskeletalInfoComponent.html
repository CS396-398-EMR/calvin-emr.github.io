import { Component, OnInit } from '@angular/core';
import { Musculoskeletal } from 'src/app/types/assessment.type';
import { AssessmentInfoComponent } from '../assessment-info/assessment-info.component';

@Component({
  selector: 'app-musculoskeletal-info',
  templateUrl: './musculoskeletal-info.component.html',
  styleUrls: ['./musculoskeletal-info.component.scss']
})
export class MusculoskeletalInfoComponent extends AssessmentInfoComponent<Musculoskeletal> implements OnInit {

  constructor() { super(); }

  ngOnInit(): void {
  }

  get hasFallHistory() {
    return this.data.fallHistory ? "Yes - one fall within 6 months before admission" : "No";
  }

  get fallRiskCategory(): string {
    if(this.data.fallScore < 6) { return "Low"}
    else if(this.data.fallScore >= 6 && this.data.fallScore <= 13) { return "Medium"}
    else { return "High"}
  }

}
