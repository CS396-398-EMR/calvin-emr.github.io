import { Component, OnInit } from '@angular/core';
import { FormControl, FormGroup } from '@angular/forms';
import { AssessmentForm } from 'src/app/shared/assessment-form.component';
import { FormOption } from 'src/app/types/form-option.type';

@Component({
  selector: 'app-psychosocial-form',
  templateUrl: './psychosocial-form.component.html',
  styleUrls: ['./psychosocial-form.component.scss', '../assessments-tab.component.scss']
})
export class PsychosocialFormComponent extends AssessmentForm implements OnInit {

  constructor() { 
    super();
    this.controlList = [
      {control: "mood", default: false,  wnl: false},
    ];
  }

  ngOnInit(): void { }

  moodIndex = 0;

  moodOptions = [
    new FormOption("Non-interactive", "nonInteractive"),
    new FormOption("Elated", "elated"),
    new FormOption("Tearful", "tearful"),
    new FormOption("Anxious", "anxious"),
    new FormOption("Agitated", "agitated"),
    new FormOption("Withdrawn", "withdrawn"),
    new FormOption("Flat", "flat"),
  ]

  get moodDefault() {
    return this.controlList[this.moodIndex].default;
  }

  get moodFormGroup(): FormGroup {
    return this.formGroup.controls.mood as FormGroup;
  }

  get suicidalFormControl(): FormControl {
    return this.formGroup.controls.suicidal as FormControl;
  }

}
