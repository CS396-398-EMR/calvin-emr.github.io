import { Component, OnInit } from '@angular/core';
import { FormControl } from '@angular/forms';
import { AssessmentForm } from 'src/app/shared/assessment-form.component';
import { FormOption } from 'src/app/types/form-option.type';

@Component({
  selector: 'app-cardiovascular-form',
  templateUrl: './cardiovascular-form.component.html',
  styleUrls: ['./cardiovascular-form.component.scss', '../assessments-tab.component.scss']
})
export class CardiovascularFormComponent extends AssessmentForm implements OnInit {

  constructor() { 
    super();
    this.controlList = [
      {control: "heartTones", default: "WNL" , wnl: false},
      {control: "rhythm", default: "NSR", wnl: false},
      {control: "rightHand", default: "WNL", wnl: false},
      {control: "rightFoot", default: "WNL", wnl: false},
      {control: "leftFoot", default: "WNL", wnl: false},
      {control: "leftHand", default: "WNL", wnl: false},
      {control: "generalizedPerfusion", default: "WNL", wnl: false},
      {control: "rue", default: "+2 normal", wnl: false},
      {control: "rle", default: "+2 normal", wnl: false},
      {control: "lue", default: "+2 normal", wnl: false},
      {control: "lle", default: "+2 normal", wnl: false},
    ];
  }

  ngOnInit(): void {
    // Clear the "Other Rhythm" control if the user switches from "Other" to a different option
    this.rhythmFormControl.valueChanges.subscribe((value: string) => {
      if(value != "Other") {
        this.otherRhythmFormControl.setValue('');
      }
    })
  }

  /**
   * Heart Tones
   */

  heartTonesIndex = 0;
  rhythmIndex = 1;

  heartTonesOptions = [
    new FormOption("Murmur"),
    new FormOption("Gallup"),
    new FormOption("WNL"),
  ]

  rhythmOptions = [
    new FormOption("Atrial Fibrillation"),
    new FormOption("PVC's"),
    new FormOption("PAC's"),
    new FormOption("SVT"),
    new FormOption("Atrial Flutter"),
    new FormOption("Other"),
    new FormOption("NSR"),
  ]

  get heartTonesDefault() {
    return this.controlList[this.heartTonesIndex].default;
  }

  get rhythmDefault() {
    return this.controlList[this.rhythmIndex].default;
  }

  get pulsePatternFormControl(): FormControl {
    return this.formGroup.get("heartTones.pulsePattern") as FormControl;
  }

  get heartTonesFormControl(): FormControl {
    return this.formGroup.get("heartTones.heartTones") as FormControl;
  }

  get rhythmFormControl(): FormControl {
    return this.formGroup.get("heartTones.rhythm") as FormControl;
  }

  get otherRhythmFormControl(): FormControl {
    return this.formGroup.get("heartTones.otherRhythm") as FormControl;
  }

  get isOtherRhythm() {
    return this.rhythmFormControl.value == "Other";
  }
  
  /**
   * Edema
   */

  rightHandIndex = 2;
  rightFootIndex = 3;
  leftFootIndex = 4;
  leftHandIndex = 5;

  pittingEdemaOptions = [
    new FormOption("1+"),
    new FormOption("2+"),
    new FormOption("3+"),
    new FormOption("4+"),
    new FormOption("WNL"),
  ]

  get pittingEdemaDefault() {
    return this.controlList[this.rightHandIndex].default;
  }

  get jvdFormControl(): FormControl {
    return this.formGroup.get("edema.jvd") as FormControl;
  }

  get gnpeFormControl(): FormControl {
    return this.formGroup.get("edema.gnpe") as FormControl;
  }

  get rightHandFormControl(): FormControl {
    return this.formGroup.get("edema.pittingEdema.rightHand") as FormControl;
  }

  get rightFootFormControl(): FormControl {
    return this.formGroup.get("edema.pittingEdema.rightFoot") as FormControl;
  }

  get leftFootFormControl(): FormControl {
    return this.formGroup.get("edema.pittingEdema.leftFoot") as FormControl;
  }

  get leftHandFormControl(): FormControl {
    return this.formGroup.get("edema.pittingEdema.leftHand") as FormControl;
  }

  /**
   * Other
   */

  generalizedPerfusionIndex = 6;
  rueIndex = 7;
  rleIndex = 8;
  lueIndex = 9;
  lleIndex = 10;

  generalizedPerfusionOptions = [
    new FormOption("Cold"),
    new FormOption("Pale"),
    new FormOption("Cyanotic"),
    new FormOption("Palpitations"),
    new FormOption("WNL"),
  ]

  extremityOptions = [
    new FormOption("RUE"),
    new FormOption("RLE"),
    new FormOption("LUE"),
    new FormOption("LLE"),
  ]

  pulseOptions = [
    new FormOption("+3 strong"),
    new FormOption("+2 normal"),
    new FormOption("+1 weak, thready"),
    new FormOption("0 absent"),
  ]

  get generalizedPerfusionDefault() {
    return this.controlList[this.generalizedPerfusionIndex].default
  }

  get pulseDefault() {
    return this.controlList[this.rueIndex].default;
  }

  get generalizedPerfusionFormControl(): FormControl {
    return this.formGroup.get("generalizedPerfusion") as FormControl;
  }

  get capRefillFormControl(): FormControl {
    return this.formGroup.get("capRefill") as FormControl;
  }

  get rueFormControl(): FormControl {
    return this.formGroup.get("pulses.rue") as FormControl;
  }

  get rleFormControl(): FormControl {
    return this.formGroup.get("pulses.rle") as FormControl;
  }

  get lueFormControl(): FormControl {
    return this.formGroup.get("pulses.lue") as FormControl;
  }

  get lleFormControl(): FormControl {
    return this.formGroup.get("pulses.lle") as FormControl;
  }

}
