import { Component, OnInit, Output, EventEmitter, Input } from '@angular/core';
import { UntypedFormBuilder, UntypedFormControl, Validators } from '@angular/forms';
import { Subscription } from 'rxjs';
import { Timestamp } from '@firebase/firestore';
import { Medication } from 'src/app/shared/models/medication.model';
import { MedicationService } from 'src/app/shared/services/medication.service';
import { prnOption, scheduledOption } from 'src/app/global';
import { FormOption } from 'src/app/shared/models/form-option.model';
import { ProviderOrder } from 'src/app/shared/models/provider-order.model';

@Component({
  selector: 'app-order-form',
  templateUrl: './order-form.component.html',
  styleUrls: ['./order-form.component.scss'],
})
export class OrderFormComponent implements OnInit {
  /**
   * A ProviderOrder used to prefill the form. For example,
   * if an order is being edited by the user.
   */
  @Input() order?: ProviderOrder;

  /**
   * The list of all medications.
   */
  medications: Medication[] = [];

  /**
   * The list of all medications along with their IDs concatenated in a string.
   */
  medicationOptions: string[] = [];

  /**
   * Form used to hold information about a provider order.
   */
  providerOrderForm = this.fb.group({
    medication: ['', Validators.required],
    route: ['', Validators.required],
    dosage: ['', Validators.required],
    quantity: ['', [Validators.required, Validators.min(0)]],
    type: ['', Validators.required],
    startDate: [''],
    order: [''],
    signature: ['', Validators.required],
  });

  orderTypeOptions: FormOption[] = [new FormOption(scheduledOption), new FormOption(prnOption)];

  medicationSubscription!: Subscription;

  /**
   * Used to emit new order objects up to the parent component.
   */
  @Output() newOrderEvent = new EventEmitter<ProviderOrder>();

  /**
   * Gets all medications from Firestore and initializes the filter for the
   * 'Medication Name' form input.
   *
   * @param fb FormBuilder used to create the providerOrderForm.
   * @param medService Service used to retrieve medications from Firestore.
   */
  constructor(private fb: UntypedFormBuilder, private medService: MedicationService) {}

  ngOnInit(): void {
    // Get all medications from backend
    this.medicationSubscription = this.medService.getAllMedications().subscribe((res) => {
      this.medications = res.map((item) => {
        let med = new Medication();
        let obj = { id: item.payload.doc.id, ...(item.payload.doc.data() as object) };
        Object.assign(med, obj);
        return med;
      });

      this.medicationOptions = this.medications.map((med) => {
        return `${med.name} (${med.id})`;
      });

      this.prefillForm();
    });
  }

  ngOnDestroy(): void {
    this.medicationSubscription.unsubscribe();
  }

  /**
   * If the form is specified to be prefilled (for example, when an order is being edited),
   * set all values in the form to match those of the provided ProviderOrder.
   * ngOnInit must be used to set the form; otherwise, the form's value will be set before
   * the Order object is passed to the OrderForm component.
   */
  prefillForm() {
    // For pre-filling form in OrderEditDialogComponent:
    //  If an order is provided, fill the form fields with their respective values
    if (this.isPrefilled && this.order) {
      if (this.order.isScheduled) {
        const date = this.order.startDate?.toDate();
        this.startDateFormControl.setValue(date);
      }

      this.providerOrderForm.patchValue({
        medication: this.medications.find((med) => med.id === this.order?.medId),
        route: this.order.route,
        dosage: this.order?.dosage,
        type: this.order.type,
        order: this.order?.order,
        quantity: this.order?.quantity,
        signature: this.order?.signature,
      });
    }
  }

  /**
   * Retrieves and formats data from the form to put into a ProviderOrder object.
   *
   * @returns A new ProviderOrder with values taken from the form.
   */
  public getOrderFromForm() {
    // Get medication information
    let med = this.medicationFormControl.value as Medication;
    let medId = med.getId();
    let medName = med.getName();

    // Remove whitespace
    const dosage = this.providerOrderForm.value.dosage.trim();
    const order = this.providerOrderForm.value.order.trim();
    const signature = this.providerOrderForm.value.signature.trim();

    // Create variables to use to store pre-filled or default data
    let id, quantityGiven, dateCreated;
    // Set variable values based on whether form is pre-filled or not
    if (this.isPrefilled && this.order) {
      id = this.order.id;
      quantityGiven = this.order.quantityGiven;
      dateCreated = this.order.dateCreated;
    } else {
      // Set order ID to be the date in milliseconds
      id = '' + Date.now();
      quantityGiven = 0;
      dateCreated = Timestamp.now();
    }

    let startDate = null;
    let endDate = null;

    // Get times for start date and end date
    if (this.isScheduled) {
      startDate = Timestamp.fromDate(new Date(this.startDateFormControl.value));
      endDate = new Date(this.startDateFormControl.value);
      // Set end date time to be 1 hour after start date time
      endDate.setTime(endDate.getTime() + 60 * 60 * 1000);
      endDate = Timestamp.fromDate(endDate);
    }

    return new ProviderOrder(
      id,
      medId,
      medName,
      this.routeFormControl.value,
      dosage,
      this.quantityFormControl.value,
      quantityGiven,
      this.orderTypeFormControl.value,
      order,
      signature,
      dateCreated,
      startDate,
      endDate
    );
  }

  /**
   * Creates a new ProviderOrder from the form and emits the instance to the parent.
   */
  onSubmit() {
    let newOrder = this.getOrderFromForm();
    this.providerOrderForm.reset();

    // Emit the new provider order to provider-charts-tab, which calls addOrder(newOrder)
    this.newOrderEvent.emit(newOrder);
  }

  /**
   * Gets the possible routes associated with the selected medication to display
   * as options in the form.
   */
  get medRoutes(): string[] {
    if (this.medicationFormControl.value === null || this.medicationFormControl.value === '') {
      return [];
    }

    return (this.medicationFormControl.value as Medication).routes;
  }

  get isPrefilled() {
    return this.order != null;
  }

  get isScheduled() {
    return this.orderTypeFormControl.value == scheduledOption;
  }

  get didSelectMedication() {
    return (
      this.medicationFormControl.value !== undefined &&
      this.medicationFormControl.value !== null &&
      this.medicationFormControl.value !== ''
    );
  }

  get medicationFormControl(): UntypedFormControl {
    return this.providerOrderForm.controls.medication as UntypedFormControl;
  }

  get routeFormControl(): UntypedFormControl {
    return this.providerOrderForm.controls.route as UntypedFormControl;
  }

  get dosageFormControl(): UntypedFormControl {
    return this.providerOrderForm.controls.dosage as UntypedFormControl;
  }

  get quantityFormControl(): UntypedFormControl {
    return this.providerOrderForm.controls.quantity as UntypedFormControl;
  }

  get orderTypeFormControl(): UntypedFormControl {
    return this.providerOrderForm.controls.type as UntypedFormControl;
  }

  get startDateFormControl(): UntypedFormControl {
    return this.providerOrderForm.controls.startDate as UntypedFormControl;
  }

  get orderFormControl(): UntypedFormControl {
    return this.providerOrderForm.controls.order as UntypedFormControl;
  }

  get signatureFormControl(): UntypedFormControl {
    return this.providerOrderForm.controls.signature as UntypedFormControl;
  }
}
