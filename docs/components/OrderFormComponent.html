import { Component, OnInit, Output, EventEmitter, Input } from '@angular/core';
import { FormBuilder, Validators } from '@angular/forms';
import { ProviderOrder } from 'src/app/types/order.type';
import {Observable} from 'rxjs';
import { Timestamp } from '@firebase/firestore';
import { Medication } from 'src/app/types/medication.type';
import { MedicationService } from 'src/app/services/medication.service';

import {map, startWith} from 'rxjs/operators';

@Component({
  selector: 'app-order-form',
  templateUrl: './order-form.component.html',
  styleUrls: ['./order-form.component.scss']
})
export class OrderFormComponent implements OnInit {

  @Input() order?: ProviderOrder;

  // --- begin autcomplete tutorial ---
  medications: Medication[] = [];
  filteredOptions!: Observable<string[]>; // TODO: scrumbag needs to remove "!"
  // --- end autcomplete tutorial ---
  providerOrderForm = this.fb.group({
    medication: ['', Validators.required],
    route: ['', Validators.required],
    dosage: ['', Validators.required],
    quantity: ['', [Validators.required, Validators.min(0)]],
    startDate: ['', Validators.required],
    order: [''],
    signature: ['', Validators.required]
  });

  constructor(private fb: FormBuilder,
    private medService: MedicationService) { }

  ngOnInit(): void {
    this.medService.getMedications().subscribe(res => {
      this.medications = res.map(item => {
        let med = new Medication();
        let obj = { id: item.payload.doc.id, ...item.payload.doc.data() as object };
        Object.assign(med, obj)
        return med;
      })
    })
    // --- begin autcomplete tutorial ---
    this.filteredOptions = this.providerOrderForm.get("medication")!.valueChanges.pipe(
      startWith(''),
      map(value => this._filter(value)),
    );
    // --- end autcomplete tutorial ---

    // For pre-filling form in OrderEditDialogComponent
    if(this.isPrefilled && this.order) {
      let date = new Date(this.order?.startDate.seconds * 1000)
      let start = date?.toLocaleDateString("en-CA", {
        year: 'numeric',
        month: '2-digit',
        day: '2-digit',
        weekday:"long",
        hour: '2-digit', 
        hour12: false, 
        minute:'2-digit'})      // "Saturday, 2022-02-26, 19:23"       
      .split(", ")              // ["Saturday", "2022-02-26", "19:23"]
      .slice(1)                 // ["2022-02-26", "19:23"]
      .join("T");               // "2022-02-26T19:23"

      this.providerOrderForm.setValue({
        medication: this.order?.medName + " (" + this.order?.medId + ")",
        route: this.order?.route,
        dosage: this.order?.dosage,
        startDate: start,
        order: this.order?.order,
        quantity: this.order?.quantity,
        signature: this.order?.signature
      })
    }
  }

  private _filter(value: string): string[] {
    if(value != null) {
      const filterValue = value.toLowerCase();
      return this.medications
      .map(
        med => med.name + " (" + med.id + ")"
      )
      .filter(med => med.toLowerCase().includes(filterValue));
    }
    return [];
  }

  @Output() newOrderEvent = new EventEmitter<ProviderOrder>();

  public getOrderFromForm() {
    // We assume nothing from the providerOrders array can be truly deleted, so use
    //  array length as provider order ID
    let medInfo = this.providerOrderForm.controls.medication.value.split(" ")
    let medId = medInfo.pop().replace(/[\(\)]+/g, "");
    let medName = medInfo.join(" ");

    // Get times for start date and end date
    let startDate = new Date(this.providerOrderForm.value.startDate);
    let endDate = new Date(this.providerOrderForm.value.startDate);
    endDate.setTime( endDate.getTime() + (60*60*1000) );


    // Create variables to use to store pre-filled or default data
    let id, quantityGiven, dateCreated;
    // Set variable values based on whether form is pre-filled or not
    if(this.isPrefilled && this.order) {
      id = this.order.id;
      quantityGiven = this.order.quantityGiven;
      dateCreated = this.order.dateCreated;
    }
    else {
      id = "" + Date.now();
      quantityGiven = 0;
      dateCreated = Timestamp.now();
    }

    return new ProviderOrder(
      id,
      medId,
      medName,
      this.providerOrderForm.value.route,
      this.providerOrderForm.value.dosage,
      this.providerOrderForm.value.quantity,
      quantityGiven,
      this.providerOrderForm.value.order,
      this.providerOrderForm.value.signature,
      Timestamp.fromDate(startDate),
      Timestamp.fromDate(endDate),
      dateCreated);
  }

  addNewOrder() {
    let newOrder = this.getOrderFromForm();
    this.providerOrderForm.reset();

    // Emit the new provider order to provider-charts-tab, which calls addOrder(newOrder)
    this.newOrderEvent.emit(newOrder);
  }

  get medRoutes(){
    let medName = this.providerOrderForm.controls.medication.value
    if(medName != null) {
      let name = medName.split("(")[0];
      let medIndex = this.medications.findIndex(med => med.name.trim() == name.trim())
      let med = this.medications[medIndex] as Medication
      if(med) {
        let routes = med.getRoutes();
        if(routes.length == 1) {
          this.providerOrderForm.controls.route.setValue(routes[0]);
        }
        return routes;
      }
    }
    return []
  }

  get isPrefilled() {
    return this.order != null;
  }
}
