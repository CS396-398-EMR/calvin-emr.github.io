import { Component, OnInit } from '@angular/core';
import { Respiratory } from 'src/app/types/assessment.type';
import { AssessmentInfoComponent } from '../assessment-info/assessment-info.component';

@Component({
  selector: 'app-respiratory-info',
  templateUrl: './respiratory-info.component.html',
  styleUrls: ['./respiratory-info.component.scss']
})
export class RespiratoryInfoComponent extends AssessmentInfoComponent<Respiratory> implements OnInit {

  constructor() { super(); }

  ngOnInit(): void {
  }

  get isRegularRhythm() {
    return !this.data.irregularRhythm ? "Yes" : "No";
  }

  get isCoughPresent() {
    return this.data.cough ? "Yes" : "No";
  }

  get isProductiveCough() {
    return this.data.coughDescription.toLocaleLowerCase() == "productive";
  }
}
