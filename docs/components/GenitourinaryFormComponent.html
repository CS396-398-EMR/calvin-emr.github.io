import { Component, OnInit } from '@angular/core';
import { UntypedFormControl } from '@angular/forms';
import { AssessmentForm } from 'src/app/shared/assessment-form.component';
import { urinaryPatternOptions, urineCharacteristicsOptions, urineColorOptions } from './genitourinary-form.metadata';

@Component({
  selector: 'app-genitourinary-form',
  templateUrl: './genitourinary-form.component.html',
  styleUrls: ['./genitourinary-form.component.scss', '../assessments-tab.component.scss'],
})
export class GenitourinaryFormComponent extends AssessmentForm implements OnInit {
  constructor() {
    super();
    this.reset();
  }

  ngOnInit(): void {}

  reset() {
    this.controlList = [
      { control: 'urineColor', default: 'WNL', wnl: false },
      { control: 'urineCharacteristics', default: [], wnl: false },
      { control: 'urinaryPattern', default: [], wnl: false },
    ];
  }

  colorIndex = 0;
  characteristicsIndex = 1;
  patternIndex = 2;

  urineColorOptions = urineColorOptions;
  urineCharacteristicsOptions = urineCharacteristicsOptions;
  urinaryPatternOptions = urinaryPatternOptions;

  get urineColorDefault() {
    return this.controlList[this.colorIndex].default;
  }

  get urineCharacteristicsDefault() {
    return this.controlList[this.characteristicsIndex].default;
  }

  get urinaryPatternDefault() {
    return this.controlList[this.patternIndex].default;
  }

  get urineColorFormControl(): UntypedFormControl {
    return this.formGroup.controls.urineColor as UntypedFormControl;
  }

  get urineCharacteristicsFormControl(): UntypedFormControl {
    return this.formGroup.controls.urineCharacteristics as UntypedFormControl;
  }

  get urinaryPatternFormControl(): UntypedFormControl {
    return this.formGroup.controls.urinaryPattern as UntypedFormControl;
  }

  /**
   * Foley
   */

  get foleyJustificationFormControl(): UntypedFormControl {
    return this.formGroup.get('foley.justification') as UntypedFormControl;
  }

  get foleySecurementFormControl(): UntypedFormControl {
    return this.formGroup.get('foley.securement') as UntypedFormControl;
  }
}
