import { Component, OnInit } from '@angular/core';
import { FormControl, Validators } from '@angular/forms';
import { AuthService } from 'src/app/services/auth.service';

import {
  getAuth,
} from 'firebase/auth'

@Component({
  selector: 'app-user-registration',
  templateUrl: './user-registration.component.html',
  styleUrls: ['./user-registration.component.scss']
})
export class UserRegistrationComponent implements OnInit {
  usrEmail = new FormControl('', [Validators.required, Validators.email]);
  usrPassword = new FormControl('', [Validators.required, Validators.min(6)]);
  confPassword = new FormControl('', [Validators.required, Validators.min(6)]);
  hide = true;

  constructor(private auth: AuthService) { }

  ngOnInit(): void {
  }

  /**
   * Returns appropriate email authentication error message.
   * 
   * @returns error message string
   */
  getEmailErrorMessage() {
    if (this.usrEmail.hasError('required')) {
      return 'You must enter a value';
    }
    return this.usrEmail.hasError('email') ? 'Not a valid email' : '';
  }

  /**
   * Returns appropriate password authentication error message.
   * 
   * @returns error message string
   */
  getPasswordErrorMessage() {
    if (this.usrPassword.hasError('required')) {
      return 'You must enter a value';
    }
    return this.usrPassword.hasError('password') ? 'Not a valid length' : '';
  }

  /**
   * Returns appropriate password confirmation error message.
   * 
   * @returns error message string
   */
  getPasswordConfirmErrorMessage() {
    if (this.usrPassword.hasError('required')) {
      return 'You must enter a value';
    }
    return this.usrPassword.hasError('password') ? 'Passwords does not match' : '';
  }

  /**
   * Register user with entered email and password authentication.
   */
  register(): void {
    let usrEmail = this.usrEmail.value;
    let usrPassword = this.usrPassword.value;

    const auth = getAuth();

    // add if-else statement to check passwords
    this.emailRegister(usrEmail, usrPassword);
  }

  /**
   * Registration using email and password 
   * (Fireauth provides other means of registration as well)
   * 
   * @param {string} email E-mail
   * @param {string} pw    Password
   */
  emailRegister(email: string, pw: string): void {
    this.auth.emailSignup(email, pw);
    // this.auth.login(email, pw);
  }
}
