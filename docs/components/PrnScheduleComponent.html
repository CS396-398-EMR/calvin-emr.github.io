import { Component, OnInit } from '@angular/core';
import { DialogService } from 'primeng/dynamicdialog';
import { PatientDashboardStore } from 'src/app/patient-dashboard/state/patient-dashboard.store';
import { AdministrationScheduleComponent } from '../administration-schedule/administration-schedule.component';

@Component({
  selector: 'app-prn-schedule',
  templateUrl: './prn-schedule.component.html',
  styleUrls: ['./prn-schedule.component.scss'],
  providers: [DialogService],
})
export class PrnScheduleComponent extends AdministrationScheduleComponent implements OnInit {
  constructor(patientStore: PatientDashboardStore, dialogService: DialogService) {
    super(patientStore, dialogService);

    this.patient$.subscribe((patient) => {
      this.orders = patient.providerOrders.filter((order) => order.isPRN);
    });
  }

  ngOnInit(): void {}
}
