import { Component, Input, OnInit } from '@angular/core';
import { Validators, UntypedFormBuilder, UntypedFormGroup, UntypedFormArray, UntypedFormControl } from '@angular/forms';
import { ActivatedRoute, Router } from '@angular/router';
import { Timestamp } from 'firebase/firestore';
import { map } from 'rxjs/operators';
import { Subscription } from 'rxjs';
import { uniqueEntryValidator } from 'src/app/shared/directives/unique-entry.directive';
import { COLLECTION_PATIENTS, FEMALE, MALE } from 'src/app/global';
import { PatientService } from 'src/app/shared/services/patient.service';
import { UserService } from 'src/app/shared/services/user.service';
import { Patient } from 'src/app/shared/models/patient.model';
import { SimPerson } from 'src/app/shared/models/sim-person.model';
import { MedicalCondition } from 'src/app/shared/models/medical-condition.model';
import { FormOption } from 'src/app/shared/models/form-option.model';
import { AuthService } from 'src/app/shared/services/auth.service';

@Component({
  selector: 'app-patient-form',
  templateUrl: './patient-form.component.html',
  styleUrls: ['./patient-form.component.scss'],
})
export class PatientFormComponent implements OnInit {
  @Input() type: string = '';
  @Input() prefill: boolean = false;
  @Input() patientId: string | null = null;

  // Limits for certain inputs

  /**
   * The minimum number that can be input into any numerical FormControl.
   */
  minMeasurement: number = 0;

  /**
   * The maximum age that can be input in the age FormControl.
   */
  maxAge: number = 130;

  /**
   * The maximum number of kilograms that can be input in the weight FormControl.
   */
  maxWeight: number = 300;

  /**
   * The maximum number of feet that can be input in the heightFeet FormControl.
   */
  maxFeet: number = 10;

  /**
   * The maximum number of inches that can be input in the heightInches FormControl.
   */
  maxInches: number = 11;

  patient: Patient = new Patient();

  /**
   * The form containing all controls.
   */
  patientForm = this.fb.group({
    firstName: ['', Validators.required],
    lastName: ['', Validators.required],
    middleName: [''],
    id: ['', Validators.required], // also used for MRN

    // General info
    generalInfo: this.fb.group({
      codeStatus: ['', Validators.required],
      dob: ['', [Validators.required]],
      age: [null, [Validators.required, Validators.min(this.minMeasurement), Validators.max(this.maxAge)]],
      sex: ['', Validators.required],
      weight: [null, [Validators.required, Validators.min(this.minMeasurement), Validators.max(this.maxWeight)]],
      heightFeet: [null, [Validators.required, Validators.min(this.minMeasurement), Validators.max(this.maxFeet)]],
      heightInches: [null, [Validators.required, Validators.min(this.minMeasurement), Validators.max(this.maxInches)]],
      isolation: ['', Validators.required],
    }),

    // Interdisciplinary team
    interdisciplinaryTeam: this.fb.array([]),
    activeMedicalProblems: this.fb.array([]),
    allergies: this.fb.array([]),

    // profile pic name
    profilePicName: [''],
  });

  /**
   * The patient's profile picture.
   */
  profilePic?: File;

  /**
   * All MRNs that are already in use in the "patients" collection in Firestore.
   */
  mrns: string[] = [];

  /**
   * Stores the subscription to the Observable that retrieves all patients in
   * the "patients" collection.
   */
  mrnsSub?: Subscription;

  /**
   * The options for the "Code Status" input.
   */
  codeStatusOptions: FormOption[] = [
    new FormOption('Full'),
    new FormOption('Limited'),
    new FormOption('DNR'),
    new FormOption('Comfort Care'),
  ];

  /**
   * The options for the "Sex" input.
   */
  sexOptions: FormOption[] = [new FormOption(FEMALE), new FormOption(MALE)];

  constructor(
    private router: Router,
    private route: ActivatedRoute,
    private fb: UntypedFormBuilder,
    private patientService: PatientService,
    private userService: UserService,
    private authService: AuthService
  ) {}

  /**
   * Get all MRNs that are already in use by existing patients.
   */
  ngOnInit(): void {
    // Push each existing patient MRN into the mrns list
    this.mrnsSub = this.patientService
      .getAllMRNs()
      .pipe(
        map((actionArray) => {
          actionArray.forEach((val) => {
            this.mrns.push(val.payload.doc.id);
          });
        })
      )
      .subscribe(() => {
        // Add validator to MRN FormControl to ensure it is unique
        this.patientForm.controls['id'].addValidators([uniqueEntryValidator(this.mrns)]);
      });

    this.prefillForm();
  }

  ngOnDestroy(): void {
    this.mrnsSub?.unsubscribe();
  }

  randomizeForm() {
    this.patientForm.patchValue({
      firstName: 'Testfirstname',
      lastName: 'Testlastname',
      id: 'deletethis' + Date.now(),

      // General info
      generalInfo: {
        codeStatus: 'Full',
        dob: new Date(),
        age: Math.random() * 100,
        sex: Math.round(Math.random()) ? 'Male' : 'Female',
        weight: Math.random() * 300,
        heightFeet: Math.random() * 10,
        heightInches: Math.random() * 11,
        isolation: 'none',
      },
    });
  }

  prefillForm() {
    if (this.prefill) {
      const pid = this.route.snapshot.paramMap.get('id');
      if (pid) {
        let patientSubscription = this.patientService.getPatient(pid, 'patients').subscribe((res) => {
          let patient = new Patient();
          let obj = { id: res.payload.id, ...(res.payload.data() as object) };
          Object.assign(patient, obj);
          patient.toJSObject();
          this.patient = patient;
          this.setFormValues();
          patientSubscription.unsubscribe();
        });
      }
    }
  }

  setFormValues() {
    // Split height into feet and inches
    const totalInches = this.patient.getHeightInches();
    const feet = Math.floor(totalInches / 12);
    const inches = totalInches % 12;

    this.patientForm.patchValue({
      firstName: this.patient.firstName,
      middleName: this.patient.middleName,
      lastName: this.patient.lastName,
      id: this.patient.id,

      // General info
      generalInfo: {
        codeStatus: this.patient.getCodeStatus(),
        dob: this.patient.getDOB().toDate(),
        age: this.patient.getAge(),
        sex: this.patient.getSex(),
        weight: this.patient.getWeight(),
        heightFeet: feet,
        heightInches: inches,
        isolation: this.patient.getIsolation(),
      },

      // profile pic name
      profilePicName: this.patient.getProfilePicURL(),
    });

    // Disable ID field
    this.idFormControl.disable();

    // Set medical info
    this.patient.interdisciplinaryTeam.forEach((member: SimPerson) => {
      this.addInterdisciplinaryTeamFormGroup(member);
    });
    this.patient.activeMedicalProblems.forEach((problem: MedicalCondition) => {
      this.addMedicalProblemFormGroup(problem);
    });
    this.patient.allergies.forEach((allergy: MedicalCondition) => {
      this.addAllergyFormGroup(allergy);
    });
  }

  /**
   * Generate a random 6-digit number, where the first digit
   * is never 0. Set the ID FormControl to this value.
   */
  generateMRN() {
    let mrn = '' + Math.floor(1000000000 + Math.random() * 9000000000);
    this.idFormControl.setValue(mrn);
  }

  /**
   * Adds a FormGroup to a FormArray, depending on the desired FormArray.
   * Default values of the controls in the FormGroup are empty strings.
   *
   * @param type Specifies which FormGroup to add.
   */
  addEmptyFormGroup(type: string) {
    switch (type.toLocaleLowerCase()) {
      case 'team': {
        this.addInterdisciplinaryTeamFormGroup();
        break;
      }
      case 'problem': {
        this.addMedicalProblemFormGroup();
        break;
      }
      case 'allergy': {
        this.addAllergyFormGroup();
        break;
      }
    }
  }

  /**
   * Adds a FormGroup for an interdisciplinary team member to the form.
   *
   * @param member Optional parameter that is used to prefill the FormGroup's control values.
   */
  addInterdisciplinaryTeamFormGroup(member?: SimPerson) {
    let specialization = member?.specialization ? member.specialization : '';
    let name = member?.name ? member.name : '';
    let title = member?.title ? member.title : '';

    const formGroup = this.fb.group({
      specialization: [specialization, Validators.required],
      name: [name, Validators.required],
      title: [title],
    });

    this.interdisciplinaryTeamFormArray.push(formGroup);
  }

  /**
   * Adds a FormGroup for an active medical problem to the form.
   *
   * @param problem Optional parameter that is used to prefill the FormGroup's control values.
   */
  addMedicalProblemFormGroup(problem?: MedicalCondition) {
    let name = problem?.name ? problem.name : '';
    let altName = problem?.altName ? problem.altName : '';

    const formGroup = this.fb.group({
      name: [name, Validators.required],
      altName: [altName],
    });

    this.activeMedProblemsFormArray.push(formGroup);
  }

  /**
   * Adds a FormGroup for an allergy to the form.
   *
   * @param allergy Optional parameter that is used to prefill the FormGroup's control values.
   */
  addAllergyFormGroup(allergy?: MedicalCondition) {
    let name = allergy?.name ? allergy.name : '';
    let altName = allergy?.altName ? allergy.altName : '';

    const formGroup = this.fb.group({
      name: [name, Validators.required],
      altName: [altName],
    });

    this.allergiesFormArray.push(formGroup);
  }

  /**
   * Deletes the FormGroup at the specified index in the specified FormArray.
   *
   * @param index The index of the FormGroup to delete in the FormArray.
   */
  deleteFormGroup(index: number, type: string) {
    switch (type.toLocaleLowerCase()) {
      case 'team': {
        this.interdisciplinaryTeamFormArray.removeAt(index);
        break;
      }
      case 'problem': {
        this.activeMedProblemsFormArray.removeAt(index);
        break;
      }
      case 'allergy': {
        this.allergiesFormArray.removeAt(index);
        break;
      }
      default: {
        console.log(type, 'is not a valid FormArray descriptor!');
      }
    }
  }

  /**
   * Get the image selected as a File object, and store the object and its name.
   *
   * @param event Object passed from p-fileUpload containing file information.
   */
  onFileChange(event: any) {
    let files = event.currentFiles as File[];
    if (files && files.length > 0) {
      this.profilePic = event.currentFiles[0];
      this.profilePicNameFormControl.setValue(this.profilePic?.name);
    } else {
      this.profilePic = undefined;
      this.profilePicNameFormControl.setValue('');
    }
  }

  /**
   *
   * @returns A Patient object using the data from the form.
   */
  getPatientFromForm() {
    // Format form as Patient object
    let patient = new Patient();

    // Change height in feet to height in inches, store only inches in patient
    const heightInches = this.heightFeetFormControl.value * 12 + this.heightInchesFormControl.value;

    // Remove heightFeet from the object, since we converted it to inches
    let formValue = this.patientForm.getRawValue();
    delete formValue.heightFeet;

    // Assign form value to Patient object
    Object.assign(patient, formValue);

    // Set specific fields
    patient.setHeight(heightInches);
    patient.setDOB(this.getDOBTimestamp());
    patient.setMRN(patient.id);

    return patient;
  }

  /**
   * Create patient using the patient form and stores height information as inch unit.
   */
  upsertPatient(): void {
    this.patient = this.getPatientFromForm();

    if (!this.patientId) {
      // Add patient to user's patient list
      this.userService.addPatientIdToUser(this.patient.getId(), this.authService.currentUser.id);
    }

    // Send formatted patient to PatientService
    this.patientService.upsertPatient(this.patient, COLLECTION_PATIENTS, this.profilePic).subscribe(() => {
      this.router.navigate(['patients', this.patient.id]);
      this.patientForm.reset();
    });
  }

  /**
   *
   * @returns The date of birth as a Timestamp object.
   */
  getDOBTimestamp() {
    let date = new Date(this.dobFormControl.value);
    return Timestamp.fromDate(date);
  }

  get isCreateType() {
    return this.type === 'create-patient';
  }

  get isEditType() {
    return this.type === 'edit-patient';
  }

  get pageTitle() {
    if (this.isCreateType) {
      return 'Create Patient';
    } else {
      return 'Edit Patient';
    }
  }

  get submitButtonLabel() {
    if (this.isCreateType) {
      return 'Create Patient';
    } else {
      return 'Update Patient';
    }
  }

  get firstNameFormControl(): UntypedFormControl {
    return this.patientForm.controls.firstName as UntypedFormControl;
  }

  get middleNameFormControl(): UntypedFormControl {
    return this.patientForm.controls.middleName as UntypedFormControl;
  }

  get lastNameFormControl(): UntypedFormControl {
    return this.patientForm.controls.lastName as UntypedFormControl;
  }

  get idFormControl(): UntypedFormControl {
    return this.patientForm.controls.id as UntypedFormControl;
  }

  get generalInfoFormGroup(): UntypedFormGroup {
    return this.patientForm.get('generalInfo') as UntypedFormGroup;
  }

  get codeStatusFormControl(): UntypedFormControl {
    return this.patientForm.get('generalInfo.codeStatus') as UntypedFormControl;
  }

  get dobFormControl(): UntypedFormControl {
    return this.patientForm.get('generalInfo.dob') as UntypedFormControl;
  }

  get ageFormControl(): UntypedFormControl {
