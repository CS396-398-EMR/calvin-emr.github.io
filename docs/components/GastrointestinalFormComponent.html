import { Component, OnInit } from '@angular/core';
import { AssessmentForm } from 'src/app/shared/assessment-form.component';
import { UntypedFormControl } from '@angular/forms';
import { bowelSoundsOptions, emesisOptions, outputOptions, stomaSiteOptions } from './gastrointestinal-form.metadata';

@Component({
  selector: 'app-gastrointestinal-form',
  templateUrl: './gastrointestinal-form.component.html',
  styleUrls: ['./gastrointestinal-form.component.scss', '../assessments-tab.component.scss'],
})
export class GastrointestinalFormComponent extends AssessmentForm implements OnInit {
  // Options for DatePicker
  minDate: Date;
  maxDate: Date;

  constructor() {
    super();
    const currentYear = new Date().getFullYear();
    this.minDate = new Date(currentYear - 20, 0, 1);
    this.maxDate = new Date();
    this.reset();
  }

  ngOnInit(): void {}

  reset() {
    this.controlList = [
      { control: 'ruq', default: 'WNL', wnl: false },
      { control: 'rlq', default: 'WNL', wnl: false },
      { control: 'luq', default: 'WNL', wnl: false },
      { control: 'llq', default: 'WNL', wnl: false },
      { control: 'nausea', default: false, wnl: false },
      { control: 'emesis', default: 'WNL', wnl: false },
      { control: 'incontinence', default: false, wnl: false },
      { control: 'stomaSite', default: 'None', wnl: false },
      { control: 'output', default: 'Formed', wnl: false },
    ];
  }

  /**
   * Bowel Sounds
   */

  ruqIndex = 0;
  rlqIndex = 1;
  luqIndex = 2;
  llqIndex = 3;

  bowelSoundsOptions = bowelSoundsOptions;

  get bowelSoundsDefault() {
    return this.controlList[0].default;
  }

  get ruqFormControl(): UntypedFormControl {
    return this.formGroup.get('bowelSounds.ruq') as UntypedFormControl;
  }

  get rlqFormControl(): UntypedFormControl {
    return this.formGroup.get('bowelSounds.rlq') as UntypedFormControl;
  }

  get luqFormControl(): UntypedFormControl {
    return this.formGroup.get('bowelSounds.luq') as UntypedFormControl;
  }

  get llqFormControl(): UntypedFormControl {
    return this.formGroup.get('bowelSounds.llq') as UntypedFormControl;
  }

  /**
   * Other
   */

  nauseaIndex = 4;
  emesisIndex = 5;
  bowelControlIndex = 6;

  emesisOptions = emesisOptions;

  get nauseaDefault() {
    return this.controlList[this.nauseaIndex].default;
  }

  get emesisDefault() {
    return this.controlList[this.emesisIndex].default;
  }

  get bowelControlDefault() {
    return this.controlList[this.bowelControlIndex].default;
  }

  get dateLastBMFormControl(): UntypedFormControl {
    return this.formGroup.controls.dateLastBM as UntypedFormControl;
  }

  get nauseaFormControl(): UntypedFormControl {
    return this.formGroup.controls.nausea as UntypedFormControl;
  }

  get emesisFormControl(): UntypedFormControl {
    return this.formGroup.controls.emesis as UntypedFormControl;
  }

  get incontinenceFormControl(): UntypedFormControl {
    return this.formGroup.controls.incontinence as UntypedFormControl;
  }

  get hemorrhoidsFormControl(): UntypedFormControl {
    return this.formGroup.controls.hemorrhoids as UntypedFormControl;
  }

  /**
   * Ostomy
   */

  stomaSiteIndex = 7;
  outputIndex = 8;

  stomaSiteOptions = stomaSiteOptions;

  outputOptions = outputOptions;

  get stomaSiteDefault() {
    return this.controlList[this.stomaSiteIndex].default;
  }

  get outputDefault() {
    return this.controlList[this.outputIndex].default;
  }

  get stomaSiteFormControl(): UntypedFormControl {
    return this.formGroup.get('ostomy.stomaSite') as UntypedFormControl;
  }

  get outputFormControl(): UntypedFormControl {
    return this.formGroup.get('ostomy.output') as UntypedFormControl;
  }
}
