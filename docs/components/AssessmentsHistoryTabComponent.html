import { Component, Input, OnInit } from '@angular/core';
import { MenuItem } from 'primeng/api';
import { HOUR24_DATE_CONFIG } from 'src/app/global';
import { PatientDashboardStore } from 'src/app/patient-dashboard/state/patient-dashboard.store';
import { PatientDashboardTabComponent } from 'src/app/patient-dashboard/dashboard-tab/dashboard-tab.component';
import { AuthService } from 'src/app/shared/services/auth.service';
import { Assessment } from 'src/app/shared/models/assessment.model';
import { Patient } from 'src/app/shared/models/patient.model';

@Component({
  selector: 'app-assessments-history-tab',
  templateUrl: './assessments-history-tab.component.html',
  styleUrls: ['./assessments-history-tab.component.scss'],
})
export class AssessmentsHistoryTabComponent extends PatientDashboardTabComponent implements OnInit {
  assessments: Assessment[] = [];
  menuItems: MenuItem[] = [];
  selectedAssessment: Assessment = new Assessment();

  constructor(patientStore: PatientDashboardStore, authService: AuthService) {
    super(patientStore, authService);
    this.patient$.subscribe((patient: Patient) => {
      this.assessments = patient.assessments;

      this.menuItems = this.assessments.map((assessment) => {
        return {
          label: this.getAssessmentDateString(assessment),
          command: () => {
            this.selectedAssessment = assessment;
          },
        };
      });

      this.menuItems.sort((a, b) => {
        if (a.label && b.label) {
          return b.label.localeCompare(a.label);
        }
        return 1;
      });

      if (this.hasAssessments) {
        this.selectedAssessment = this.assessments[0];
      }
    });
  }

  ngOnInit(): void {}

  getAssessmentDateString(assessment: Assessment) {
    return assessment.date.toDate().toLocaleString([], HOUR24_DATE_CONFIG).replace(',', ' @');
  }

  get hasAssessments(): boolean {
    return this.assessments.length > 0;
  }

  get selectedAssessmentDateString(): string {
    if (this.hasAssessments && this.selectedAssessment) {
      return this.selectedAssessment?.date.toDate().toLocaleString([], HOUR24_DATE_CONFIG).replace(',', ' at');
    }
    return 'NULL';
  }

  get cardiovascular() {
    return this.selectedAssessment.cardiovascular;
  }

  get respiratory() {
    return this.selectedAssessment.respiratory;
  }

  get neurological() {
    return this.selectedAssessment.neurological;
  }

  get musculoskeletal() {
    return this.selectedAssessment.musculoskeletal;
  }

  get gastrointestinal() {
    return this.selectedAssessment.gastrointestinal;
  }

  get genitourinary() {
    return this.selectedAssessment.genitourinary;
  }

  get psychosocial() {
    return this.selectedAssessment.psychosocial;
  }

  get integumentary() {
    return this.selectedAssessment.integumentary;
  }
}
