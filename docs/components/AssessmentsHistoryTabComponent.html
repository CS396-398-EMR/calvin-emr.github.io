import { Component, Input, OnInit } from '@angular/core';
import { hour24DateOptions } from 'src/app/global';
import { Assessment } from 'src/app/types/assessment.type';

@Component({
  selector: 'app-assessments-history-tab',
  templateUrl: './assessments-history-tab.component.html',
  styleUrls: ['./assessments-history-tab.component.scss']
})
export class AssessmentsHistoryTabComponent implements OnInit {

  @Input() assessments: Assessment[] = [];

  displayedColumns = ["date/time"]
  selectedRow: number = 0;

  constructor() { }

  ngOnInit(): void {
  }

  selectRow(row: any, index: number): void {
    this.selectedRow = index
  }

  getAssessmentDateString(assessment: Assessment) {
    return assessment.date
      .toDate()
      .toLocaleString([], hour24DateOptions)
      .replace(',', ' @'); 
  }

  get hasAssessments(): boolean {
    return this.assessments.length > 0;
  }

  get selectedAssessmentDateString(): string {
    return this.assessments[this.selectedRow].date
      .toDate()
      .toLocaleString([], hour24DateOptions)
      .replace(',', ' at');
  }

  get cardiovascular() {
    return this.assessments[this.selectedRow].cardiovascular;
  }

  get respiratory() {
    return this.assessments[this.selectedRow].respiratory;
  }

  get neurological() {
    return this.assessments[this.selectedRow].neurological;
  }

  get musculoskeletal() {
    return this.assessments[this.selectedRow].musculoskeletal;
  }

  get gastrointestinal() {
    return this.assessments[this.selectedRow].gastrointestinal;
  }

  get genitourinary() {
    return this.assessments[this.selectedRow].genitourinary;
  }

  get psychosocial() {
    return this.assessments[this.selectedRow].psychosocial;
  }

  get integumentary() {
    return this.assessments[this.selectedRow].integumentary;
  }
}
