import { Component, OnInit } from '@angular/core';
import { AssessmentForm } from 'src/app/shared/assessment-form.component';
import { FormOption } from 'src/app/types/form-option.type';
import { FormControl, FormGroup } from '@angular/forms';

@Component({
  selector: 'app-respiratory-form',
  templateUrl: './respiratory-form.component.html',
  styleUrls: ['./respiratory-form.component.scss', '../assessments-tab.component.scss']
})
export class RespiratoryFormComponent extends AssessmentForm implements OnInit {

  constructor() { 
    super() 
    this.controlList = [
      {control: "quality", default: "WNL", wnl: false},
      {control: "rul", default: "WNL", wnl: false},
      {control: "rll", default: "WNL", wnl: false},
      {control: "rml", default: "WNL", wnl: false},
      {control: "lul", default: "WNL", wnl: false},
      {control: "lll", default: "WNL", wnl: false},
    ]
  }

  ngOnInit(): void {
    // Create listener on cough form control to reset cough Description form control value
    this.coughFormControl.valueChanges.subscribe(() => {
      // Only reset the other controls if the current value is false
      if(this.coughFormControl.value == false) {
        this.coughDescriptionFormControl.setValue('');
        Object.keys(this.productiveDescriptionFormGroup.controls).forEach(key => {
          this.productiveDescriptionFormGroup.controls[key].setValue(false);
        });
      }
    })

    // Clear the "Other Lung Sound" controls if the user switches from "Other" to a different option
    this.rulFormControl.valueChanges.subscribe((value: string) => {
      if(value != "Other") {
        this.otherRULFormControl.setValue('');
      }
    });
    this.rmlFormControl.valueChanges.subscribe((value: string) => {
      if(value != "Other") {
        this.otherRMLFormControl.setValue('');
      }
    });
    this.rllFormControl.valueChanges.subscribe((value: string) => {
      if(value != "Other") {
        this.otherRLLFormControl.setValue('');
      }
    });
    this.lulFormControl.valueChanges.subscribe((value: string) => {
      if(value != "Other") {
        this.otherLULFormControl.setValue('');
      }
    });
    this.lllFormControl.valueChanges.subscribe((value: string) => {
      if(value != "Other") {
        this.otherLLLFormControl.setValue('');
      }
    });
  }

  /**
   * Respiratory Pattern
   */

  qualityIndex = 0;

  qualityOptions = [
    new FormOption("Dyspnea", "dyspnea"),
    new FormOption("Tachypnea", "tachypnea"),
    new FormOption("Bradypnea", "bradypnea"),
    new FormOption("Apnea", "apnea"),
    new FormOption("WNL"),
  ]

  get qualityDefault() {
    return this.controlList[this.qualityIndex].default;
  }

  get rhythmFormControl(): FormControl {
    return this.formGroup.get("respiratoryPattern.irregularRhythm") as FormControl;
  }

  get qualityFormControl(): FormControl {
    return this.formGroup.get("respiratoryPattern.quality") as FormControl;
  }

  /**
   * Lung Sounds
   */

  rulIndex = 1;
  rllIndex = 2;
  rmlIndex = 3;
  lulIndex = 4;
  lllIndex = 5;

  lungSoundOptions = [
    new FormOption("Coarse rhonchi"),
    new FormOption("Crackles/rales"),
    new FormOption("Wheezes expiratory"),
    new FormOption("Wheezes inspiratory"),
    new FormOption("Other"),
    new FormOption("WNL"),
  ]

  get lungSoundDefault() {
    return this.controlList[1].default;
  }

  get rulFormControl(): FormControl {
    return this.formGroup.get("lungSounds.rul") as FormControl;
  }

  get otherRULFormControl(): FormControl {
    return this.formGroup.get("lungSounds.otherRul") as FormControl;
  }

  get isOtherRUL() {
    return this.rulFormControl.value == "Other";
  }

  get rllFormControl(): FormControl {
    return this.formGroup.get("lungSounds.rll") as FormControl;
  }

  get otherRLLFormControl(): FormControl {
    return this.formGroup.get("lungSounds.otherRll") as FormControl;
  }

  get isOtherRLL() {
    return this.rllFormControl.value == "Other";
  }

  get rmlFormControl(): FormControl {
    return this.formGroup.get("lungSounds.rml") as FormControl;
  }

  get otherRMLFormControl(): FormControl {
    return this.formGroup.get("lungSounds.otherRml") as FormControl;
  }

  get isOtherRML() {
    return this.rmlFormControl.value == "Other";
  }

  get lulFormControl(): FormControl {
    return this.formGroup.get("lungSounds.lul") as FormControl;
  }

  get otherLULFormControl(): FormControl {
    return this.formGroup.get("lungSounds.otherLul") as FormControl;
  }

  get isOtherLUL() {
    return this.lulFormControl.value == "Other";
  }

  get lllFormControl(): FormControl {
    return this.formGroup.get("lungSounds.lll") as FormControl;
  }

  get otherLLLFormControl(): FormControl {
    return this.formGroup.get("lungSounds.otherLll") as FormControl;
  }

  get isOtherLLL() {
    return this.lllFormControl.value == "Other";
  }

  /**
   * Cough
   */

  coughDescriptionOptions = [
    new FormOption("Productive"),
    new FormOption("Nonproductive"),
    new FormOption("Dry"),
  ]

  productiveDescriptionOptions = [
    new FormOption("Scant", "scant"),
    new FormOption("Moderate", "moderate"),
    new FormOption("Large", "large"),
    new FormOption("Purulent", "purulent"),
    new FormOption("Blood tinged", "bloodTinged"),
    new FormOption("Black", "black"),
    new FormOption("Green", "green"),
    new FormOption("Yellow", "yellow"),
    new FormOption("Thick", "thick"),
    new FormOption("Thin", "thin"),
    new FormOption("Tenacious", "tenacious"),
  ]

  get coughFormControl(): FormControl {
    return this.formGroup.controls.cough as FormControl;
  }

  get coughDescriptionFormControl(): FormControl {
    return this.formGroup.controls.coughDescription as FormControl;
  }

  get productiveDescriptionFormGroup(): FormGroup {
    return this.formGroup.controls.productiveDescription as FormGroup;
  }

  get isCoughPresent(): boolean {
    return this.coughFormControl.value;
  }

  get isCoughProductive(): boolean {
    return (this.coughDescriptionFormControl.value as string).toLocaleLowerCase() == "productive";
  }
}
