import { Component, Input, OnInit } from '@angular/core';
import { TabComponent } from 'src/app/shared/tab.component';
import { ProviderOrder } from 'src/app/types/order.type';
import { ProgressNote } from 'src/app/types/progress-note.type';
import { Patient } from 'src/app/types/patient.type';
import { MatDialog } from '@angular/material/dialog';
import { Update } from 'src/app/types/update.type';

@Component({
  selector: 'app-provider-charts-tab',
  templateUrl: './provider-charts-tab.component.html',
  styleUrls: ['./provider-charts-tab.component.scss']
})
export class ProviderChartsTabComponent extends TabComponent implements OnInit {

  @Input() providerOrders: ProviderOrder[] = [];
  @Input() progressNotes: ProgressNote[] = [];
  @Input() historyAndPhysical: Patient = new Patient();

  constructor(public dialog: MatDialog) {
    super();
  }

  emitUpdateToParent(update: Update) {
    this.updatePatientEvent.emit(update);
  } 
}
