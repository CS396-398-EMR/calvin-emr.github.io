import { Component, OnInit, Provider } from '@angular/core';
import { OrderEditDialogComponent } from 'src/app/patient-dashboard/provider-orders-tab/order-edit-dialog/order-edit-dialog.component';
import { PatientDashboardTabComponent } from 'src/app/patient-dashboard/dashboard-tab/dashboard-tab.component';
import { ProviderOrder } from 'src/app/shared/models/provider-order.model';
import { AuthService } from 'src/app/shared/services/auth.service';
import { DialogService } from 'primeng/dynamicdialog';
import { PatientDashboardStore } from 'src/app/patient-dashboard/state/patient-dashboard.store';
import { Patient } from 'src/app/shared/models/patient.model';

@Component({
  selector: 'app-provider-orders-tab',
  templateUrl: './provider-orders-tab.component.html',
  styleUrls: ['./provider-orders-tab.component.scss'],
  providers: [DialogService],
})
export class ProviderOrdersTabComponent extends PatientDashboardTabComponent implements OnInit {
  providerOrders: ProviderOrder[] = [];

  constructor(patientStore: PatientDashboardStore, authService: AuthService, public dialogService: DialogService) {
    super(patientStore, authService);
    this.patient$.subscribe((patient: Patient) => {
      this.providerOrders = patient.providerOrders;
    });
  }

  ngOnInit(): void {}

  viewOrder(order: ProviderOrder) {
    // this.dialogService.open()
  }

  addOrder(order: ProviderOrder) {
    this.providerOrders.push(order);
    this.updatePatientField('providerOrders', this.providerOrders);
  }

  editOrder(order: ProviderOrder) {
    const dialogRef = this.dialogService.open(OrderEditDialogComponent, {
      header: 'Edit Order',
      data: order,
      width: '75%',
    });

    dialogRef.onClose.subscribe((newOrder: ProviderOrder) => {
      if (newOrder) {
        const indexToRemove = this.providerOrders.findIndex((o) => o.id == order.id);
        this.providerOrders.splice(indexToRemove, 1, newOrder);
        this.updatePatientField('providerOrders', this.providerOrders);
      }
    });
  }

  deleteOrder(order: ProviderOrder) {
    this.providerOrders = this.providerOrders.filter((o) => {
      return o != order;
    });
    this.updatePatientField('providerOrders', this.providerOrders);
  }
}
