import { Component, Input, OnInit } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { Timestamp } from 'firebase/firestore';
import { editActionName, deleteActionName, undoDeleteActionName, providerOrdersField } from 'src/app/global';
import { UserService } from 'src/app/services/user.service';
import { OrderEditDialog } from 'src/app/shared/order-edit-dialog/order-edit-dialog.component';
import { TabComponent } from 'src/app/shared/tab.component';
import { ProviderOrder } from 'src/app/types/order.type';
import { ActionName, TableAction, TableEvent } from 'src/app/types/table-action.enum';
import { TableColumn } from 'src/app/types/table-column.type';
import { Update } from 'src/app/types/update.type';

@Component({
  selector: 'app-provider-orders-tab',
  templateUrl: './provider-orders-tab.component.html',
  styleUrls: ['./provider-orders-tab.component.scss']
})
export class ProviderOrdersTabComponent extends TabComponent implements OnInit {

 /**
  * Columns passed into the emr-table used to display provider orders.
  */
  columns: TableColumn[] = [
    new TableColumn("Medication", "medName",
      (element: ProviderOrder) => element.medName),
    new TableColumn("Dosage", "dosage",
      (element: ProviderOrder) => element.dosage),
    new TableColumn("Quantity Given", "quantityGiven",
      (element: ProviderOrder) => element.quantityGiven),
    new TableColumn("Quantity", "quantity",
      (element: ProviderOrder) => element.quantity),
    new TableColumn("Route", "route",
      (element: ProviderOrder) => element.route),
    new TableColumn("Order", "order",
      (element: ProviderOrder) => element.order),
    new TableColumn("When to Give", "startDate",
      (element: ProviderOrder) => {
        return this.getDateString(element.startDate.toDate());
      }),
    new TableColumn("Latest to Give", "endDate",
    (element: ProviderOrder) => {
      return this.getDateString(element.endDate.toDate());
    }),
    new TableColumn("Signature", "signature", 
      (element: ProviderOrder) => element.signature),
    new TableColumn("Administered", "administered",
      (element: ProviderOrder) => element.getAdministeredString()),
  ];

  /**
   * List of actions to display in the emr-table.
   */
  actions: TableAction[] = [];

  
  @Input() providerOrders: ProviderOrder[] = [];

  constructor(private userService: UserService, public dialog: MatDialog) { 
    super();
    if(this.isAdmin) {
      this.columns.push(
        new TableColumn("Created", "createdDate",
        (element: ProviderOrder) => {
          return this.getDateString(element.dateCreated.toDate());
        })
      )
    }
  }

  ngOnInit(): void {
  }

  ngOnChanges(): void {
    // EMR table only has actions if user is admin
    if (this.isAdmin) {
      this.actions = [
        new TableAction(editActionName, "edit"),
        new TableAction(deleteActionName, "delete", undoDeleteActionName, "undo"),
      ]
    }
    else {
      this.actions = []
    }
    
    this.stopLoading();
  }

  handleActionClick(event: TableEvent) {
    if (event.action.name == ActionName.Edit) {
      this.editOrder(event.data as ProviderOrder);
    }
    else if (event.action.name == ActionName.Delete) {
      this.deleteOrder(event.data as ProviderOrder)
    }

    this.updateProviderOrders();
  }

  addOrder(order: ProviderOrder) {
    this.providerOrders.push(order);
    this.updateProviderOrders()
  }

  editOrder(order: ProviderOrder) {
    const dialogRef = this.dialog.open(OrderEditDialog, {
      data: order,
      disableClose: true,
    });

    dialogRef.afterClosed().subscribe((newOrder: ProviderOrder) => {
      if(newOrder) {
        let indexToRemove = this.providerOrders.findIndex(o => o.id == order.id)
        this.providerOrders.splice(indexToRemove, 1, newOrder)
        this.updateProviderOrders();
      }
    })
  }

  deleteOrder(order: ProviderOrder) {
    this.providerOrders = this.providerOrders.filter((o) => {
      return o != order;
    })
  }

  updateProviderOrders(): void {
    // Merge providerOrders array with empty array - this triggers the
    //  ngOnChanges() in the emr-table component, since pushing to an
    //  array is not seen as a change by onChanges()
    this.providerOrders = ([] as any[]).concat(this.providerOrders);
    this.updatePatient(providerOrdersField, this.providerOrders);
  }

  updatePatient(field: string, data: any[]): void {
    // Create new update object
    let update = new Update(field, data);

    // Reformat update to JSON
    update = JSON.parse(JSON.stringify(update))

    // Reformat datestrings in data to Timestamp
    update.data.forEach((obj: any) => {
      obj.dateCreated = Timestamp.fromMillis(obj.dateCreated.seconds * 1000)
      obj.startDate = Timestamp.fromMillis(obj.startDate.seconds * 1000);
      obj.endDate = Timestamp.fromMillis(obj.endDate.seconds * 1000);
    });

    this.updatePatientEvent.emit(update);
  }

  get isAdmin() {
    return this.userService.isCurrentUserAdmin;
  }

}

<app-order-form *ngIf="isAdmin"
  (newOrderEvent)="addOrder($event)">
</app-order-form>
<app-emr-table
  [tableData]="providerOrders"
  [tableColumns]="columns"
  [tableActions]="actions"
  (rowAction)="handleActionClick($event)"
  [paginated]="true"
>  
</app-emr-table>
