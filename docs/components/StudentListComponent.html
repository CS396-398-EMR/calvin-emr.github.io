  /**
   * Compare two objects and return the differences and common values
   * Potentially used for the updating studentPatient object in firebase
   * Code URL: https://newbedev.com/javascript-typescript-get-difference-between-two-objects-code-example
   *
   * @param origObj   Object 1 for comparison
   * @param newObj    Object 2 for comparison
   * @returns object type that contains difference or commonality of two objects
   */

  objectDifference(origObj: Object, newObj: Object) {
    let changes = (newObj: Object, origObj: Object) => {
      let arrayIndexCounter = 0;
      return transform(newObj, function (result: any, value, key) {
        if (!isEqual(value, origObj[key])) {
          let resultKey = isArray(origObj) ? arrayIndexCounter++ : key;
          result[resultKey] = isObject(value) && isObject(origObj[key]) ? changes(value, origObj[key]) : value;
        }
      });
    };
    return changes(newObj, origObj);
  }

  objectSame(origObj: Object, newObj: Object) {
    let changes = (newObj: Object, origObj: Object) => {
      let arrayIndexCounter = 0;
      return transform(newObj, function (result: any, value, key) {
        if (isEqual(value, origObj[key])) {
          let resultKey = isArray(origObj) ? arrayIndexCounter++ : key;
          result[resultKey] = isObject(value) && isObject(origObj[key]) ? changes(value, origObj[key]) : value;
        }
      });
    };
    return changes(newObj, origObj);
  }

  /**
   *
   * @param result The string created by the FileReader.
   */
  getStudentsFromFile(result: string) {
    let studentArray: User[] = [];

    // Clear any previous import errors
    this.importErrors = [];

    // Ignore first row, since first row is column headers
    let newStudents = result.split('\r').slice(1);

    newStudents.forEach((studentInfoString) => {
      // Try to add data; if data is not valid, add to errors array
      try {
        let params = studentInfoString
          .split(',')
          .filter((x) => x != '')
          .map((x) => x.trim());
        let student = new User(...params);
        if (!this.isValidUser(student)) {
          throw new Error();
        }
        studentArray.push(student);
      } catch (e) {
        this.importErrors.push(studentInfoString);
      }
    });

    // Display errors, if any
    if (this.importErrors.length > 0) {
      let errorString = this.importErrors.join('\n');
      console.error(
        'Could not import the following data:',
        errorString,
        '\nVerify that your file is correctly formatted and that no data is missing.'
      );
      // TODO: make this a dialog box
    }

    return studentArray;
  }

  importStudents(studentArray: User[]) {
    this.toastService.clear();
    studentArray.forEach((student: User) => {
      this.userService.createUserAccount(student);
      forkJoin([this.userService.createUserAccount(student), this.userService.createUserDocument(student)]).subscribe(
        (res) => {
          console.log(res);
          this.toastService.addOne({
            severity: ERROR,
            summary: 'Error',
            detail: `Unable to create account for: ${student.id} ${student.firstNameLastName}!`,
            life: 5000,
          });
        }
      );
    });
  }

  isValidUser(student: User) {
    return student.hasValidEmail;
  }

  get numStudents() {
    let students = this.currentUser.studentIds;
    return students ? students.length : 0;
  }
