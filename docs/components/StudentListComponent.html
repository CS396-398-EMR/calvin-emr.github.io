   * Interprets the event passed by the user clicking a button in the table.
   * Essentially, it calls a specific function based on the button pressed.
   * 
   * @param event Specifies the action to carry out and the Medication object to act upon.
   */
   handleAction(event?: TableEvent) {
    // If event was not specified, assume action is an "Add Student"
    if (!event) {
      // Create new TableEvent for "Add Student" operation
      // Pass a new student and the list of student IDs for validation
      event = new TableEvent(
        new TableAction(ActionName.Add, "add"),
        { student: new User(), ids: this.ids }
      )
    }
    // If the event is specified (edit or delete), don't pass any IDs since
    // they are not needed
    else {
      // Translate button labels into ActionName enumeration and update the action name 
      //  in the TableEvent to pass to the dialog
      const action = this.getActionName(event.action.name);
      event = new TableEvent(
        new TableAction(action, ""),
        { student: event.data, ids: []}
      )
    }

    this.openStudentDialog(event);
  }

  /**
   * Opens a StudentDialogComponent. Passes a TableEvent object as data to the dialog.
   * The TableEvent object contains:
   *    (1) a TableAction object, and
   *    (2) an object containing a User object and an array of all existing student IDs.
   * The array of IDs is used in the "Add" action to check an inputted user's ID
   * against currently existing user IDs. This is to ensure the admin does not
   * enter an aleady-existing user ID.
   * After the dialog is closed, executes a REST operation on the current user
   * based on what action was selected.
   * 
   * @param event Specifies the action to carry out and the User object to act upon.
   */
  openStudentDialog(event: TableEvent) {
    const dialogRef = this.dialog.open(StudentDialogComponent, {
      data: event
    })

    // Execute REST operations based on action
    dialogRef.afterClosed().subscribe((student: User) => {
      if(student) {
        switch(event.action.name) {
          case ActionName.Add: {
            this.userService.createUser(student);
            break;
          }
          case ActionName.Edit: {
            this.userService.updateUser(student);
            break;
          }
          case ActionName.Delete: {
            this.userService.deleteUser(student.id);
            break;
          }
          default: {
            console.log(event.action.name, "is not a valid action!")
          }
        }
      }
    })
  }

  /**
   * Compare two objects and return the differences and common values
   * Potentially used for the updating studentPatient object in firebase
   * Code URL: https://newbedev.com/javascript-typescript-get-difference-between-two-objects-code-example
   * 
   * @param origObj   Object 1 for comparison
   * @param newObj    Object 2 for comparison
   * @returns object type that contains difference or commonality of two objects
   */

  objectDifference(origObj:Object, newObj:Object) {
    let changes = (newObj:Object, origObj:Object) => {
      let arrayIndexCounter = 0;
      return transform(newObj, function(result: any, value, key) {
        if (!isEqual(value, origObj[key])) {
          let resultKey = isArray(origObj) ? arrayIndexCounter++ : key
          result[resultKey] = (isObject(value) && isObject(origObj[key])) ? changes(value, origObj[key]) : value
        }
      })
    }
    return changes(newObj, origObj)
  }

  objectSame(origObj:Object, newObj:Object) {
    let changes = (newObj:Object, origObj:Object) => {
      let arrayIndexCounter = 0;
      return transform(newObj, function(result: any, value, key) {
        if (isEqual(value, origObj[key])) {
          let resultKey = isArray(origObj) ? arrayIndexCounter++ : key
          result[resultKey] = (isObject(value) && isObject(origObj[key])) ? changes(value, origObj[key]) : value
        }
      })
    }
    return changes(newObj, origObj)
  }

    let students = this.currentUser.studentIds;
    return students ? students.length : 0;
  }
}
