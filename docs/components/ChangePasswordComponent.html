import { Component, OnInit, SimpleChange } from '@angular/core';
import { FormBuilder, FormControl, Validators } from '@angular/forms';
import { Router } from '@angular/router';
import { getAuth, updatePassword } from 'firebase/auth';

@Component({
  selector: 'app-change-password',
  templateUrl: './change-password.component.html',
  styleUrls: ['./change-password.component.scss']
})
export class ChangePasswordComponent implements OnInit {

  // Form controls for password form
  passwordForm = this.fb.group({
    password: [''],
    retypedPassword: ['']
  })

  // Booleans for hiding password text
  hidePassword = true;
  hideRetypedPassword = true;

  // Boolean for error messages
  submitted = false;      // false if the user is not clicking the submit button

  // Error message to display on client end
  error: string = "";

  constructor(private fb: FormBuilder,
              private router: Router) { }

  ngOnInit(): void {
    // Set submitted to false whenever the form's value changes;
    // this gets rid of any error messages when the user starts typing
    this.passwordForm.valueChanges.subscribe(() => {
      this.submitted = false;
    })
  }

  onSubmit() {
    // If password is valid, update it with new value
    if(this.passwordsMatch && this.isValidLength) {
      // Get current user information
      const auth = getAuth();
      const user = auth.currentUser;

      if(user) {
        // Update user's password with new password information
        updatePassword(user, this.password.value).then(() => {
          // Update successful
          this.router.navigateByUrl('/patients');
        }).catch((error) => {
          error = error.message;
        });
      }
    }
    // If password is not valid, display any error messages
    else {
      this.submitted = true;
    }
  }

  /**
   * Returns various error messages to be displayed on client side.
   * The error messages are checked below in descending precedence.
   */
  getError(): string {
    if(!this.isValidLength) {
      return "Password must be between 6 and 32 characters in length!"
    }
    if(!this.passwordsMatch) {
      return "Passwords do not match!"
    }
    return "Something went wrong - try again."
  }

  get password(): FormControl {
    return this.passwordForm.controls.password as FormControl;
  }

  get retypedPassword(): FormControl {
    return this.passwordForm.controls.retypedPassword as FormControl;
  }
  
  get passwordsMatch() {
    return this.password.value === this.retypedPassword.value;
  }

  get isValidLength() {
    let pwdLength = (this.password.value as String).length;
    return pwdLength >= 6 && pwdLength <= 32;
  }

}
