import { Component, OnInit } from '@angular/core';
import { DialogService } from 'primeng/dynamicdialog';
import { Observable } from 'rxjs';
import { takeUntil } from 'rxjs/operators';
import { Patient } from 'src/app/shared/models/patient.model';
import { ProviderOrder } from 'src/app/shared/models/provider-order.model';
import { PatientDashboardStore } from '../../state/patient-dashboard.store';
import { AdministerMedDialogComponent } from '../administer-med-dialog/administer-med-dialog.component';

@Component({
  selector: 'app-administration-schedule',
  template: ` <p>Administration schedule works!</p> `,
  styles: [],
  providers: [DialogService],
})
export class AdministrationScheduleComponent implements OnInit {
  patient$: Observable<Patient> = this.patientStore.patient$.pipe(takeUntil(this.patientStore.destroy$));
  patient: Patient = new Patient();
  orders: ProviderOrder[] = [];

  constructor(private patientStore: PatientDashboardStore, private dialogService: DialogService) {
    this.patient$.subscribe((patient) => {
      this.patient = patient;
    });
  }

  ngOnInit(): void {}

  openAdministerMedDialog(order: ProviderOrder): void {
    const dialogRef = this.dialogService.open(AdministerMedDialogComponent, {
      header: `Administering medication for: ${this.patient.getNameFirstMiddleLast()}`,
      data: { patient: this.patient, order: order },
      width: '75%',
      closable: false,
    });

    dialogRef.onClose.subscribe((res: { order: ProviderOrder; administered: boolean }) => {
      if (res.order && res.administered) {
        let indexToSplice = this.patient.providerOrders.findIndex((o) => o.id == order.id);
        this.patient.providerOrders.splice(indexToSplice, 1, order);
        this.patientStore.setField({ field: 'providerOrders', data: this.patient.providerOrders });
      }
    });
  }
}
