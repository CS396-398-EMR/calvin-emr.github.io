import { Component, OnInit, ViewChild } from '@angular/core';
import { PatientDashboardTabComponent } from 'src/app/patient-dashboard/dashboard-tab/dashboard-tab.component';
import { WARN } from 'src/app/global';
import { ToastService } from 'src/app/shared/services/toast.service';
import { UntypedFormControl } from '@angular/forms';
import { AuthService } from 'src/app/shared/services/auth.service';
import { Patient } from 'src/app/shared/models/patient.model';
import { PatientDashboardStore } from 'src/app/patient-dashboard/state/patient-dashboard.store';
import { DialogService } from 'primeng/dynamicdialog';
import { MedicationScheduleComponent } from '../medication-schedule/medication-schedule.component';

@Component({
  selector: 'app-mar-tab',
  templateUrl: './mar-tab.component.html',
  styleUrls: ['./mar-tab.component.scss'],
  providers: [DialogService],
})
export class MarTabComponent extends PatientDashboardTabComponent implements OnInit {
  /**
   * Flag indicating if the user has scanned the barcode of the patient.
   * This is set to true if the user is an admin.
   */
  isVerified: boolean;

  /**
   * The current patient's MRN.
   */
  wristbandId: UntypedFormControl = new UntypedFormControl('');

  patientMrn: string = '';

  @ViewChild('medicationSchedule') medicationSchedule!: MedicationScheduleComponent;

  constructor(
    patientStore: PatientDashboardStore,
    authService: AuthService,
    public dialogService: DialogService,
    private toastService: ToastService
  ) {
    super(patientStore, authService);

    this.patient$.subscribe((patient: Patient) => {
      this.patientMrn = patient.mrn;
    });

    // Skip verifying patient if current user is admin
    this.isVerified = this.isAdmin;
  }

  ngOnInit(): void {
    this.stopLoading();
  }

  tabChanged(event: any) {
    if (event.index === 1) {
      this.medicationSchedule.refresh();
    }
  }

  verify() {
    this.isVerified = this.patientMrn == this.wristbandIdFormControl.value;
    if (!this.isVerified) {
      this.toastService.clear();
      this.toastService.addOne({
        severity: WARN,
        detail: 'Wristband ID does not match patient MRN!',
        life: 3000,
      });
    }
  }

  get wristbandIdFormControl(): UntypedFormControl {
    return this.wristbandId;
  }
}

<div *ngIf="!isVerified" class="verification-container">

  <button mat-raised-button color="primary" (click)="verify()">Verify Patient</button>
