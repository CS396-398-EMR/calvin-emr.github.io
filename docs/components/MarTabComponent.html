import { Component, Input, OnInit, ViewChild } from '@angular/core';
import {MatDialog} from '@angular/material/dialog';
import { DxSchedulerModule, DxSchedulerComponent, 
  DxTemplateModule, DxAutocompleteModule } from 'devextreme-angular';
import { Timestamp } from 'firebase/firestore';
import { AdministerMedDialogComponent } from 'src/app/shared/administer-med-dialog/administer-med-dialog.component';
import { TabComponent } from 'src/app/shared/tab.component';
import { Order } from 'src/app/type/order.type';
import { Update } from 'src/app/type/update.type';

import { UserService } from 'src/app/services/user.service';

@Component({
  selector: 'app-mar-tab',
  templateUrl: './mar-tab.component.html',
  styleUrls: ['./mar-tab.component.scss'],
})
export class MarTabComponent extends TabComponent implements OnInit {
  isVerified: boolean;
  wristbandId: string = "";
  error: string = "";

  showCurrentTimeIndicator : boolean = true;
  shadeUntilCurrentTime : boolean = true;

  aptsCopy: any[] = []; // array for passing appointment data into dx-scheduler

  // currentDate is s|n|D because there will be compilation errors
  // if it is only Date. No idea why; maybe TypeScript issue?
  @Input() currentDate : string|number|Date = new Date();
  @Input() apts: Order[] = [];
  @Input() patientName: string = ""
  @Input() patientId: string = "";

  @ViewChild(DxSchedulerComponent, {static: false}) scheduler!: DxSchedulerComponent;

  constructor(public dialog: MatDialog,
    private userSerivce: UserService){
    super();
    this.isVerified = this.userSerivce.getCurrentUserRole() == "admin";
  }

  ngOnInit(): void {
    this.copyApts();

    this.stopLoading();
  }

  /**
   * repaint
   * 
   * Repaints the dx-scheduler so that the appointments render correctly.
   */
  refresh() {
    if(this.isVerified) {
      this.copyApts();
      this.scheduler.instance.repaint();
    }
  }

  verify() {
    this.isVerified = this.patientId == this.wristbandId;
    if(!this.isVerified) {
      alert("Wristband ID does not match patient ID!")
    }
  }

  /**
   * copyApts
   * 
   * Copies all medication orders from the array passed in to the MAR component
   * to a new array. This new array is passed to the dx-scheduler component.
   */
  copyApts() {
    // Format the Timestamp objects in each Appointment to Date objects
    this.aptsCopy = [];
    this.apts.forEach((apt) => {
      this.aptsCopy.push({
        id: apt.id,
        medId: apt.medId,
        medName: apt.medName,
        route: apt.route,
        dosage: apt.dosage,
        quantity: apt.quantity,
        quantityGiven: apt.quantityGiven,
        order: apt.order,
        signature: apt.signature,
        startDate: new Date(apt.startDate.seconds * 1000),
        endDate: new Date(apt.endDate.seconds * 1000),
        administered: apt.administered,
        dateCreated: new Date(apt.dateCreated.seconds * 1000),
      })
    })
  }
  
  /**
   * renderAppointment
   * 
   * @param event object containing information about the appointment that is being rendered, including
   *  the Appointment object and the appointment styling.
   * 
   * Sets the background color of the current appointment, and formats the subject line to contain
   * the medication name and time to administer.
   */
  renderAppointment(event: any) {
    // Set background color of appointment
    event.appointmentElement.style.backgroundColor = this.getBackgroundColor(event.appointmentData);  
  }

  /**
   * overrideDefaultPopup
   * 
   * @param aptEvent object containing information about the appointment clicked on the dx-scheduler
   * 
   * Cancels the default dx-scheduler popup and instead opens the VerifyPatientDialog component,
   * if the appointment has not yet been administered.
   */
  overrideDefaultPopup(aptEvent: any) {
    let apt = aptEvent.appointmentData as Order;

    if(apt) {  
      // Cancel the default popup that dx-scheduler creates
      aptEvent.cancel = true;
      
      // TODO: check with nursing profs if clickable after max qty administered
      // if(!apt.administered) {
      // let med = new Order();
      // Object.assign(med, apt)
      // this.openVerifyDialog(aptEvent, this.patientName, this.patientId, med);
      // }

      // Open custom dialog regardless of whether med has/hasn't been administered
      let med = new Order();
      Object.assign(med, apt)
      this.openVerifyDialog(aptEvent, this.patientName, this.patientId, med);
    }
  }

  /**
   * 
   * @param aptEvent object containing information about the appointment clicked on the dx-scheduler
   * @param patientName Name of the current patient
   * @param patientId ID of the current patient
   * @param med order object that is to be administered
   */
  openVerifyDialog(aptEvent: any, patientName: string, patientId: string, med: Order): void {
    let aptData = aptEvent.appointmentData;
    const dialogRef = this.dialog.open(AdministerMedDialogComponent, {
      width: '50%',
      data: {patientName: patientName, patientId: patientId, med: med},
      disableClose: true,
    });

    dialogRef.afterClosed().subscribe((order: Order) => {
      // Set 'administered' to true/false for the specific appointment; since we can access the
      // appointment directly, we will just set administered to true.
      aptData.administered = order.isAdministered()
      let actualApt = this.apts.find(apt => apt.id == aptData.id);
      // If appointment has been administered, set the actual appointment to be administered
      if(actualApt) {
        if(order.isAdministered()) {
          actualApt.administered = true;
        }
        // Update local appointment data
        aptData.quantity = order.quantity;
        aptData.quantityGiven = order.quantityGiven;

        // Update appointment data to send to backend
        actualApt.quantity = order.quantity;
        actualApt.quantityGiven = order.quantityGiven;
      }

      // Re-render the appointment to change color
      this.renderAppointment(aptEvent);

      // Update appointments in Firebase
      this.updateMar();
    });
  }

  updateMar(): void {
    // Reformat dates in data to Timestamp; create a copy to work separately
    // from the dx-scheduler
    let updateApts = this.apts.map(x => Object.assign(new Order(), x));

    // Create new update object
    let update = new Update("providerOrders", updateApts);

    // Reformat update to JSON
    update = JSON.parse(JSON.stringify(update))

    // Convert dates in JSON to Timestamps
    update.data.forEach((apt: any) => {
      apt.dateCreated = Timestamp.fromMillis(apt.dateCreated.seconds * 1000)
      apt.startDate = Timestamp.fromMillis(apt.startDate.seconds * 1000);
      apt.endDate = Timestamp.fromMillis(apt.endDate.seconds * 1000);
    });

    this.updatePatientEvent.emit(update);
  }

  getTimeRange(start: Date, end: Date): string {
    let extraZeroStart = start.getMinutes() < 10 ? "0" : "";
    let extraZeroEnd = end.getMinutes() < 10 ? "0" : "";
    return start.getHours() + ":" + extraZeroStart + start.getMinutes() + " - " +
                      end.getHours() + ":" + extraZeroEnd + end.getMinutes()
  }

  /**
   * 
   * @param apt Order object
   * @returns hex code of "administered"/"not administered" color
   * 
   * Hello
   */
  getBackgroundColor(apt: any) {
    if(apt.administered) {
      return "#999999"
    }
    return "#3F51B5"
  }

  /**
   * @returns whether or not current user has admin role
   */
  get isAdmin(): boolean {
    return this.userSerivce.getCurrentUserRole() == "admin";
  }

}
