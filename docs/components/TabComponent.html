import { Component, EventEmitter, OnInit, Output } from '@angular/core';
import { ProgressSpinnerMode } from '@angular/material/progress-spinner';
import { Timestamp } from 'firebase/firestore';
import { hour24DateOptions, hour24TimeOptions } from '../global';
import { Update } from '../types/update.type';

@Component({
  selector: 'app-base',
  template: `
    <p>
      tab works!
    </p>
  `,
  styles: [
  ]
})
export class TabComponent implements OnInit {

  // For loading
  isLoading = true;
  mode: ProgressSpinnerMode = 'indeterminate';
  value = 50;

  // Event emitter to tell parent component to update patient.
  @Output() updatePatientEvent = new EventEmitter<Update>();

  constructor() { }

  ngOnInit(): void {
  }

  startLoading(): void {
    this.isLoading = true;
  }

  stopLoading(): void {
    this.isLoading = false;
  }

  /**
   * 
   * @param date The date to get the modified locale string for.
   * @returns A date as string, formatted as 'DD/MM/YYYY @ HH:MM' (24-hour time).
   */
  getDateString(date: Date): string {
    return date.toLocaleString([], hour24DateOptions).replace(',', ' @');
  }

  /**
   * 
   * @param date The date to get the modified locale string for.
   * @returns A time as string, formatted as 'HH:MM' (24-hour time).
   */
  getTimeString(date: Date): string {
    return date.toLocaleString([], hour24TimeOptions);
  }

  /**
   * toggleDelete
   * 
   * @param obj an object containing the property 'deleted' (which must be a boolean value)
   * 
   * Toggles the 'deleted' property between true and false.
   */
  toggleDelete(obj: any) {
    if(obj.hasOwnProperty("deleted")) {
      obj.deleted = !obj.deleted;
    }
    else{
      console.error("Object ", obj, " has no 'deleted' property!")
    }
  }

  /**
   * updatePatient
   * 
   * Emit the update to the parent component (the PatientDashboard component), 
   * indicating that we update the [field] field of current patient
   *  
   * Assumes that data is in an array object.
   * Assumes that each data element contains a date property.
   * 
   * Children of the Tab component should override this function if
   * objects do not have a "date" property
   */ 
  updatePatient(field: string, data: any[], fn?: () => void): void {
    // Create new update object
    let update = new Update(field, data);

    // Reformat update to JSON
    update = JSON.parse(JSON.stringify(update))

    // Reformat datestrings in data to Timestamp
    update.data.forEach((obj: any) => {
      obj.date = Timestamp.fromMillis(obj.date.seconds * 1000)
    });

    this.updatePatientEvent.emit(update);
  }

}
