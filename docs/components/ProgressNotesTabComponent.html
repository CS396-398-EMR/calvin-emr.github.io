import { Component, Input, OnInit } from '@angular/core';
import { PROGRESS_NOTES_FIELD } from 'src/app/global';
import { HtmlDialogComponent } from 'src/app/shared/html-dialog/html-dialog.component';
import { NoteEditDialogComponent } from 'src/app/patient-dashboard/progress-notes-tab/note-edit-dialog/note-edit-dialog.component';
import { PatientDashboardTabComponent } from 'src/app/patient-dashboard/dashboard-tab/dashboard-tab.component';
import { ProgressNote } from 'src/app/shared/models/progress-note.model';
import { AuthService } from 'src/app/shared/services/auth.service';
import { DialogService } from 'primeng/dynamicdialog';
import { PatientDashboardStore } from 'src/app/patient-dashboard/state/patient-dashboard.store';
import { Patient } from 'src/app/shared/models/patient.model';

@Component({
  selector: 'app-progress-notes-tab',
  templateUrl: './progress-notes-tab.component.html',
  styleUrls: ['./progress-notes-tab.component.scss'],
  providers: [DialogService],
})
export class ProgressNotesTabComponent extends PatientDashboardTabComponent implements OnInit {
  /**
   * Columns passed into the table used to display progress notes.
   */
  menuItems = [
    {
      icon: 'pi pi-pencil',
      command: () => {
        console.log('edit');
      },
    },
    {
      icon: 'pi pi-trash',
      command: () => {
        console.log('delete');
      },
    },
    {
      icon: 'pi pi-right-arrow',
      command: () => {
        console.log('view');
      },
    },
  ];

  progressNotes: ProgressNote[] = [];

  constructor(patientStore: PatientDashboardStore, authService: AuthService, public dialogService: DialogService) {
    super(patientStore, authService);
    this.patient$.subscribe((patient: Patient) => {
      this.progressNotes = patient.progressNotes;
    });
  }

  ngOnInit(): void {
    this.stopLoading();
  }

  viewNote(note: ProgressNote) {
    const dialogRef = this.dialogService.open(HtmlDialogComponent, {
      header: note.title,
      data: note,
      width: '50%',
    });
  }

  addNote(note: ProgressNote) {
    this.progressNotes.push(note);
    this.updatePatientField(PROGRESS_NOTES_FIELD, this.progressNotes);
  }

  editNote(note: ProgressNote) {
    const dialogRef = this.dialogService.open(NoteEditDialogComponent, {
      header: 'Edit Note',
      data: note,
    });

    dialogRef.onClose.subscribe((note: ProgressNote) => {
      if (note) {
        let indexToRemove = this.progressNotes.findIndex((n) => n.id == note.id);
        this.progressNotes.splice(indexToRemove, 1, note);
        this.updatePatientField(PROGRESS_NOTES_FIELD, this.progressNotes);
      }
    });
  }

  deleteNote(id: string) {
    this.progressNotes = this.progressNotes.filter((note) => {
      return note.id !== id;
    });
    this.updatePatientField(PROGRESS_NOTES_FIELD, this.progressNotes);
  }
}
