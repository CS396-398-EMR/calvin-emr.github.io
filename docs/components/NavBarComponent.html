import { Component, OnInit } from '@angular/core';
import { Router } from '@angular/router';
import { MenuItem } from 'primeng/api';
import { Observable } from 'rxjs';
import { PATH_HOME, PATH_LOGIN, PATH_PATIENTS, PATH_ABOUT_US, PATH_STUDENTS, PATH_MEDICATIONS } from '../global';
import { AuthService } from '../shared/services/auth.service';
import { PATH_SETTINGS, PATH_SETTINGS_GENERAL, PATH_SETTINGS_HELP } from '../settings/settings.global';

@Component({
  selector: 'app-nav-bar',
  templateUrl: './nav-bar.component.html',
  styleUrls: ['./nav-bar.component.scss'],
})
export class NavBarComponent implements OnInit {
  // Local constants
  homePath = PATH_HOME;
  loginPath = PATH_LOGIN;
  patientsPath = PATH_PATIENTS;
  studentsPath = PATH_STUDENTS;
  medicationsPath = PATH_MEDICATIONS;
  aboutUsPath = PATH_ABOUT_US;

  /**
   * Menu items to display under the "Profile" split button.
   */
  profileItems: MenuItem[] = [
    {
      label: 'Settings',
      icon: 'pi pi-cog',
      command: () => {
        this.navigateToSettingsGeneral();
      },
    },
    {
      label: 'Help',
      icon: 'pi pi-question-circle',
      command: () => {
        this.navigateToSettingsHelp();
      },
    },
    { separator: true },
    {
      label: 'Log Out',
      icon: 'pi pi-sign-out',
      command: () => {
        this.logout();
      },
    },
  ];

  /**
   * Local reference to the loggedIn Observable in order to determine the current
   * auth state.
   */
  loggedIn: Observable<boolean>;

  constructor(private router: Router, private authService: AuthService) {
    this.loggedIn = authService.loggedIn$;
  }

  ngOnInit(): void {}

  /**
   * Navigates to '/settings' path.
   */
  navigateToSettings() {
    this.router.navigateByUrl(PATH_SETTINGS);
  }

  /**
   * Navigates to '/settings/general' path.
   */
  navigateToSettingsGeneral() {
    this.router.navigateByUrl(`${PATH_SETTINGS}/${PATH_SETTINGS_GENERAL}`);
  }

  /**
   * Navigates to '/settings/help' path.
   */
  navigateToSettingsHelp() {
    this.router.navigateByUrl(`${PATH_SETTINGS}/${PATH_SETTINGS_HELP}`);
  }

  /**
   * Calls the AuthService to log the user out.
   */
  logout() {
    this.authService.logout();
  }

  get isAdmin() {
    return this.authService.isCurrentUserAdmin;
  }
}
