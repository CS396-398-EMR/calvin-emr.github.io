import { Component, OnInit } from '@angular/core';
import { UntypedFormControl } from '@angular/forms';
import { AssessmentForm } from 'src/app/shared/assessment-form.component';
import {
  extremityStrengthOptions,
  locOptions,
  orientationOptions,
  pupillaryDilationOptions,
  pupillaryResponseOptions,
  sensationOptions,
  tremorsOptions,
} from './neurological-form.metadata';

@Component({
  selector: 'app-neurological-form',
  templateUrl: './neurological-form.component.html',
  styleUrls: ['./neurological-form.component.scss', '../assessments-tab.component.scss'],
})
export class NeurologicalFormComponent extends AssessmentForm implements OnInit {
  constructor() {
    super();
    this.reset();
  }

  ngOnInit(): void {}

  reset() {
    this.controlList = [
      { control: 'loc', default: 'WNL', wnl: false },
      { control: 'handGrips', default: 'WNL', wnl: false },
      { control: 'lePushPull', default: 'WNL', wnl: false },
      { control: 'pupillaryResponse', default: 'WNL', wnl: false },
      { control: 'leftPupillaryDilation', default: 'WNL', wnl: false },
      { control: 'rightPupillaryDilation', default: 'WNL', wnl: false },
      { control: 'rue', default: 'Full sensation', wnl: false },
      { control: 'rle', default: 'Full sensation', wnl: false },
      { control: 'lle', default: 'Full sensation', wnl: false },
      { control: 'lue', default: 'Full sensation', wnl: false },
      { control: 'tremors', default: [], wnl: false },
    ];
  }

  locIndex = 0;
  handGripsIndex = 1;
  lePushPullIndex = 2;
  pupillaryResponseIndex = 3;

  locOptions = locOptions;
  orientationOptions = orientationOptions;

  // Options for handGrips and lePushPull
  extremityStrengthOptions = extremityStrengthOptions;
  pupillaryResponseOptions = pupillaryResponseOptions;

  get locDefault() {
    return this.controlList[this.locIndex].default;
  }

  get handGripsDefault() {
    return this.controlList[this.handGripsIndex].default;
  }

  get lePushPullDefault() {
    return this.controlList[this.lePushPullIndex].default;
  }

  get pupillaryResponseDefault() {
    return this.controlList[this.pupillaryResponseIndex].default;
  }

  get locFormControl(): UntypedFormControl {
    return this.formGroup.controls.loc as UntypedFormControl;
  }

  get orientationFormControl(): UntypedFormControl {
    return this.formGroup.controls.orientation as UntypedFormControl;
  }

  get extremityMovementFormControl(): UntypedFormControl {
    return this.formGroup.controls.extremityMovement as UntypedFormControl;
  }

  get handGripsFormControl(): UntypedFormControl {
    return this.formGroup.controls.handGrips as UntypedFormControl;
  }

  get lePushPullFormControl(): UntypedFormControl {
    return this.formGroup.controls.lePushPull as UntypedFormControl;
  }

  get pupillaryResponseFormControl(): UntypedFormControl {
    return this.formGroup.controls.pupillaryResponse as UntypedFormControl;
  }

  /**
   * Pupillary Dilation
   */

  leftPupillaryDilationIndex = 4;
  rightPupillaryDilationIndex = 5;

  pupillaryDilationOptions = pupillaryDilationOptions;

  get pupillaryDilationDefault() {
    return this.controlList[this.leftPupillaryDilationIndex].default;
  }

  get leftPupillaryDilationFormControl(): UntypedFormControl {
    return this.formGroup.get('pupillaryDilation.left') as UntypedFormControl;
  }

  get rightPupillaryDilationFormControl(): UntypedFormControl {
    return this.formGroup.get('pupillaryDilation.right') as UntypedFormControl;
  }

  /**
   * Glasgow Coma Scale Score
   */

  get gcssFormControl(): UntypedFormControl {
    return this.formGroup.controls.gcss as UntypedFormControl;
  }

  /**
   * Sensation
   */

  rueIndex = 6;
  rleIndex = 7;
  lleIndex = 8;
  lueIndex = 9;

  sensationOptions = sensationOptions;

  get sensationDefault() {
    return this.controlList[this.rueIndex].default;
  }

  get rueFormControl(): UntypedFormControl {
    return this.formGroup.get('sensation.rue') as UntypedFormControl;
  }

  get rleFormControl(): UntypedFormControl {
    return this.formGroup.get('sensation.rle') as UntypedFormControl;
  }

  get lueFormControl(): UntypedFormControl {
    return this.formGroup.get('sensation.lue') as UntypedFormControl;
  }

  get lleFormControl(): UntypedFormControl {
    return this.formGroup.get('sensation.lle') as UntypedFormControl;
  }

  /**
   * Tremors
   */

  tremorsIndex = 10;

  tremorsOptions = tremorsOptions;

  get tremorsFormControl(): UntypedFormControl {
    return this.formGroup.controls.tremors as UntypedFormControl;
  }
}
