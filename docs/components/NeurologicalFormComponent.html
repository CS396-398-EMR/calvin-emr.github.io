import { Component, OnInit } from '@angular/core';
import { FormControl, FormGroup } from '@angular/forms';
import { AssessmentForm } from 'src/app/shared/assessment-form.component';
import { FormOption } from 'src/app/type/form-option.type';

@Component({
  selector: 'app-neurological-form',
  templateUrl: './neurological-form.component.html',
  styleUrls: ['./neurological-form.component.scss', '../assessments-tab.component.scss']
})
export class NeurologicalFormComponent extends AssessmentForm implements OnInit {

  constructor() { 
    super(); 
    this.controlList = [
      {control: "loc", default: "WNL" , wnl: false},
      {control: "handGrips", default: "WNL" , wnl: false},
      {control: "lePushPull", default: "WNL" , wnl: false},
      {control: "pupillaryResponse", default: "WNL" , wnl: false},
      {control: "leftPupillaryDilation", default: "WNL" , wnl: false},
      {control: "rightPupillaryDilation", default: "WNL" , wnl: false},
      {control: "rue", default: "Full sensation" , wnl: false},
      {control: "rle", default: "Full sensation" , wnl: false},
      {control: "lle", default: "Full sensation" , wnl: false},
      {control: "lue", default: "Full sensation" , wnl: false},
      {control: "tremors", default: "None" , wnl: false},
    ];
  }

  ngOnInit(): void {
  }

  locIndex = 0;
  handGripsIndex = 1;
  lePushPullIndex = 2;
  pupillaryResponseIndex = 3;

  locOptions = [
    new FormOption("Lethargic"),
    new FormOption("Somnolent"),
    new FormOption("WNL"),
  ]

  orientationOptions = [
    new FormOption("NOT TO person", "person"),
    new FormOption("NOT TO place", "place"),
    new FormOption("NOT TO time", "time"),
    new FormOption("NOT TO situation", "situation"),
  ]

  // Options for handGrips and lePushPull
  extremityStrengthOptions = [
    new FormOption("Moderate"),
    new FormOption("Weak"),
    new FormOption("WNL"),
  ]

  pupillaryResponseOptions = [
    new FormOption("Sluggish"),
    new FormOption("Absent"),
    new FormOption("WNL"),
  ]

  get locDefault() {
    return this.controlList[this.locIndex].default;
  }

  get handGripsDefault() {
    return this.controlList[this.handGripsIndex].default;
  }

  get lePushPullDefault() {
    return this.controlList[this.lePushPullIndex].default;
  }

  get pupillaryResponseDefault() {
    return this.controlList[this.pupillaryResponseIndex].default;
  }

  get locFormControl(): FormControl {
    return this.formGroup.controls.loc as FormControl;
  }

  get orientationFormGroup(): FormGroup {
    return this.formGroup.controls.orientation as FormGroup;
  }

  get extremityMovementFormControl(): FormControl {
    return this.formGroup.controls.extremityMovement as FormControl;
  }

  get handGripsFormControl(): FormControl {
    return this.formGroup.controls.handGrips as FormControl;
  }

  get lePushPullFormControl(): FormControl {
    return this.formGroup.controls.lePushPull as FormControl;
  }

  get pupillaryResponseFormControl(): FormControl {
    return this.formGroup.controls.pupillaryResponse as FormControl;
  }

  /**
   * Pupillary Dilation
   */

  leftPupillaryDilationIndex = 4;
  rightPupillaryDilationIndex = 5;

  pupillaryDilationOptions = [
    new FormOption("2mm"),
    new FormOption("4mm"),
    new FormOption("6mm"),
    new FormOption("8mm"),
  ]

  get pupillaryDilationDefault() {
    return this.controlList[this.leftPupillaryDilationIndex].default;
  }

  get leftPupillaryDilationFormControl(): FormControl {
    return this.formGroup.get("pupillaryDilation.left") as FormControl;
  }

  get rightPupillaryDilationFormControl(): FormControl {
    return this.formGroup.get("pupillaryDilation.right") as FormControl;
  }


  /**
   * Glasgow Coma Scale Score
   */

   get gcssFormControl(): FormControl {
    return this.formGroup.controls.gcss as FormControl;
  }


  /**
   * Sensation
   */

  rueIndex = 6;
  rleIndex = 7;
  lleIndex = 8;
  lueIndex = 9;

  sensationOptions = [
    new FormOption("Numbness"),
    new FormOption("Tingling"),
    new FormOption("Full sensation")
  ]

  get sensationDefault() {
    return this.controlList[this.rueIndex].default;
  }

  get rueFormControl(): FormControl {
    return this.formGroup.get("sensation.rue") as FormControl;
  }

  get rleFormControl(): FormControl {
    return this.formGroup.get("sensation.rle") as FormControl;
  }

  get lueFormControl(): FormControl {
    return this.formGroup.get("sensation.lue") as FormControl;
  }

  get lleFormControl(): FormControl {
    return this.formGroup.get("sensation.lle") as FormControl;
  }


  /**
   * Tremors
   */

  tremorsIndex = 10;

  tremorsOptions = [
    new FormOption("RUE", "rue"),
    new FormOption("RLE", "rle"),
    new FormOption("LLE", "lle"),
    new FormOption("LUE", "lue"),
  ]

  get tremorsFormGroup(): FormGroup {
    return this.formGroup.controls.tremors as FormGroup;
  }
 
}
