import { Component, OnInit } from '@angular/core';
import { FormControl, Validators } from '@angular/forms';
import { AuthService } from 'src/app/services/auth.service';

@Component({
  selector: 'app-login',
  templateUrl: './login.component.html',
  styleUrls: ['./login.component.scss']
})
export class LoginComponent implements OnInit {

  email = new FormControl('', [Validators.required, Validators.email]);
  password = new FormControl('', [Validators.required]);
  hide = true;

  invalidLogin = false;
  
  constructor(private auth: AuthService) { }

  ngOnInit(): void { 
    this.auth.getUser()
  }

  /**
   *  returns appropriate error message for the e-mail form
   * @returns {string} error message for the wrong e-mail credential
   */
  getEmailErrorMessage() {
    if(this.email.hasError('required')) {
      return 'You must enter an email.';
    }
    return this.email.hasError('email') ? 'Not a valid email.' : '';
  }

  /**
   *  returns appropriate error message related to the password form
   * @returns {string} error message for the wrong password credential
   */
  getPasswordErrorMessage() {
    if (this.password.hasError('required')) {
      return 'You must enter a password';
    }
    return "Invalid password."
  }

  /**
   * 
   * @param event Object passed by (keyup) property of input tag
   * 
   * Calls the login function when the "Enter/Return" key is pressed.
   */
  handleKeyUp(event: any) {
    // keyCode == 13 stands for the return key
    if(event.keyCode == 13) {
      this.login();
    }
  }

  // TODO: combine login and guestLogin function to reduce number of methods
  //        it can be done with adding optional parameter for login function
  //        where it will accept string-literal value for guest id and password

  /**
   *  login as a guest account (Currently used for the debugging purpose)
   */
  guestLogin() {
    // Use appropriate e-mail and password credential for the guest login
    this.auth.login("admin@gmail.com", "111111")
    .catch((error: any) => {
      console.error(error)
      this.invalidLogin = true;
    })
  }

  /**
   *  login using email and password value
   */
  login() {
    this.auth.login(this.email.value, this.password.value)
    .catch((error: any) => {
      console.error(error)
      this.invalidLogin = true;
    })
  }
}
