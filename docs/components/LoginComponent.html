import { Component, OnInit } from '@angular/core';
import { FormControl, Validators } from '@angular/forms';
import { Router } from '@angular/router';
import { Subscription } from 'rxjs';
import { forgotPasswordPath, invalidEmailError, noEmailError, noPasswordError, invalidPasswordError, returnKeyCode, patientsPath } from 'src/app/global';
import { AuthService } from 'src/app/services/auth.service';

@Component({
  selector: 'app-login',
  templateUrl: './login.component.html',
  styleUrls: ['./login.component.scss']
})
export class LoginComponent implements OnInit {

  /**
   * Local constant to specify the path to the ForgotPassword page.
   */
  forgotPasswordPath = '/' + forgotPasswordPath;

  /**
   * Local constant to sp
   */
  passwordLabel: string = 'Enter your password';

  /**
   * FormControl for email input.
   */
  email = new FormControl('', [Validators.required, Validators.email]);

  /**
   * FormControl for password input.
   */
  password = new FormControl('', [Validators.required]);

  /**
   * Keeps track of the subscription to email FormControl valueChanges property.
   */
  emailSub: Subscription;

  /**
   * Keeps track of the subscription to password FormControl valueChanges property.
   */
  passwordSub: Subscription;

  /**
   * Flag to indicate whether to display error message on screen.
   */
  invalidLogin = false;
  
  /**
   * Subscribe to changes on each FormControl to clear error messages
   * when typing after a failed login attempt.
   */
  constructor(private auth: AuthService, private router: Router) { 
    this.emailSub = this.email.valueChanges.subscribe(() => {
      this.invalidLogin = false;
    })
    this.passwordSub = this.password.valueChanges.subscribe(() => {
      this.invalidLogin = false;
    })
  }

  ngOnInit(): void { }

  /**
   * Unsubscribes from both FormControl valueChanges.
   */
  ngOnDestroy(): void {
    this.emailSub.unsubscribe();
    this.passwordSub.unsubscribe();
  }

  /**
   * Returns appropriate error message for the email form.
   * 
   * @returns Error message for the wrong email credential.
   */
  getEmailErrorMessage(): string {
    if(this.email.hasError('required')) {
      return noEmailError;
    }
    return this.email.hasError('email') ? invalidEmailError : '';
  }

  /**
   * Returns appropriate error message related to the password form.
   * 
   * @returns Error message for the wrong password credential.
   */
  getPasswordErrorMessage(): string {
    if (this.password.hasError('required')) {
      return noPasswordError;
    }
    return invalidPasswordError;
  }

  /**
   * Calls the login function when the "Enter/Return" key is pressed.
   * 
   * @param event Object passed by (keyup) property of input tag.
   */
  handleKeyUp(event: any) {
    if(event.keyCode == returnKeyCode) {
      this.login();
    }
  }

  /**
   * Attempt to log in using the provided email and password.
   * If the attempt succeeds, navigate to the patient-list page.
   * If the attempt fails, set the invalidLogin flag to true to show
   * an error message.
   */
  login() {
    this.auth.login(this.email.value, this.password.value)
    .then(() => {
      this.router.navigateByUrl(patientsPath);
    })
    .catch(() => {
      this.invalidLogin = true;
    })
  }
}
