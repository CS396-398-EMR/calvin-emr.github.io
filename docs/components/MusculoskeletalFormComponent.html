import { Component, OnInit } from '@angular/core';
import { UntypedFormControl, UntypedFormGroup } from '@angular/forms';
import { AssessmentForm } from 'src/app/shared/assessment-form.component';
import {
  ageOptions,
  cognitionOptions,
  COGNITION_FALL_RISK_AWARENESS,
  COGNITION_FALL_RISK_IMPULSIVE,
  COGNITION_FALL_RISK_UNDERSTANDING,
  deviceOptions,
  ebuOptions,
  fallHistoryOption,
  medicationsOptions,
  mobilityAndTransfersOptions,
  mobilityFallRiskOptions,
  MOBILITY_FALL_RISK_ASSISTANCE,
  MOBILITY_FALL_RISK_GAIT,
  MOBILITY_FALL_RISK_IMPAIRMENT,
  pceOptions,
} from './musculoskeletal-form.metadata';

@Component({
  selector: 'app-musculoskeletal-form',
  templateUrl: './musculoskeletal-form.component.html',
  styleUrls: ['./musculoskeletal-form.component.scss', '../assessments-tab.component.scss'],
})
export class MusculoskeletalFormComponent extends AssessmentForm implements OnInit {
  constructor() {
    super();
    this.reset();
  }

  ngOnInit(): void {
    // Add listener to changes in the fall risk form, which updates
    //  fall score each time a change occurs
    this.fallRiskFormGroup.valueChanges.subscribe(() => {
      this.fallScoreFormControl.setValue(this.fallRiskScore);
    });
  }

  reset() {
    this.controlList = [
      { control: 'mobilityAndTransfers', default: 'No assists', wnl: false },
      { control: 'assistiveDevices', default: [], wnl: false },
    ];
  }

  /**
   * Other
   */

  mobilityAndTransfersIndex = 0;
  assistiveDevicesIndex = 1;

  mobilityAndTransfersOptions = mobilityAndTransfersOptions;
  deviceOptions = deviceOptions;

  get mobilityAndTransfersDefault() {
    return this.controlList[this.mobilityAndTransfersIndex].default;
  }

  get assistiveDevicesDefault() {
    return this.controlList[this.assistiveDevicesIndex].default;
  }

  get mobilityAndTransfersFormControl(): UntypedFormControl {
    return this.formGroup.controls.mobilityAndTransfers as UntypedFormControl;
  }

  get assistiveDevicesFormControl(): UntypedFormControl {
    return this.formGroup.controls.assistiveDevices as UntypedFormControl;
  }

  /**
   * Fall Risk
   */

  ageOptions = ageOptions;
  fallHistoryOption = fallHistoryOption;
  ebuOptions = ebuOptions;
  medicationsOptions = medicationsOptions;
  pceOptions = pceOptions;
  mobilityFallRiskOptions = mobilityFallRiskOptions;
  cognitionOptions = cognitionOptions;

  get fallRiskDefault() {
    return this.controlList[0].default;
  }

  get fallRiskFormGroup(): UntypedFormGroup {
    return this.formGroup.controls.fallRisk as UntypedFormGroup;
  }

  get ageFormControl(): UntypedFormControl {
    return this.formGroup.get('fallRisk.age') as UntypedFormControl;
  }

  get fallHistoryFormControl(): UntypedFormControl {
    return this.formGroup.get('fallRisk.fallHistory') as UntypedFormControl;
  }

  get ebuFormControl(): UntypedFormControl {
    return this.formGroup.get('fallRisk.ebu') as UntypedFormControl;
  }

  get medicationsFormControl(): UntypedFormControl {
    return this.formGroup.get('fallRisk.medications') as UntypedFormControl;
  }

  get pceFormControl(): UntypedFormControl {
    return this.formGroup.get('fallRisk.pce') as UntypedFormControl;
  }

  get mobilityFallRiskFormControl(): UntypedFormControl {
    return this.formGroup.get('fallRisk.mobility') as UntypedFormControl;
  }

  get cognitionFormControl(): UntypedFormControl {
    return this.formGroup.get('fallRisk.cognition') as UntypedFormControl;
  }

  get fallScoreFormControl(): UntypedFormControl {
    return this.formGroup.controls.fallScore as UntypedFormControl;
  }

  get ageScore() {
    switch (this.ageFormControl.value) {
      case '60 - 69 years (1 point)':
        return 1;
      case '70 - 79 years (2 points)':
        return 2;
      case 'greater than or equal to 80 years (3 points)':
        return 3;
      default:
        return 0;
    }
  }

  get fallHistoryScore() {
    return this.fallHistoryFormControl.value ? 5 : 0;
  }

  get ebuScore() {
    switch (this.ebuFormControl.value) {
      case 'Incontinence (2 points)':
        return 2;
      case 'Urgency or frequency (2 points)':
        return 2;
      case 'Urgency/frequency and incontinence (4 points)':
        return 4;
      default:
        return 0;
    }
  }

  get medicationsScore() {
    switch (this.medicationsFormControl.value) {
      case 'On 1 high fall risk drug (3 points)':
        return 3;
      case 'On 2 or more high fall risk drugs (5 points)':
        return 5;
      case 'Sedated procedure within past 24 hours (7 points)':
        return 7;
      default:
        return 0;
    }
  }

  get pceScore() {
    switch (this.pceFormControl.value) {
      case 'One present (1 point)':
        return 1;
      case 'Two present (2 points)':
        return 2;
      case '3 or more present (3 points)':
        return 3;
      default:
        return 0;
    }
  }

  get mobilityFallRiskScore() {
    let score = 0;
    let mobilityFallRiskFactors: string[] = this.mobilityFallRiskFormControl.value;

    mobilityFallRiskFactors.forEach((factor) => {
      if (
        factor === MOBILITY_FALL_RISK_ASSISTANCE ||
        factor === MOBILITY_FALL_RISK_GAIT ||
        factor === MOBILITY_FALL_RISK_IMPAIRMENT
      ) {
        score += 2;
      }
    });

    return score;
  }

  get cognitionScore() {
    let score = 0;
    let cognitionFallRiskFactors: string[] = this.cognitionFormControl.value;

    cognitionFallRiskFactors.forEach((factor) => {
      if (factor === COGNITION_FALL_RISK_AWARENESS) {
        score += 1;
      } else if (factor === COGNITION_FALL_RISK_IMPULSIVE) {
        score += 2;
      } else if (factor === COGNITION_FALL_RISK_UNDERSTANDING) {
        score += 4;
      }
    });

    return score;
  }

  get fallRiskScore() {
    return (
      this.ageScore +
      this.fallHistoryScore +
      this.ebuScore +
      this.medicationsScore +
      this.pceScore +
      this.mobilityFallRiskScore +
      this.cognitionScore
    );
  }

  get fallRiskCategory(): string {
    if (this.fallRiskScore < 6) {
      return 'Low';
    } else if (this.fallRiskScore >= 6 && this.fallRiskScore <= 13) {
      return 'Medium';
    } else {
      return 'High';
    }
  }
}
