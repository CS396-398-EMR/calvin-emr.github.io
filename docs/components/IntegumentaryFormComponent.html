import { Component, OnInit } from '@angular/core';
import { FormControl, FormGroup } from '@angular/forms';
import { AssessmentForm } from 'src/app/shared/assessment-form.component';
import { FormOption } from 'src/app/type/form-option.type';

@Component({
  selector: 'app-integumentary-form',
  templateUrl: './integumentary-form.component.html',
  styleUrls: ['./integumentary-form.component.scss', '../assessments-tab.component.scss']
})
export class IntegumentaryFormComponent extends AssessmentForm implements OnInit {

  constructor() {
    super();
    this.controlList = [
      {control: "skinAppearance", default: false,  wnl: false},
    ];
  }

  ngOnInit(): void {
  }

  skinAppearanceOptions = [
    new FormOption("Diaphoresis", "diaphoresis"),
    new FormOption("Pallor", "pallor"),
    new FormOption("Poor turgor", "poorTurgor"),
    new FormOption("Cool", "cool"),
    new FormOption("Modeling", "modeling"),
    new FormOption("Jaundice", "jaundice"),
  ];

  stagingOptions = [
    new FormOption("Stage One", "stageOne"),
    new FormOption("Stage Two", "stageTwo"),
    new FormOption("Stage Three", "stageThree"),
    new FormOption("Stage Four", "stageFour"),
    new FormOption("Unstageable", "unstageable"),
  ];

  careOptions = [
    new FormOption("ABD", "abd"),
    new FormOption("Gauze", "gauze"),
    new FormOption("Wet to dry dressing change", "dressingChange"),
    new FormOption("Pharmaceutical agent (See MAR)", "pharmAgent"),
    new FormOption("Steri strips", "steriStrips"),
    new FormOption("Staples", "staples"),
    new FormOption("Well Approximated", "wellApproximated"),
    new FormOption("Dehiscence", "dehiscence"),
  ];

  get skinAppearanceDefault() {
    return this.controlList[0].default;
  }

  get skinAppearanceFormGroup(): FormGroup {
    return this.formGroup.controls.skinAppearance as FormGroup;
  }

  get locationFormControl(): FormControl {
    return this.formGroup.get('wounds.location') as FormControl;
  }

  get stagingFormControl(): FormControl {
    return this.formGroup.get('wounds.staging') as FormControl;
  }

  get careFormGroup(): FormGroup {
    return this.formGroup.get('wounds.care') as FormGroup;
  }

}
