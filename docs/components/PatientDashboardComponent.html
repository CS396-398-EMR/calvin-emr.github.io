import { Component, OnInit, ViewChild } from '@angular/core';
import { Patient } from '../../shared/models/patient.model';
import { PatientService } from '../../shared/services/patient.service';
import { PageComponent } from 'src/app/shared/page.component';
import { ActivatedRoute, Router } from '@angular/router';
import { getStorage, ref, getDownloadURL } from 'firebase/storage';
import * as QRCode from 'qrcode';
import { saveAs } from 'file-saver';
import { MarTabComponent } from 'src/app/patient-dashboard/mar/mar-tab/mar-tab.component';
import { ERROR_PROFILE_PIC_LOADING, QR_CODE_SUFFIX } from 'src/app/global';
import { Subscription } from 'rxjs';
import { MenuItem } from 'primeng/api';
import { ToastService } from 'src/app/shared/services/toast.service';
import { AuthService } from 'src/app/shared/services/auth.service';
import { ProviderOrder } from 'src/app/shared/models/provider-order.model';
import { PatientDashboardStore } from '../state/patient-dashboard.store';
import { map, skip, takeUntil } from 'rxjs/operators';

@Component({
  selector: 'app-patient-dashboard',
  templateUrl: './patient-dashboard.component.html',
  styleUrls: ['./patient-dashboard.component.scss'],
  providers: [PatientDashboardStore],
})
export class PatientDashboardComponent extends PageComponent implements OnInit {
  // Stores current patient information
  currentPatient = new Patient();
  patientId = '';
  collection = '';
  profilePicSrc = '';
  providerOrders: ProviderOrder[] = [];

  patientSub?: Subscription;

  menuItems: MenuItem[] = [
    { label: 'Edit Patient Info', icon: 'pi pi-pencil', command: () => this.navigateToEditPatientPage() },
    { label: 'Download QR Code', icon: 'pi pi-qrcode', command: () => this.downloadQR() },
  ];

  tabsUrls = ['patient-charting', 'vital-signs', 'provider-charts', 'mar'];

  // Needed to repaint the MAR scheduler so that it loads correctly on the first open
  // and reloads each time thereafter
  @ViewChild(MarTabComponent, { static: false }) mar!: MarTabComponent;

  constructor(
    authService: AuthService,
    private router: Router,
    private route: ActivatedRoute,
    private readonly patientStore: PatientDashboardStore,
    private patientService: PatientService,
    private toastService: ToastService
  ) {
    super(authService);
  }

  ngOnInit(): void {
    super.ngOnInit();

    this.patientSub = this.patientStore.patient$
      .pipe(
        takeUntil(this.patientStore.destroy$),
        // Skip the first patient emitted, since it is empty
        skip(1),
        map((patient, index) => [patient, index])
      )
      .subscribe((res) => {
        const patient = res[0] as Patient;
        const index = res[1] as number;

        this.currentPatient = patient;

        // Only load the profile picture on the first non-empty patient emitted
        if (index === 0) {
          this.getProfilePicture();
        }
      });

    // Get id from the router URL, set from PatientList component
    let pId = this.route.snapshot.paramMap.get('id');
    let col = this.route.snapshot.paramMap.get('collection');

    // TODO: change default patient or make "patient not loaded" page
    if (!pId || !col) {
      this.router.navigate(['/page-not-found']);
    } else {
      this.patientId = pId;
      this.collection = col;
      this.patientStore.getPatient({ patientId: pId, collection: col });
    }
  }

  /**
   * Unsubscribes from the patientSub subscription.
   */
  ngOnDestroy(): void {
    super.ngOnDestroy();
  }

  getProfilePicture() {
    // Get profile picture
    const storage = getStorage();
    getDownloadURL(ref(storage, this.currentPatient.getProfilePicURL()))
      .then((url) => {
        this.profilePicSrc = url;
        this.stopLoading();
      })
      .catch((error) => {
        console.error(ERROR_PROFILE_PIC_LOADING, error);
      });
  }

  /**
   * tabClick
   *
   * @param event event containing the index of the tab clicked
   *
   * Runs whenever the user clicks the MAR tab; repaints the scheduler
   * in the MAR so that it loads normally. Without this, appointments do
   * not load on first opening the MAR.
   */
  onTabChange(event: any) {
    //   if (event.index === 3) {
    //     this.mar.refresh();
    //   }
  }

  /**
   * Navigates user to the Edit Patient page.
   */
  navigateToEditPatientPage() {
    this.router.navigate(['edit-patient', this.currentPatient.id]);
  }

  /**
   *  Downloads the current patient's MRN as a QR code image.
   */
  downloadQR() {
    QRCode.toDataURL(this.currentPatient.getId(), (err, url) => {
      saveAs(url, this.currentPatient.getId() + QR_CODE_SUFFIX);
    });
  }
}
