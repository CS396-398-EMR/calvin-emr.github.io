import { Component, OnInit } from '@angular/core';
import { UntypedFormControl, Validators } from '@angular/forms';
import { ActionCodeSettings, getAuth, sendPasswordResetEmail } from "firebase/auth";

@Component({
  selector: 'app-forgot-password',
  templateUrl: './forgot-password.component.html',
  styleUrls: ['./forgot-password.component.scss']
})
export class ForgotPasswordComponent implements OnInit {

  email = new UntypedFormControl('', [Validators.required, Validators.email]);

  // Boolean for error messages
  invalidEmail = false;

  // Boolean to control what card is displayed
  emailSent = false;
  noEmailExists = false;

  constructor() { }

  ngOnInit(): void {
  }

  onSubmit() {
    // Check for errors
    if(!this.email.valid) {
      this.invalidEmail = true;
      this.noEmailExists = false;
    }
    else {
      this.sendEmail();
    }
  }

  /**
   * Send password reset email and redirect user to the login page after the reset
   */
  sendEmail() {
    const auth = getAuth();

    // Create ActionCodeSettings object to redirect back to login page after password reset
    let actionCodeSettings = {
      url: "https://calvin-emr.web.app"
    }

    sendPasswordResetEmail(auth, this.email.value, actionCodeSettings)
      .then(() => {
        this.emailSent = true;
      })
      .catch((error) => {
        console.log(error)
        this.invalidEmail = false;
        this.noEmailExists = true;
      });
  }
}
